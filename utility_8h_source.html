192 template <typename T>
193 constexpr bool is_pow2(T number) {
194 static_assert(std::is_integral<T>::value, "is_pow2 must be used with integer types");
195
196 return (number & (number - 1)) == 0;
197 }
198
199 template <typename T>
200 constexpr T closest_pow2(T number) {
201 static_assert(std::is_integral<T>::value, "closest_pow2 must be used with integer types");
202
203 if (is_pow2(number))
204 return number;
205
206 // Collapse all bits below the highest set bit
207 number |= number >> 1;
208 number |= number >> 2;
209 number |= number >> 4;
210 if constexpr (sizeof(T) > 1)
211 number |= number >> 8;
212 if constexpr (sizeof(T) > 2)
213 number |= number >> 16;
214 if constexpr (sizeof(T) > 4)
215 number |= number >> 32;
216
217 // The result is now one less than the next power of two, so shift back
218 return number - (number >> 1);
219 }

563 void each_tuple_impl(Func func, std::integer_sequence<decltype(FirstIdx), Is...> /*no_name*/) {
564 if constexpr ((std::is_invocable_v<
565 Func&&, decltype(std::tuple_element_t<FirstIdx + Is, Tuple>{}),
566 std::integral_constant<decltype(FirstIdx), Is>> &&
567 ...))
568 // func(Args&& arg, uint32_t idx)

570 std::tuple_element_t<FirstIdx + Is, Tuple>{},
571 std::integral_constant<decltype(FirstIdx), FirstIdx + Is>{}),
572 ...);

574 // func(Args&& arg)
575 (func(std::tuple_element_t<FirstIdx + Is, Tuple>{}), ...);

779 template <typename InputIt, typename T>
780 constexpr InputIt find(InputIt first, InputIt last, const T& value) {
781 if constexpr (std::is_pointer_v<InputIt>) {
782 auto size = distance(first, last);
783 for (decltype(size) i = 0; i < size; ++i) {
784 if (first[i] == value)
785 return &first[i];
786 }
787 } else if constexpr (is_random_iter_v<InputIt>) {
788 auto size = distance(first, last);
789 for (decltype(size) i = 0; i < size; ++i) {
790 if (*(first[i]) == value)
791 return first[i];
792 }
793 } else {
794 for (; first != last; ++first) {
795 if (*first == value)
796 return first;
797 }
798 }
799 return last;
800 }

810 template <typename InputIt, typename Func>
811 constexpr InputIt find_if(InputIt first, InputIt last, Func func) {
812 if constexpr (std::is_pointer_v<InputIt>) {
813 auto size = distance(first, last);
814 for (decltype(size) i = 0; i < size; ++i) {
815 if (func(first[i]))
816 return &first[i];
817 }
818 } else if constexpr (std::is_same_v<typename InputIt::iterator_category, core::random_access_iterator_tag>) {
819 auto size = distance(first, last);
820 for (decltype(size) i = 0; i < size; ++i) {
821 if (func(*(first[i])))
822 return first[i];
823 }
824 } else {
825 for (; first != last; ++first) {
826 if (func(*first))
827 return first;
828 }
829 }
830 return last;

843 if constexpr (std::is_pointer_v<InputIt>) {
844 auto size = distance(first, last);
845 for (decltype(size) i = 0; i < size; ++i) {
846 if (!func(first[i]))
847 return &first[i];
848 }
849 } else if constexpr (std::is_same_v<typename InputIt::iterator_category, core::random_access_iterator_tag>) {
850 auto size = distance(first, last);
851 for (decltype(size) i = 0; i < size; ++i) {
852 if (!func(*(first[i])))
853 return first[i];
854 }
855 } else {
856 for (; first != last; ++first) {
857 if (!func(*first))
858 return first;
859 }

1444 swapFunc(pivotIdx, r.high);

1452 swapFunc(++i, r.high);

1457 stack[sp++] = {r.low, i - 1};

1460 } else {

1463 if (r.low < i - 1)
1464 stack[sp++] = {r.low, i - 1};
1465 }
