82 void init(
83 uint32_t cntEntities, const Entity* ids, const Component* comps, const ChunkDataOffsets& headerOffsets,
84 const ChunkDataOffset* compOffs) {
85 m_header.cntEntities = (uint8_t)cntEntities;
86
87 // Cache pointers to versions
88 m_records.pVersions = (ComponentVersion*)&data(headerOffsets.firstByte_Versions);
89
90 // Cache entity ids
91 if (cntEntities > 0) {
92 auto* dst = m_records.pCompEntities = (Entity*)&data(headerOffsets.firstByte_CompEntities);
93
94 // We treat the entity array as if were MAX_COMPONENTS long.
95 // Real size can be smaller.
96 uint32_t j = 0;
97 for (; j < cntEntities; ++j)
98 dst[j] = ids[j];
99 for (; j < ChunkHeader::MAX_COMPONENTS; ++j)
100 dst[j] = EntityBad;
101 }
102
103 // Cache component records
104 if (cntEntities > 0) {
105 auto* dst = m_records.pRecords = (ComponentRecord*)&data(headerOffsets.firstByte_Records);
106 GAIA_FOR_(cntEntities, j) {
107 dst[j].comp = comps[j];
108 dst[j].pData = &data(compOffs[j]);
109 dst[j].pItem = m_header.cc->find(comps[j].id());
110 }
111 }
112
113 m_records.pEntities = (Entity*)&data(headerOffsets.firstByte_EntityData);
114
115 // Now that records are set, we use the cached component descriptors to set ctor/dtor masks.
116 {
117 auto recs = comp_rec_view();
118 const auto recs_cnt = recs.size();
119 GAIA_FOR(recs_cnt) {
120 const auto& rec = recs[i];
121 if (!component_has_inline_data(rec.comp))
122 continue;
123
124 const auto e = m_records.pCompEntities[i];
125 if (e.kind() == EntityKind::EK_Gen) {
126 m_header.hasAnyCustomGenCtor |= (rec.pItem->func_ctor != nullptr);
127 m_header.hasAnyCustomGenDtor |= (rec.pItem->func_dtor != nullptr);
128 } else {
129 m_header.hasAnyCustomUniCtor |= (rec.pItem->func_ctor != nullptr);
130 m_header.hasAnyCustomUniDtor |= (rec.pItem->func_dtor != nullptr);
131
132 // We construct unique components right away if possible
133 call_ctor(0, *rec.pItem);
134 }
135 }
136 }
137
138 // Make sure world versions are set initially.
139 update_world_version_init();
140 }

509 void save(ser::serializer& s) const {
510 s.save(m_header.count);
511 if (m_header.count == 0)
512 return;
513
514 s.save(m_header.countEnabled);
515
516 const uint16_t dead = m_header.dead;
517 const uint16_t lifespanCountdown = m_header.lifespanCountdown;
518 s.save(dead);
519 s.save(lifespanCountdown);
520
521 const auto cnt = (uint32_t)m_header.count;
522 const auto cap = (uint32_t)m_header.capacity;
523
524 // Store entity data
525 {
526 const auto* pData = m_records.pEntities;
527 GAIA_FOR(cnt) s.save(pData[i]);
528 }
529
530 // Store component data
531 {
532 for (const auto& rec: comp_rec_view()) {
533 // Skip the component if there's no size associated with it
534 if (!component_has_inline_data(rec.comp))
535 continue;
536
537 rec.pItem->save(s, rec.pData, 0, cnt, cap);
538 }
539 }

542 void load(ser::serializer& s) {
543 uint16_t prevCount = m_header.count;
544 s.load(m_header.count);
545 if (m_header.count == 0)
546 return;
547
548 s.load(m_header.countEnabled);
549
550 uint16_t dead = 0;
551 uint16_t lifespanCountdown = 0;
552 s.load(dead);
553 s.load(lifespanCountdown);
554 m_header.dead = dead != 0;
555 m_header.lifespanCountdown = lifespanCountdown;
556
557 const auto cnt = (uint32_t)m_header.count;
558 const auto cap = (uint32_t)m_header.capacity;
559
560 // Load entity data
561 {
562 GAIA_FOR(cnt) {
563 Entity e;
564 s.load(e);
565 entity_view_mut()[i] = e;
566 }
567 }
568
569 // Load component data. Call constructors first as necessary.
570 call_gen_ctors(prevCount, cnt);
571 {
572 for (const auto& rec: comp_rec_view()) {
573 // Skip the component if there's no size associated with it
574 if (!component_has_inline_data(rec.comp))
575 continue;
576
577 rec.pItem->load(s, rec.pData, 0, cnt, cap);
578 }
579 }

1130 void remove_entity_inter(uint16_t row, EntityContainers& recs) {
1131 GAIA_PROF_SCOPE(Chunk::remove_entity_inter);
1132
1133 const uint16_t rowA = row;
1134 const uint16_t rowB = m_header.count - 1;
1135 // The "rowA" entity is the one we are going to destroy so it needs to precede the "rowB"
1136 GAIA_ASSERT(rowA <= rowB);
1137
1138 // To move anything, we need at least 2 entities
1139 if GAIA_LIKELY (rowA < rowB) {
1140 GAIA_ASSERT(m_header.count > 1);
1141
1142 auto ev = entity_view_mut();
1143
1144 // Update entity data
1145 const auto entityB = ev[rowB];
1146 auto& ecB = recs[entityB];
1147#if GAIA_ASSERT_ENABLED
1148 const auto entityA = ev[rowA];
1149 auto& ecA = recs[entityA];
1150
1151 GAIA_ASSERT(ecA.pArchetype == ecB.pArchetype);
1152 GAIA_ASSERT(ecA.pChunk == ecB.pChunk);
1153#endif
1154
1155 ev[rowA] = entityB;
1156
1157 // Move component data from entityB to entityA
1158 auto recView = comp_rec_view();
1159 GAIA_FOR(m_header.genEntities) {
1160 const auto& rec = recView[i];
1161 if (!component_has_inline_data(rec.comp))
1162 continue;
1163
1164 auto* pSrc = (void*)comp_ptr_mut(i);
1165 rec.pItem->move(pSrc, pSrc, rowA, rowB, capacity(), capacity());
1166
1167 pSrc = (void*)comp_ptr_mut(i, rowB);
1168 rec.pItem->dtor(pSrc);
1169 }
1170
1171 // Entity has been replaced with the last one in our chunk. Update its container record.
1172 ecB.row = rowA;
1173 ecB.pEntity = &ev[rowA];
1174 } else if (m_header.hasAnyCustomGenDtor) {
1175 // This is the last entity in the chunk so simply destroy its data
1176 auto recView = comp_rec_view();
1177 GAIA_FOR(m_header.genEntities) {
1178 const auto& rec = recView[i];
1179 if (!component_has_inline_data(rec.comp))
1180 continue;
1181
1182 auto* pSrc = (void*)comp_ptr_mut(i, rowA);
1183 rec.pItem->dtor(pSrc);
1184 }
1185 }

1194 void remove_entity(uint16_t row, EntityContainers& recs) {
1195 if GAIA_UNLIKELY (m_header.count == 0)
1196 return;
1197
1198 GAIA_PROF_SCOPE(Chunk::remove_entity);
1199
1200 if (enabled(row)) {
1201 // Entity was previously enabled. Swap with the last entity
1202 remove_entity_inter(row, recs);
1203 // If this was the first enabled entity make sure to update the row
1204 if (m_header.rowFirstEnabledEntity > 0 && row == m_header.rowFirstEnabledEntity)
1205 --m_header.rowFirstEnabledEntity;
1206 // At this point the last entity is no longer valid so remove it
1207 remove_last_entity();
1208 --m_header.countEnabled;
1209 } else {
1210 // Entity was previously disabled. Swap with the last disabled entity
1211 const uint16_t pivot = size_disabled() - 1;
1212 swap_chunk_entities(row, pivot, recs);
1213 // Once swapped, try to swap with the last (enabled) entity in the chunk.
1214 remove_entity_inter(pivot, recs);
1215 --m_header.rowFirstEnabledEntity;
1216 // At this point the last entity is no longer valid so remove it
1217 remove_last_entity();
1218 }

1228 void swap_chunk_entities(uint16_t rowA, uint16_t rowB, EntityContainers& recs) {
1229 // If there are at least two different entities inside to swap
1230 if GAIA_UNLIKELY (m_header.count <= 1 || rowA == rowB)
1231 return;
1232
1233 GAIA_PROF_SCOPE(Chunk::swap_chunk_entities);
1234
1235 // Update entity data
1236 auto ev = entity_view_mut();
1237 const auto entityA = ev[rowA];
1238 const auto entityB = ev[rowB];
1239
1240 auto& ecA = recs[entityA];
1241 auto& ecB = recs[entityB];
1242 GAIA_ASSERT(ecA.pArchetype == ecB.pArchetype);
1243 GAIA_ASSERT(ecA.pChunk == ecB.pChunk);
1244
1245 ev[rowA] = entityB;
1246 ev[rowB] = entityA;
1247
1248 // Swap component data
1249 auto recView = comp_rec_view();
1250 GAIA_FOR(m_header.genEntities) {

1256 rec.pItem->swap(rec.pData, rec.pData, rowA, rowB, capacity(), capacity());
1257 }
1258
1259 // Update indices in entity container.
1260 ecA.row = rowB;
1261 ecB.row = rowA;
1262 ecA.pEntity = &ev[rowB];
1263 ecB.pEntity = &ev[rowA];

1321 void enable_entity(uint16_t row, bool enableEntity, EntityContainers& recs) {
1322 GAIA_ASSERT(row < m_header.count && "Entity chunk row out of bounds!");
1323
1324 if (enableEntity) {
1325 // Nothing to enable if there are no disabled entities
1326 if (!m_header.has_disabled_entities())
1327 return;
1328 // Trying to enable an already enabled entity
1329 if (enabled(row))
1330 return;
1331 // Try swapping our entity with the last disabled one
1332 const auto entity = entity_view()[row];
1333 swap_chunk_entities(--m_header.rowFirstEnabledEntity, row, recs);
1334 recs[entity].data.dis = 0;
1335 ++m_header.countEnabled;
1336 } else {
1337 // Nothing to disable if there are no enabled entities
1338 if (!m_header.has_enabled_entities())
1339 return;
1340 // Trying to disable an already disabled entity
1341 if (!enabled(row))
1342 return;
1343 // Try swapping our entity with the last one in our chunk
1344 const auto entity = entity_view()[row];
1345 swap_chunk_entities(m_header.rowFirstEnabledEntity++, row, recs);
1346 recs[entity].data.dis = 1;
1347 --m_header.countEnabled;
1348 }
