98 explicit RuntimeSparseComponentStore(const ComponentCacheItem& item): pItem(&item) {
99 const auto size = item.comp.size();
100 if (size == 0)
101 return;
102
103 const auto itemAlignment = item.comp.alig();
104 payloadAlignment = itemAlignment < alignof(void*) ? (uint32_t)alignof(void*) : itemAlignment;
105 const auto payloadSize = size < sizeof(void*) ? (uint32_t)sizeof(void*) : size;
106 payloadStride = mem::align(payloadSize, payloadAlignment);
107 payloadsPerPage = PayloadPageBytes / payloadStride;
108 if (payloadsPerPage == 0)
109 payloadsPerPage = 1;
110 else if (payloadsPerPage > MaxPayloadsPerPage)
111 payloadsPerPage = MaxPayloadsPerPage;
112 nextPayload = payloadsPerPage;
113 }

120 GAIA_NODISCARD void* alloc_payload() {
121 if (pFreePayload != nullptr) {
122 auto* pData = pFreePayload;
123 memcpy(&pFreePayload, pData, sizeof(pFreePayload));
124 return pData;
125 }
126
127 if (nextPayload == payloadsPerPage) {
128 const auto pageBytes = payloadStride * payloadsPerPage;
129 auto* pPage = payloadAlignment <= alignof(std::max_align_t)
130 ? mem::mem_alloc("Runtime sparse payload page", pageBytes)
131 : mem::mem_alloc_alig("Runtime sparse payload page", pageBytes, payloadAlignment);
132 GAIA_ASSERT(pPage != nullptr);
133 payloadPages.push_back(pPage);
134 nextPayload = 0;
135 }
136
137 auto* pData = (uint8_t*)payloadPages.back() + ((uintptr_t)payloadStride * nextPayload);
138 ++nextPayload;
139 return pData;
140 }
141
143 void free_payload(void* pData) {
144 memcpy(pData, &pFreePayload, sizeof(pFreePayload));
145 pFreePayload = pData;
146 }
147
149 void free_payload_pages() {
150 for (auto* pPage: payloadPages) {
151 if (payloadAlignment <= alignof(std::max_align_t))
152 mem::mem_free("Runtime sparse payload page", pPage);
153 else
154 mem::mem_free_alig("Runtime sparse payload page", pPage);
155 }
156 payloadPages.clear();
157 pFreePayload = nullptr;
158 nextPayload = payloadsPerPage;
159 }

161 void* add(Entity entity) {
162 const auto sparseId = sid(entity);
163 if (data.has(sparseId))
164 return data[sparseId].pData;
165
166 void* pData = nullptr;
167 const auto size = pItem->comp.size();
168 if (size != 0) {
169 pData = alloc_payload();
170 GAIA_ASSERT(pData != nullptr);
171 if (pItem->func_ctor != nullptr)
172 pItem->func_ctor(pData, 1);
173 }
174
175 data.add(RuntimeSparseComponentRecord{entity, pData});
176 return pData;
177 }

189 void del_entity(Entity entity) {
190 const auto sparseId = sid(entity);
191 if (data.empty() || !data.has(sparseId))
192 return;
193
194 auto* pData = data[sparseId].pData;
195 if (pData != nullptr) {
196 pItem->dtor(pData);
197 free_payload(pData);
198 }
199 data.del(sparseId);

203 return !data.empty() && data.has(sid(entity));
204 }

206 bool copy_entity(Entity dstEntity, Entity srcEntity) {
207 if (!has(srcEntity))

209
210 auto* pDst = add(dstEntity);
211 const auto* pSrc = get(srcEntity);
212 if (pDst != nullptr)
213 pItem->copy(pDst, pSrc, 0, 0, pItem->comp.size(), pItem->comp.size());

215 }

221 void collect_entities(cnt::darray<Entity>& out) const {
222 out.reserve(out.size() + (uint32_t)data.size());
223 for (const auto& item: data)
224 out.push_back(item.entity);
225 }

227 void clear_store() {
228 while (!data.empty())
229 del_entity(data.begin()->entity);
230 free_payload_pages();
231 }
