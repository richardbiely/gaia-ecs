71 static ComponentCacheItem::SymbolLookupKey short_name_key(const ComponentCacheItem& item) noexcept {
72 const auto* pName = item.name.str();
73 const auto len = item.name.len();
74 for (uint32_t i = len; i > 1; --i) {
75 const auto idx = i - 1;
76 if (pName[idx] != ':' || pName[idx - 1] != ':')
77 continue;
78
79 return ComponentCacheItem::SymbolLookupKey(pName + idx + 1, len - idx - 1, 0);
80 }
81
82 return ComponentCacheItem::SymbolLookupKey();
83 }

114 void validate_runtime_fields(const ecs::ComponentDesc& desc) const noexcept {
115 if (desc.fieldCount == 0)
116 return;
117
118 GAIA_ASSERT(desc.fields != nullptr);
119 if (desc.fields == nullptr)
120 return;
121
122 GAIA_FOR(desc.fieldCount) {
123 const auto& field = desc.fields[i];
124 GAIA_ASSERT(!field.name.empty());
125 GAIA_ASSERT(field.name.size() < ComponentCacheItem::MaxNameLength);
126 GAIA_ASSERT(field.type != EntityBad);
127
128 const auto* pType = find(field.type);
129 GAIA_ASSERT(pType != nullptr);
130 const auto elementSize = pType != nullptr ? pType->comp.size() : 0;
131 GAIA_ASSERT(elementSize > 0);
132 const auto elementCount = ComponentCacheItem::field_element_count(field);
133 GAIA_ASSERT(elementCount > 0);
134 GAIA_ASSERT(field.offset <= desc.size);
135 const auto availableSize = field.offset <= desc.size ? desc.size - field.offset : 0;
136 GAIA_ASSERT(elementSize <= availableSize);
137 GAIA_ASSERT(elementCount <= availableSize / elementSize);
138
139 GAIA_FOR2_(i + 1, desc.fieldCount, otherIdx) {
140 const auto& other = desc.fields[otherIdx];
141 const bool sameLen = field.name.size() == other.name.size();
142 const bool sameName = sameLen && field.name.data() != nullptr && other.name.data() != nullptr &&
143 strncmp(field.name.data(), other.name.data(), field.name.size()) == 0;
144 GAIA_ASSERT(!sameName);
145 }
146 }

150 static bool build_default_path(util::str& out, util::str_view symbol) {
151 out.clear();
152 bool changed = false;
153 out.reserve(symbol.size());
154 for (uint32_t i = 0; i < symbol.size(); ++i) {
155 if (i + 1 < symbol.size() && symbol.data()[i] == ':' && symbol.data()[i + 1] == ':') {
156 out.append('.');
157 ++i;
158 changed = true;
159 continue;
160 }
161
162 out.append(symbol.data()[i]);
163 }
164
165 if (!changed)
166 out.clear();
167
168 return changed;
169 }

176 static bool build_scoped_path(util::str& out, util::str_view scopePath, util::str_view leaf) {
177 out.clear();
178 if (scopePath.empty() || leaf.empty())
179 return false;
180
181 out.reserve(scopePath.size() + 1 + leaf.size());
182 out.append(scopePath);
183 out.append('.');
184 out.append(leaf);
185 return true;
186 }

191 static void initialize_names(ComponentCacheItem& item, util::str_view scopePath = {}) {
192 item.path.clear();
193 const auto symbol = util::str_view{item.name.str(), item.name.len()};
194 if (is_internal_symbol(symbol))
195 return;
196
197 const auto shortName = short_name_key(item);
198 const bool hasShortName =
199 shortName.str() != nullptr && shortName.len() != 0 && shortName.len() != item.name.len();
200 const auto leaf = hasShortName ? util::str_view{shortName.str(), shortName.len()} : symbol;
201
202 if (!build_scoped_path(item.path, scopePath, leaf))
203 (void)build_default_path(item.path, symbol);
204 }

206 static void add_lookup_mapping(
207 cnt::map<ComponentCacheItem::SymbolLookupKey, ResolvedLookupEntry>& map, util::str_view view,
208 const ComponentCacheItem& item) {
209 if (view.empty())
210 return;
211
212 const auto key = lookup_key(view);
213 const auto it = map.find(key);
214 if (it == map.end()) {
215 map.emplace(key, ResolvedLookupEntry{&item, 1});
216 return;
217 }
218
219 auto& entry = it->second;
220 GAIA_ASSERT(entry.matches != 0);
221 ++entry.matches;
222 }

228 static void push_unique_entity(cnt::darray<Entity>& out, Entity entity) {
229 if (entity == EntityBad)
230 return;
231
232 for (const auto existing: out) {
233 if (existing == entity)
234 return;
235 }

237 out.push_back(entity);

240 void remove_path_mapping(const ComponentCacheItem& item) {
241 const auto pathValue = item.path.view();
242 if (pathValue.empty())
243 return;
244
245 const auto key = lookup_key(pathValue);
246 const auto it = m_compByPath.find(key);
247 if (it == m_compByPath.end())
248 return;
249
250 auto& entry = it->second;
251 GAIA_ASSERT(entry.matches != 0);
252
253 if (entry.matches == 1) {
254 GAIA_ASSERT(entry.pItem == &item);
255 m_compByPath.erase(it);
256 return;
257 }
258
259 --entry.matches;
260 if (entry.pItem != &item)
261 return;
262
263 entry.pItem = nullptr;
264 for (const auto& [entityId, pItem]: m_compByEntityId) {
265 (void)entityId;
266 GAIA_ASSERT(pItem != nullptr);
267 if (pItem == &item)
268 continue;
269
270 if (pItem->path.view() != pathValue)
271 continue;
272
273 entry.pItem = pItem;
274 return;
275 }
276
277 GAIA_ASSERT(false);
278 }

570 void add_path_matches(cnt::darray<Entity>& out, util::str_view name) const {
571 if (name.empty())
572 return;
573
574 const auto it = m_compByPath.find(lookup_key(name));
575 if (it == m_compByPath.end())
576 return;
577
578 const auto& entry = it->second;
579 GAIA_ASSERT(entry.matches != 0);
580
581 if (entry.matches == 1) {
582 GAIA_ASSERT(entry.pItem != nullptr);
583 push_unique_entity(out, entry.pItem->entity);
584 return;
585 }
586
587 for (const auto& [entityId, pItem]: m_compByEntityId) {
588 (void)entityId;
589 GAIA_ASSERT(pItem != nullptr);
590 if (pItem->path.view() == name)
591 push_unique_entity(out, pItem->entity);
592 }
