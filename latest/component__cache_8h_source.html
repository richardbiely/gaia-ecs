71 static ComponentCacheItem::SymbolLookupKey short_name_key(const ComponentCacheItem& item) noexcept {
72 const auto* pName = item.name.str();
73 const auto len = item.name.len();
74 for (uint32_t i = len; i > 1; --i) {
75 const auto idx = i - 1;
76 if (pName[idx] != ':' || pName[idx - 1] != ':')
77 continue;
78
79 return ComponentCacheItem::SymbolLookupKey(pName + idx + 1, len - idx - 1, 0);
80 }
81
82 return ComponentCacheItem::SymbolLookupKey();
83 }

114 void validate_runtime_type(const ecs::ComponentDesc& desc) const noexcept {
115 ser::serialization_type_id type = ser::serialization_type_id::ignore;
116 if (desc.typeKind == RuntimeTypeKind::Enum || desc.typeKind == RuntimeTypeKind::Bitmask) {
117 GAIA_ASSERT(runtime_primitive_serialization_type(desc.underlyingType, type));
118 GAIA_ASSERT(desc.elementType == EntityBad);
119 GAIA_ASSERT(desc.elementCount == 0);
120 GAIA_ASSERT(desc.opaqueAsType == EntityBad);
121 return;
122 }
123
124 if (desc.typeKind == RuntimeTypeKind::Array) {
125 GAIA_ASSERT(desc.underlyingType == EntityBad);
126 GAIA_ASSERT(desc.elementType != EntityBad);
127 GAIA_ASSERT(desc.elementCount > 0);
128 GAIA_ASSERT(desc.fields == nullptr);
129 GAIA_ASSERT(desc.fieldCount == 0);
130 GAIA_ASSERT(desc.constants == nullptr);
131 GAIA_ASSERT(desc.constantCount == 0);
132 GAIA_ASSERT(desc.opaqueAsType == EntityBad);

137 GAIA_ASSERT(elementSize > 0);

140 GAIA_ASSERT(desc.size == elementSize * desc.elementCount);
141 }
142 GAIA_ASSERT(desc.alig == elementAlignment);

191 void validate_runtime_fields(const ecs::ComponentDesc& desc) const noexcept {
192 if (desc.fieldCount == 0)
193 return;
194
195 GAIA_ASSERT(desc.fields != nullptr);
196 if (desc.fields == nullptr)
197 return;
198
199 GAIA_FOR(desc.fieldCount) {
200 const auto& field = desc.fields[i];
201 GAIA_ASSERT(!field.name.empty());
202 GAIA_ASSERT(field.name.size() < ComponentCacheItem::MaxNameLength);
203 GAIA_ASSERT(field.type != EntityBad);
204
205 const auto* pType = find(field.type);
206 GAIA_ASSERT(pType != nullptr);
207 GAIA_ASSERT(pType == nullptr || pType->typeKind != RuntimeTypeKind::Array || field.count == 0);
208 const auto elementSize = pType != nullptr ? pType->comp.size() : 0;
209 GAIA_ASSERT(elementSize > 0);
210 const auto elementCount = ComponentCacheItem::field_element_count(field);
211 GAIA_ASSERT(elementCount > 0);
212 GAIA_ASSERT(field.offset <= desc.size);
213 const auto availableSize = field.offset <= desc.size ? desc.size - field.offset : 0;
214 GAIA_ASSERT(elementSize <= availableSize);
215 GAIA_ASSERT(elementCount <= availableSize / elementSize);
216
217 GAIA_FOR2_(i + 1, desc.fieldCount, otherIdx) {
218 const auto& other = desc.fields[otherIdx];
219 const bool sameLen = field.name.size() == other.name.size();
220 const bool sameName = sameLen && field.name.data() != nullptr && other.name.data() != nullptr &&
221 strncmp(field.name.data(), other.name.data(), field.name.size()) == 0;
222 GAIA_ASSERT(!sameName);
223 }
224 }
225 }

255 static bool build_default_path(util::str& out, util::str_view symbol) {
256 out.clear();
257 bool changed = false;
258 out.reserve(symbol.size());
259 for (uint32_t i = 0; i < symbol.size(); ++i) {
260 if (i + 1 < symbol.size() && symbol.data()[i] == ':' && symbol.data()[i + 1] == ':') {
261 out.append('.');
262 ++i;
263 changed = true;
264 continue;
265 }
266
267 out.append(symbol.data()[i]);
268 }
269
270 if (!changed)
271 out.clear();
272
273 return changed;
274 }

281 static bool build_scoped_path(util::str& out, util::str_view scopePath, util::str_view leaf) {
282 out.clear();
283 if (scopePath.empty() || leaf.empty())
284 return false;
285
286 out.reserve(scopePath.size() + 1 + leaf.size());
287 out.append(scopePath);
288 out.append('.');
289 out.append(leaf);
290 return true;

296 static void initialize_names(ComponentCacheItem& item, util::str_view scopePath = {}) {
297 item.path.clear();
298 const auto symbol = util::str_view{item.name.str(), item.name.len()};
299 if (is_internal_symbol(symbol))
300 return;
301
302 const auto shortName = short_name_key(item);
303 const bool hasShortName =
304 shortName.str() != nullptr && shortName.len() != 0 && shortName.len() != item.name.len();
305 const auto leaf = hasShortName ? util::str_view{shortName.str(), shortName.len()} : symbol;
306
307 if (!build_scoped_path(item.path, scopePath, leaf))
308 (void)build_default_path(item.path, symbol);
309 }

311 static void add_lookup_mapping(
312 cnt::map<ComponentCacheItem::SymbolLookupKey, ResolvedLookupEntry>& map, util::str_view view,
313 const ComponentCacheItem& item) {
314 if (view.empty())
315 return;
316
317 const auto key = lookup_key(view);
318 const auto it = map.find(key);
319 if (it == map.end()) {
320 map.emplace(key, ResolvedLookupEntry{&item, 1});
321 return;
322 }
323
324 auto& entry = it->second;
325 GAIA_ASSERT(entry.matches != 0);
326 ++entry.matches;
327 }

333 static void push_unique_entity(cnt::darray<Entity>& out, Entity entity) {
334 if (entity == EntityBad)
335 return;

338 if (existing == entity)

342 out.push_back(entity);

345 void remove_path_mapping(const ComponentCacheItem& item) {
346 const auto pathValue = item.path.view();
347 if (pathValue.empty())
348 return;
349
350 const auto key = lookup_key(pathValue);
351 const auto it = m_compByPath.find(key);
352 if (it == m_compByPath.end())
353 return;
354
355 auto& entry = it->second;
356 GAIA_ASSERT(entry.matches != 0);
357
358 if (entry.matches == 1) {
359 GAIA_ASSERT(entry.pItem == &item);
360 m_compByPath.erase(it);
361 return;
362 }
363
364 --entry.matches;
365 if (entry.pItem != &item)
366 return;
367
368 entry.pItem = nullptr;
369 for (const auto& [entityId, pItem]: m_compByEntityId) {
370 (void)entityId;
371 GAIA_ASSERT(pItem != nullptr);
372 if (pItem == &item)
373 continue;
374
375 if (pItem->path.view() != pathValue)
376 continue;
377
378 entry.pItem = pItem;
379 return;
380 }
381
382 GAIA_ASSERT(false);
383 }

646 void add_path_matches(cnt::darray<Entity>& out, util::str_view name) const {
647 if (name.empty())
648 return;
649
650 const auto it = m_compByPath.find(lookup_key(name));
651 if (it == m_compByPath.end())
652 return;
653
654 const auto& entry = it->second;
655 GAIA_ASSERT(entry.matches != 0);
656
657 if (entry.matches == 1) {
658 GAIA_ASSERT(entry.pItem != nullptr);
659 push_unique_entity(out, entry.pItem->entity);
660 return;
661 }
662
663 for (const auto& [entityId, pItem]: m_compByEntityId) {
664 (void)entityId;
665 GAIA_ASSERT(pItem != nullptr);
666 if (pItem->path.view() == name)
667 push_unique_entity(out, pItem->entity);
668 }
