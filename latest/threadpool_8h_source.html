168 void set_max_workers(uint32_t count, uint32_t countHighPrio) {
169 const auto workersCnt = core::get_max(core::get_min(MaxWorkers, count), 1U);
170 if (countHighPrio > count)
171 countHighPrio = count;
172
173 // Stop all threads first
174 reset();
175
176 // Reset previous worker contexts
177 for (auto& ctx: m_workersCtx)
178 ctx.reset();
179
180 // Resize or array
181 m_workersCtx.resize(workersCnt);
182 // We also have the main thread so there's always one less worker spawned
183 m_workers.resize(workersCnt - 1);
184
185 // First worker is considered the main thread.
186 // It is also assigned high priority but it doesn't really matter.
187 // The main thread can steal any jobs, both low and high priority.
188 detail::tl_workerCtx = m_workersCtx.data();
189 m_workersCtx[0].tp = this;
190 m_workersCtx[0].workerIdx = 0;
191 m_workersCtx[0].prio = JobPriority::High;
192
193 // Reset the workers
194 for (auto& worker: m_workers)
195 worker = {};
196
197 // Create a new set of high and low priority threads (if any)
198 uint32_t workerIdx = 1;
199 set_workers_high_prio_inter(workerIdx, countHighPrio);
200 }
201
206 void set_workers_high_prio_inter(uint32_t& workerIdx, uint32_t count) {
207 count = gaia::core::get_min(count, m_workers.size());
208 if (count == 0) {
209 m_workersCnt[0] = 1; // main thread
210 m_workersCnt[1] = 0;
211 } else {
212 m_workersCnt[0] = count + 1; // Main thread is always a priority worker
213 m_workersCnt[1] = m_workers.size() - count;
214 }
215
216 // Create a new set of high and low priority threads (if any)
217 create_worker_threads(workerIdx, JobPriority::High, m_workersCnt[0] - 1);
218 create_worker_threads(workerIdx, JobPriority::Low, m_workersCnt[1]);
219 }
220
225 void set_workers_low_prio_inter(uint32_t& workerIdx, uint32_t count) {
226 const uint32_t realCnt = gaia::core::get_max(count, m_workers.size());
227 if (realCnt == 0) {
228 m_workersCnt[0] = 0;
229 m_workersCnt[1] = 1; // main thread
230 } else {
231 m_workersCnt[0] = m_workers.size() - realCnt;
232 m_workersCnt[1] = realCnt + 1; // Main thread is always a priority worker;
233 }
234
235 // Create a new set of high and low priority threads (if any)
236 create_worker_threads(workerIdx, JobPriority::High, m_workersCnt[0]);
237 create_worker_threads(workerIdx, JobPriority::Low, m_workersCnt[1]);
238 }

385 void submit(std::span<JobHandle> jobHandles) {
386 if (jobHandles.empty())
387 return;
388
389 GAIA_PROF_SCOPE(tp::submit);
390
391 auto* pHandles = (JobHandle*)alloca(sizeof(JobHandle) * jobHandles.size());
392
393 uint32_t cnt = 0;
394 for (auto handle: jobHandles) {
395 GAIA_ASSERT(handle != (JobHandle)JobNull_t{});
396
397 auto& jobData = m_jobManager.data(handle);
398 const auto state = m_jobManager.submit(jobData) & JobState::DEP_BITS_MASK;
399 // Jobs that were already submitted won't be submitted again.
400 // We can only accept the job if it has no pending dependencies.
401 if (state != 0)
402 continue;
403
404 pHandles[cnt++] = handle;
405 }
406
407 auto* ctx = detail::tl_workerCtx;
408 process(std::span(pHandles, cnt), ctx);
409 }

438 void reset(std::span<JobHandle> jobHandles) {
439 if (jobHandles.empty())
440 return;
441
442 GAIA_ASSERT(main_thread());
443 GAIA_PROF_SCOPE(tp::reset_wait);
444
445 // Wait first to avoid resetting one handle while another one still depends on it.
446 for (auto handle: jobHandles) {
447 if (handle == (JobHandle)JobNull_t{})
448 continue;
449 wait(handle);
450 }
451
452 for (auto handle: jobHandles) {
453 if (handle == (JobHandle)JobNull_t{})
454 continue;
455
456 auto& jobData = m_jobManager.data(handle);
457 const auto state = jobData.state.load() & JobState::STATE_BITS_MASK;
458 // Auto-deleted jobs are released and cannot be reused through reset_state().
459 if (state == JobState::Released)
460 continue;
461
462 m_jobManager.reset_state(jobData);
463 }
464 }

502 JobHandle sched_par(JobParallel job, uint32_t itemsToProcess, uint32_t groupSize) {
503 GAIA_ASSERT(main_thread());
504
505 // Empty data set are considered wrong inputs
506 GAIA_ASSERT(itemsToProcess != 0);
507 if (itemsToProcess == 0)
508 return JobNull;
509
510 // Don't add new jobs once stop was requested
511 if GAIA_UNLIKELY (m_stop)
512 return JobNull;
513
514 // Make sure the right priority is selected
515 const auto prio = job.priority = final_prio(job);
516
517 // No group size was given, make a guess based on the set size
518 if (groupSize == 0) {
519 const auto cntWorkers = m_workersCnt[(uint32_t)prio];
520 groupSize = (itemsToProcess + cntWorkers - 1) / cntWorkers;
521
522 // If there are too many items we split them into multiple jobs.
523 // This way, if we wait for the result and some workers finish
524 // with our task faster, the finished worker can pick up a new
525 // job faster.
526 // On the other hand, too little items probably don't deserve
527 // multiple jobs.
528 constexpr uint32_t maxUnitsOfWorkPerGroup = 8;
529 groupSize = groupSize / maxUnitsOfWorkPerGroup;
530 if (groupSize <= 0)
531 groupSize = 1;
532 }
533
534 const auto jobs = (itemsToProcess + groupSize - 1) / groupSize;
535
536 // Only one job is created, use the job directly.
537 // Generally, this is the case we would want to avoid because it means this particular case
538 // is not worth of being scheduled via sched_par. However, we can never know for sure what
539 // the reason for that is so let's stay silent.
540 if (jobs == 1) {
541 const uint32_t groupJobIdxEnd = groupSize < itemsToProcess ? groupSize : itemsToProcess;
542 auto groupFunc = GAIA_MOV(job.func);
543 auto groupJobFunc = [func = GAIA_MOV(groupFunc), groupJobIdxEnd]() mutable {
544 JobArgs args;
545 args.idxStart = 0;
546 args.idxEnd = groupJobIdxEnd;
547 func(args);
548 };
549
550 auto handle = add(Job{GAIA_MOV(groupJobFunc), prio, JobCreationFlags::Default});
551 submit(handle);
552 return handle;
553 }
554
555 // Multiple jobs need to be parallelized.
556 // Create a sync job and assign it as their dependency.
557 auto callbackHandle = add_parallel_callback(GAIA_MOV(job.func), jobs);
558
559 auto* pHandles = (JobHandle*)alloca(sizeof(JobHandle) * (jobs + 1));
560 std::span<JobHandle> handles(pHandles, jobs + 1);
561
562 add_n(prio, handles);
563
564#if GAIA_ASSERT_ENABLED
565 for (auto jobHandle: handles)
566 GAIA_ASSERT(m_jobManager.is_clear(jobHandle));
567#endif
568
569 // Work jobs
570 for (uint32_t jobIndex = 0; jobIndex < jobs; ++jobIndex) {
571 const uint32_t groupJobIdxStart = jobIndex * groupSize;
572 const uint32_t groupJobIdxStartPlusGroupSize = groupJobIdxStart + groupSize;
573 const uint32_t groupJobIdxEnd =
574 groupJobIdxStartPlusGroupSize < itemsToProcess ? groupJobIdxStartPlusGroupSize : itemsToProcess;
575
576 auto groupJobFunc = [this, callbackHandle, groupJobIdxStart, groupJobIdxEnd]() {
577 JobArgs args;
578 args.idxStart = groupJobIdxStart;
579 args.idxEnd = groupJobIdxEnd;
580 m_jobManager.invoke_parallel_callback(callbackHandle, args);
581 release_parallel_callback(callbackHandle);
582 };
583
584 auto& jobData = m_jobManager.data(pHandles[jobIndex]);
585 jobData.func = util::SmallFunc::create(GAIA_MOV(groupJobFunc));
586 jobData.prio = prio;
587 }
588 // Sync job
589 {
590 auto& jobData = m_jobManager.data(pHandles[jobs]);
591 jobData.prio = prio;
592 }
593
594 // Assign the sync jobs as a dependency for work jobs
595 dep(handles.subspan(0, jobs), pHandles[jobs]);
596
597 // Sumbit the jobs to the threadpool.
598 // This is a point of no return. After this point no more changes to jobs are possible.
599 submit(handles);
600 return pHandles[jobs];
601 }

610 JobHandle sched_par(JobParallelRef job, uint32_t itemsToProcess, uint32_t groupSize) {
611 GAIA_ASSERT(main_thread());
612 GAIA_ASSERT(job.pCtx != nullptr);
613 GAIA_ASSERT(job.invoke != nullptr);
614
615 GAIA_ASSERT(itemsToProcess != 0);
616 if (itemsToProcess == 0)
617 return JobNull;
618
619 if GAIA_UNLIKELY (m_stop)
620 return JobNull;
621
622 const auto prio = job.priority = final_prio(job);
623
624 if (groupSize == 0) {
625 const auto cntWorkers = m_workersCnt[(uint32_t)prio];
626 groupSize = (itemsToProcess + cntWorkers - 1) / cntWorkers;
627
628 constexpr uint32_t maxUnitsOfWorkPerGroup = 8;
629 groupSize = groupSize / maxUnitsOfWorkPerGroup;
630 if (groupSize <= 0)
631 groupSize = 1;
632 }
633
634 const auto jobs = (itemsToProcess + groupSize - 1) / groupSize;
635
636 if (jobs == 1) {
637 const uint32_t groupJobIdxEnd = groupSize < itemsToProcess ? groupSize : itemsToProcess;
638 auto* pCtx = job.pCtx;
639 auto invoke = job.invoke;
640 auto groupJobFunc = [pCtx, invoke, groupJobIdxEnd]() {
641 JobArgs args;
642 args.idxStart = 0;
643 args.idxEnd = groupJobIdxEnd;
644 invoke(pCtx, args);
645 };
646
647 auto handle = add(Job{GAIA_MOV(groupJobFunc), prio, JobCreationFlags::Default});
648 submit(handle);
649 return handle;
650 }
651
652 auto* pHandles = (JobHandle*)alloca(sizeof(JobHandle) * (jobs + 1));
653 std::span<JobHandle> handles(pHandles, jobs + 1);
654
655 add_n(prio, handles);
656
657#if GAIA_ASSERT_ENABLED
658 for (auto jobHandle: handles)
659 GAIA_ASSERT(m_jobManager.is_clear(jobHandle));
660#endif
661
662 for (uint32_t jobIndex = 0; jobIndex < jobs; ++jobIndex) {
663 const uint32_t groupJobIdxStart = jobIndex * groupSize;
664 const uint32_t groupJobIdxStartPlusGroupSize = groupJobIdxStart + groupSize;
665 const uint32_t groupJobIdxEnd =
666 groupJobIdxStartPlusGroupSize < itemsToProcess ? groupJobIdxStartPlusGroupSize : itemsToProcess;
667
668 auto* pCtx = job.pCtx;
669 auto invoke = job.invoke;
670 auto groupJobFunc = [pCtx, invoke, groupJobIdxStart, groupJobIdxEnd]() {
671 JobArgs args;
672 args.idxStart = groupJobIdxStart;
673 args.idxEnd = groupJobIdxEnd;
674 invoke(pCtx, args);
675 };
676
677 auto& jobData = m_jobManager.data(pHandles[jobIndex]);
678 jobData.func = util::SmallFunc::create(GAIA_MOV(groupJobFunc));
679 jobData.prio = prio;
680 }
681 {
682 auto& jobData = m_jobManager.data(pHandles[jobs]);
683 jobData.prio = prio;
684 }
685
686 dep(handles.subspan(0, jobs), pHandles[jobs]);
687 submit(handles);
688 return pHandles[jobs];
689 }

695 void wait(JobHandle jobHandle) {
696 GAIA_PROF_SCOPE(tp::wait);
697
698 GAIA_ASSERT(main_thread());
699
700 // Skip waitinig for unset job handles.
701 if (jobHandle == (JobHandle)JobNull_t{})
702 return;
703
704 auto* ctx = detail::tl_workerCtx;
705 auto& jobData = m_jobManager.data(jobHandle);
706 auto state = jobData.state.load();
707
708 // Waiting for a job that has not been initialized is nonsense.
709 GAIA_ASSERT(state != 0);
710
711 // Wait until done
712 for (; (state & JobState::STATE_BITS_MASK) < JobState::Done; state = jobData.state.load()) {
713 // The job we are waiting for is not finished yet, try running some other job in the meantime
714 JobHandle otherJobHandle;
715 if (try_fetch_job(*ctx, otherJobHandle)) {
716 if (run(otherJobHandle, ctx))
717 continue;
718 }
719
720 // The job we are waiting for is already running.
721 // Wait until it signals it's finished.
722 if ((state & JobState::STATE_BITS_MASK) == JobState::Executing) {
723 const auto workerId = (state & JobState::DEP_BITS_MASK);
724 auto* jobDoneEvent = &m_workersCtx[workerId].event;
725 jobDoneEvent->wait();
726 continue;
727 }
728
729 // The worst case scenario.
730 // We have nothing to do and the job we are waiting for is not executing still.
731 // Let's wait for any job to start executing.
732 const auto workerBit = 1U << ctx->workerIdx;
733 const auto oldBlockedMask = m_blockedInWorkUntil.fetch_or(workerBit);
734 const auto newState = jobData.state.load();
735 if (newState == state) // still not JobState::Done?
736 Futex::wait(&m_blockedInWorkUntil, oldBlockedMask | workerBit, detail::WaitMaskAny);
737 m_blockedInWorkUntil.fetch_and(~workerBit);
738 }
739 }

756 GAIA_NODISCARD static uint32_t hw_efficiency_cores_cnt() {
757 uint32_t efficiencyCores = 0;
758#if GAIA_PLATFORM_APPLE
759 size_t size = sizeof(efficiencyCores);
760 if (sysctlbyname("hw.perflevel1.logicalcpu", &efficiencyCores, &size, nullptr, 0) != 0)
761 return 0;
762#elif GAIA_PLATFORM_FREEBSD
763 int cpuIndex = 0;
764 char oidName[32];
765 int coreType;
766 size_t size = sizeof(coreType);
767 while (true) {
768 GAIA_STRFMT(oidName, sizeof(oidName), "dev.cpu.%d.coretype", cpuIndex);
769 if (sysctlbyname(oidName, &coreType, &size, nullptr, 0) != 0)
770 break; // Stop on the last CPU index
771
772 // 0 = performance core
773 // 1 = efficiency core
774 if (coreType == 1)
775 ++efficiencyCores;
776
777 ++cpuIndex;
778 }
779#elif GAIA_PLATFORM_WINDOWS
780 DWORD length = 0;
781
782 // First, determine required buffer size
783 if (!GetLogicalProcessorInformationEx(RelationProcessorCore, nullptr, &length))
784 return 0;
785
786 // Allocate enough memory
787 auto* pBuffer = (SYSTEM_LOGICAL_PROCESSOR_INFORMATION_EX*)malloc(length);
788 if (pBuffer == nullptr)
789 return 0;
790
791 // Retrieve the data
792 if (!GetLogicalProcessorInformationEx(RelationProcessorCore, pBuffer, &length)) {
793 free(pBuffer);
794 return 0;
795 }
796
797 uint32_t heterogenousCnt = 0;
798
799 // Iterate over processor core entries.
800 // On Windows we can't directly tell whether a core is an efficiency core or a performance core.
801 // Instead:
802 // - lower EfficiencyClass values correspond to more efficient cores
803 // - higher EfficiencyClass values correspond to higher performance cores
804 // - EfficiencyClass is zero for homogeneous CPU architectures
805 // Therefore, to count efficiency cores on Windows, we will count cores where EfficiencyClass == 0.
806 // On heterogeneous this should gives us the correct results.
807 // On homogenous architectures, the value is always 0 so rather than calculating the number of efficiency
808 // cores, we would calculate the number of performance cores. For the sake of correctness, if all cores return
809 // 0, we use 0 for the number of efficiency cores.
810 for (char* ptr = (char*)pBuffer; ptr < (char*)pBuffer + length;
811 ptr += ((SYSTEM_LOGICAL_PROCESSOR_INFORMATION_EX*)ptr)->Size) {
812 auto* entry = (SYSTEM_LOGICAL_PROCESSOR_INFORMATION_EX*)ptr;
813 if (entry->Relationship == RelationProcessorCore) {
814 if (entry->Processor.EfficiencyClass == 0)
815 ++efficiencyCores;
816 else
817 ++heterogenousCnt;
818 }
819 }
820
821 if (heterogenousCnt == 0)
822 efficiencyCores = 0;
823
824 free(pBuffer);
825#elif GAIA_PLATFORM_LINUX
826 {
827 // Intel has /sys/devices/cpu_core/cpus, /sys/devices/cpu_atom/cpus on some systems
828 DIR* dir = opendir("/sys/devices/cpu_atom/cpus/");
829 if (dir == nullptr)
830 return 0;
831
832 dirent* entry;
833 while ((entry = readdir(dir)) != nullptr) {
834 if (strncmp(entry->d_name, "cpu", 3) == 0 && entry->d_name[3] >= '0' && entry->d_name[3] <= '9')
835 ++efficiencyCores;
836 }
837
838 closedir(dir);
839 }
840
841 if (efficiencyCores == 0) {
842 // TODO: Go through all CPUs packages and CPUs and determine the differences between them.
843 // There are many metrics.
844 // 1) We will assume all CPUs to be of the same architecture.
845 // 2) Same CPU architecture but different cache sizes. Smaller ones are "efficiency" cores.
846 // This is the AMD way. Still, these are about the same things so maybe we would just treat
847 // all such cores as performance cores.
848 // 3) Different max frequencies on different cores. This might be indicative enough.
849 // There is also an optional parameter present on ARM CPUs:
850 // https://www.kernel.org/doc/Documentation/devicetree/bindings/arm/cpu-capacity.txt
851 // In this case, we'd treat CPUs with the highest capacity-dmips-mhz as performance cores,
852 // and consider the rest as efficiency cores.
853
854 // ...
855 }
856#endif
857 return efficiencyCores;
858 }

863
864 detail::tl_workerCtx = &ctx;

867 // Needs to be called from inside the thread because some platforms

885 void create_thread(uint32_t workerIdx, JobPriority prio) {
886 // Idx 0 is reserved for the main thread
887 GAIA_ASSERT(workerIdx > 0);
888
889 auto& ctx = m_workersCtx[workerIdx];
890 ctx.tp = this;
891 ctx.workerIdx = workerIdx;
892 ctx.prio = prio;
893 ctx.threadCreated = false;
894
895#if GAIA_THREAD_PLATFORM == GAIA_THREAD_STD
896 m_workers[workerIdx - 1] = std::thread([&ctx]() {
897 thread_func((void*)&ctx);
898 });
899#else
900 pthread_attr_t attr{};
901 int ret = pthread_attr_init(&attr);
902 if (ret != 0) {
903 GAIA_LOG_W("pthread_attr_init failed for worker thread %u. ErrCode = %d", workerIdx, ret);
904 return;
905 }
906
908 // Apple's recommendation for Apple Silicon for games / high-perf software
909 // ========================================================================
910 // Per frame | Scheduling policy | QoS class / Priority
911 // ========================================================================
912 // Main thread | SCHED_OTHER | QOS_CLASS_USER_INTERACTIVE (47)
913 // Render/Audio thread | SCHED_RR | 45
914 // Workers High Prio | SCHED_RR | 39-41
915 // Workers Low Prio | SCHED_OTHER | QOS_CLASS_USER_INTERACTIVE (38)
916 // ========================================================================
917 // Multiple-frames | |
918 // ========================================================================
919 // Async Workers High Prio| SCHED_OTHER | QOS_CLASS_USER_INITIATED (37)
920 // Async Workers Low Prio | SCHED_OTHER | QOS_CLASS_DEFAULT (31)
921 // Prefetching/Streaming | SCHED_OTHER | QOS_CLASS_UTILITY (20)
922 // ========================================================================
923
924 if (prio == JobPriority::Low) {
925 #if GAIA_PLATFORM_APPLE
926 ret = pthread_attr_set_qos_class_np(&attr, QOS_CLASS_USER_INTERACTIVE, -9); // 47-9=38
927 if (ret != 0) {
928 GAIA_LOG_W(
929 "pthread_attr_set_qos_class_np failed for worker thread %u [prio=%u]. ErrCode = %d", workerIdx,
930 (uint32_t)prio, ret);
931 }
932 #else
933 ret = pthread_attr_setschedpolicy(&attr, SCHED_OTHER);
934 if (ret != 0) {
935 GAIA_LOG_W(
936 "pthread_attr_setschedpolicy SCHED_RR failed for worker thread %u [prio=%u]. ErrCode = %d", workerIdx,
937 (uint32_t)prio, ret);
938 }
939
940 int prioMax = core::get_min(38, sched_get_priority_max(SCHED_OTHER));
941 int prioMin = core::get_min(prioMax, sched_get_priority_min(SCHED_OTHER));
942 int prioUse = core::get_min(prioMin + 5, prioMax);
943 prioUse = core::get_max(prioUse, prioMin);
944 sched_param param{};
945 param.sched_priority = prioUse;
946
947 ret = pthread_attr_setschedparam(&attr, &param);
948 if (ret != 0) {
949 GAIA_LOG_W(
950 "pthread_attr_setschedparam %d failed for worker thread %u [prio=%u]. ErrCode = %d",
951 param.sched_priority, workerIdx, (uint32_t)prio, ret);
952 }
953 #endif
954 } else {
955 ret = pthread_attr_setschedpolicy(&attr, SCHED_RR);
956 if (ret != 0) {
957 GAIA_LOG_W(
958 "pthread_attr_setschedpolicy SCHED_RR failed for worker thread %u [prio=%u]. ErrCode = %d", workerIdx,
959 (uint32_t)prio, ret);
960 }
961
962 int prioMax = core::get_min(41, sched_get_priority_max(SCHED_RR));
963 int prioMin = core::get_min(prioMax, sched_get_priority_min(SCHED_RR));
964 int prioUse = core::get_max(prioMax - 5, prioMin);
965 prioUse = core::get_min(prioUse, prioMax);
966 sched_param param{};
967 param.sched_priority = prioUse;
968
969 ret = pthread_attr_setschedparam(&attr, &param);
970 if (ret != 0) {
971 GAIA_LOG_W(
972 "pthread_attr_setschedparam %d failed for worker thread %u [prio=%u]. ErrCode = %d",
973 param.sched_priority, workerIdx, (uint32_t)prio, ret);
974 }
975 }
976
977 // Create the thread with given attributes
978 ret = pthread_create(&m_workers[workerIdx - 1], &attr, thread_func, (void*)&ctx);
979 if (ret != 0) {
980 GAIA_LOG_W("pthread_create failed for worker thread %u. ErrCode = %d", workerIdx, ret);
981 } else {
982 ctx.threadCreated = true;
983 }
984
985 pthread_attr_destroy(&attr);
986#endif
987
988 // Stick each thread to a specific CPU core if possible
989 set_thread_affinity(workerIdx);
990 }

1152 void main_thread_tick() {
1153 auto& ctx = *detail::tl_workerCtx;
1154
1155 // Keep executing while there is work
1156 while (true) {
1157 JobHandle jobHandle;
1158 if (!try_fetch_job(ctx, jobHandle))
1159 break;
1160
1161 (void)run(jobHandle, &ctx);
1162 }
1163 }
1164
1165 bool try_fetch_job(ThreadCtx& ctx, JobHandle& jobHandle) {
1166 // Try getting a job from the local queue
1167 if (ctx.jobQueue.try_pop(jobHandle))
1168 return true;
1169
1170 // Try getting a job from the global queue
1171 if (m_jobQueue[(uint32_t)ctx.prio].try_pop(jobHandle))
1172 return true;
1173
1174 // Could not get a job, try stealing from other workers.
1175 const auto workerCnt = m_workersCtx.size();
1176 for (uint32_t i = 0; i < workerCnt;) {
1177 // We need to skip our worker
1178 if (i == ctx.workerIdx) {
1179 ++i;
1180 continue;
1181 }
1182
1183 // Try stealing a job
1184 const auto res = m_workersCtx[i].jobQueue.try_steal(jobHandle);
1185
1186 // Race condition, try again from the same context
1187 if (!res)
1188 continue;
1189
1190 // Stealing can return true if the queue is empty.
1191 // We return right away only if we receive a valid handle which means
1192 // when there was an idle job in the queue.
1193 if (res && jobHandle != (JobHandle)JobNull_t{})
1194 return true;
1195
1196 ++i;
1197 }
1198
1199 return false;
1200 }

1205 void worker_loop(ThreadCtx& ctx) {
1206 while (true) {
1207 // Wait for work
1208 m_sem.wait();
1209
1210 // Keep executing while there is work
1211 while (true) {
1212 JobHandle jobHandle;
1213 if (!try_fetch_job(ctx, jobHandle))
1214 break;
1215
1216 (void)run(jobHandle, detail::tl_workerCtx);
1217 }
1218
1219 // Check if the worker can keep running
1220 const bool stop = m_stop.load();
1221 if (stop)
1222 break;
1223 }
1224 }

1227 void reset() {
1228 if (m_workers.empty())
1229 return;
1230
1231 // Request stopping
1232 m_stop.store(true);
1233
1234 // Signal all threads
1235 m_sem.release((int32_t)m_workers.size());
1236
1237 auto* ctx = detail::tl_workerCtx;
1238 if (ctx == nullptr) {
1239 // reset() can be reached during static teardown from a thread that never
1240 // entered the pool and therefore has no TLS worker context bound.
1241 ctx = &m_workersCtx[0];
1242 detail::tl_workerCtx = ctx;
1243 }
1244
1245 // Finish remaining jobs
1246 JobHandle jobHandle;
1247 while (try_fetch_job(*ctx, jobHandle)) {
1248 run(jobHandle, ctx);
1249 }
1250
1251 detail::tl_workerCtx = nullptr;
1252
1253 // Join threads with the main one
1254 GAIA_FOR(m_workers.size()) join_thread(i + 1);
1255
1256 // All threads have been stopped. Allow new threads to run if necessary.
1257 m_stop.store(false);
1258 }

1271 void signal_edges(JobContainer& jobData) {
1272 const auto max = jobData.edges.depCnt;
1273
1274 // Nothing to do if there are no dependencies
1275 if (max == 0)
1276 return;
1277
1278 auto* ctx = detail::tl_workerCtx;
1279
1280 // One dependency
1281 if (max == 1) {
1282 auto depHandle = jobData.edges.dep;
1283#if GAIA_LOG_JOB_STATES
1284 GAIA_LOG_N("SIGNAL %u.%u -> %u.%u", jobData.idx, jobData.gen, depHandle.id(), depHandle.gen());
1285#endif
1286
1287 // See the conditions can't be satisfied for us to submit the job we skip
1288 auto& depData = m_jobManager.data(depHandle);
1289 if (!JobManager::signal_edge(depData))
1290 return;
1291
1292 // Submit all jobs that can are ready
1293 process(std::span(&depHandle, 1), ctx);
1294 return;
1295 }
1296
1297 // Multiple dependencies. The array has to be set
1298 GAIA_ASSERT(jobData.edges.pDeps != nullptr);
1299
1300 auto* pHandles = (JobHandle*)alloca(sizeof(JobHandle) * max);
1301 uint32_t cnt = 0;
1302 GAIA_FOR(max) {
1303 auto depHandle = jobData.edges.pDeps[i];
1304
1305 // See if all conditions were satisfied for us to submit the job
1306 auto& depData = m_jobManager.data(depHandle);
1307 if (!JobManager::signal_edge(depData))
1308 continue;
1309
1310 pHandles[cnt++] = depHandle;
1311 }
1312
1313 // Submit all jobs that can are ready
1314 process(std::span(pHandles, cnt), ctx);
1315 }
1316
1317 void process(std::span<JobHandle> jobHandles, ThreadCtx* ctx) {
1318 auto* pHandles = (JobHandle*)alloca(sizeof(JobHandle) * jobHandles.size());
1319 uint32_t handlesCnt = 0;
1320
1321 for (auto handle: jobHandles) {
1322 auto& jobData = m_jobManager.data(handle);
1323 m_jobManager.processing(jobData);
1324
1325 // Jobs that have no functor assigned don't need to be enqueued.
1326 // We can "run" them right away. The only time where it makes
1327 // sense to create such a job is to create a sync job. E.g. when you
1328 // need to wait for N jobs, rather than waiting for each of them
1329 // separately you make them a dependency of a dummy/sync job and
1330 // wait just for that one.
1331 if (!jobData.func.operator bool())
1332 (void)run(handle, ctx);
1333 else
1334 pHandles[handlesCnt++] = handle;
1335 }
1336
1337 std::span handles(pHandles, handlesCnt);
1338 while (!handles.empty()) {
1339 // Try pushing all jobs
1340 uint32_t pushed = 0;
1341 if (ctx != nullptr) {
1342 pushed = ctx->jobQueue.try_push(handles);
1343 } else {
1344 for (auto handle: handles) {
1345 if (m_jobQueue[(uint32_t)handle.prio()].try_push(handle))
1346 pushed++;
1347 else
1348 break;
1349 }
1350 }
1351
1352 // Lock the semaphore with the number of jobs me managed to push.
1353 // Number of workers if the upper bound.
1354 const auto cntWorkers = ctx != nullptr ? m_workersCnt[(uint32_t)ctx->prio] : m_workers.size();
1355 const auto cnt = (int32_t)core::get_min(pushed, cntWorkers);
1356 m_sem.release(cnt);
1357
1358 handles = handles.subspan(pushed);
1359 if (!handles.empty()) {
1360 // The queue was full. Execute the job right away.
1361 run(handles[0], ctx);
1362 handles = handles.subspan(1);
1363 }
1364 }
1365 }
1366
1367 bool run(JobHandle jobHandle, ThreadCtx* ctx) {
1368 if (jobHandle == (JobHandle)JobNull_t{})
1369 return false;
1370
1371 auto& jobData = m_jobManager.data(jobHandle);
1372 const bool manualDelete = (jobData.flags & JobCreationFlags::ManualDelete) != 0U;
1373 const bool canWait = (jobData.flags & JobCreationFlags::CanWait) != 0U;
1374
1375 m_jobManager.executing(jobData, ctx->workerIdx);
1376
1377 if (m_blockedInWorkUntil.load() != 0) {
1378 const auto blockedCnt = m_blockedInWorkUntil.exchange(0);
1379 if (blockedCnt != 0)
1380 Futex::wake(&m_blockedInWorkUntil, detail::WaitMaskAll);
1381 }
1382
1383 GAIA_ASSERT(jobData.idx != (uint32_t)-1 && jobData.data.gen != (uint32_t)-1);
1384
1385 // Run the functor associated with the job
1386 m_jobManager.run(jobData);
1387
1388 // Signal the edges and release memory allocated for them if possible
1389 signal_edges(jobData);
1390 JobManager::free_edges(jobData);
1391
1392 // Signal we finished
1393 ctx->event.set();
1394 if (canWait) {
1395 const auto* pFutexValue = &jobData.state;
1396 Futex::wake(pFutexValue, detail::WaitMaskAll);
1397 }
1398
1399 if (!manualDelete)
1400 del(jobHandle);
1401
1402 return true;
1403 }
