319 static bool signal_edge(JobContainer& jobData) {
320 // Subtract from dependency counter
321 const auto state = jobData.state.fetch_sub(1) - 1;
322
323 // If the job is not submitted, we can't accept it
324 const auto s = state & JobState::STATE_BITS_MASK;
325 if (s != JobState::Submitted)
326 return false;
327
328 // If the job still has some dependencies left we can't accept it
329 const auto deps = state & JobState::DEP_BITS_MASK;
330 return deps == 0;
331 }

359 void dep(std::span<JobHandle> jobsFirst, JobHandle jobSecond) {
360 GAIA_ASSERT(!jobsFirst.empty());
361
362 GAIA_PROF_SCOPE(JobManager::dep);
363
364 auto& secondData = data(jobSecond);
365
366#if GAIA_ASSERT_ENABLED
367 GAIA_ASSERT(!busy(const_cast<const JobContainer&>(secondData)));
368 for (auto jobFirst: jobsFirst) {
369 const auto& firstData = data(jobFirst);
370 GAIA_ASSERT(!busy(firstData));
371 }
372#endif
373
374 for (auto jobFirst: jobsFirst)
375 dep_internal(jobFirst, jobSecond);
376
377 // Tell jobSecond that it has new dependencies
378 // secondData.canWait = true;
379 const uint32_t cnt = (uint32_t)jobsFirst.size();
380 [[maybe_unused]] const uint32_t statePrev = secondData.state.fetch_add(cnt);
381 GAIA_ASSERT((statePrev & JobState::DEP_BITS_MASK) < DEP_BITS_MASK - 1);
382 }

390 void dep_refresh(std::span<JobHandle> jobsFirst, JobHandle jobSecond) {
391 GAIA_ASSERT(!jobsFirst.empty());
392
393 GAIA_PROF_SCOPE(JobManager::dep_refresh);
394
395 auto& secondData = data(jobSecond);
396
397#if GAIA_ASSERT_ENABLED
398 GAIA_ASSERT(!busy(const_cast<const JobContainer&>(secondData)));
399 for (auto jobFirst: jobsFirst) {

401 GAIA_ASSERT(!busy(firstData));
402 }
403
404 for (auto jobFirst: jobsFirst)

407
408 // Tell jobSecond that it has new dependencies
409 // secondData.canWait = true;
410 const uint32_t cnt = (uint32_t)jobsFirst.size();
411 [[maybe_unused]] const uint32_t statePrev = secondData.state.fetch_add(cnt);
412 GAIA_ASSERT((statePrev & JobState::DEP_BITS_MASK) < DEP_BITS_MASK - 1);
413 }

418 static uint32_t submit(JobContainer& jobData) {
419 [[maybe_unused]] const auto state = jobData.state.load() & JobState::STATE_BITS_MASK;
420 GAIA_ASSERT(state < JobState::Submitted);
421 const auto val = jobData.state.fetch_add(JobState::Submitted) + (uint32_t)JobState::Submitted;
422#if GAIA_LOG_JOB_STATES
423 GAIA_LOG_N("JobHandle %u.%u - SUBMITTED", jobData.idx, jobData.gen);
424#endif
425 return val;
426 }

538 void dep_internal(JobHandle jobFirst, JobHandle jobSecond) {
539 GAIA_ASSERT(jobFirst != (JobHandle)JobNull_t{});
540 GAIA_ASSERT(jobSecond != (JobHandle)JobNull_t{});
541
542 auto& firstData = data(jobFirst);
543 const auto depCnt0 = firstData.edges.depCnt;
544 const auto depCnt1 = ++firstData.edges.depCnt;
545
546#if GAIA_LOG_JOB_STATES
547 GAIA_LOG_N(
548 "DEP %u.%u, %u -> %u.%u", jobFirst.id(), jobFirst.gen(), firstData.edges.depCnt, jobSecond.id(),
549 jobSecond.gen());
550#endif
551
552 if (depCnt1 <= 1) {
553 firstData.edges.dep = jobSecond;
554 } else if (depCnt1 == 2) {
555 auto prev = firstData.edges.dep;
556 // TODO: Use custom allocator
557 firstData.edges.pDeps = mem::AllocHelper::alloc<JobHandle>(depCnt1);
558 firstData.edges.pDeps[0] = prev;
559 firstData.edges.pDeps[1] = jobSecond;
560 } else {
561 // Reallocate if the previous value was a power of 2
562 const bool isPow2 = core::is_pow2(depCnt0);
563 if (isPow2) {
564 const auto nextPow2 = depCnt0 << 1;
565 auto* pPrev = firstData.edges.pDeps;
566 // TODO: Use custom allocator
567 firstData.edges.pDeps = mem::AllocHelper::alloc<JobHandle>(nextPow2);
568 if (pPrev != nullptr) {
569 GAIA_FOR(depCnt0) firstData.edges.pDeps[i] = pPrev[i];
570 mem::AllocHelper::free(pPrev);
571 }
572 }
573
574 // Append new dependencies
575 firstData.edges.pDeps[depCnt0] = jobSecond;
576 }
577 }

583
