139 static uint32_t soa(std::span<uint8_t, meta::StructToTupleMaxTypes> soaSizes) {
140 if constexpr (mem::is_soa_layout_v<U>) {
141 uint32_t i = 0;
142 using TTuple = decltype(meta::struct_to_tuple(std::declval<U>()));
143 // is_soa_layout_v is always false for empty types so we know there is at least one element in the tuple
144 constexpr auto TTupleSize = std::tuple_size_v<TTuple>;
145 static_assert(TTupleSize > 0);
146 static_assert(TTupleSize <= meta::StructToTupleMaxTypes);
147 core::each_tuple<TTuple>([&](auto&& item) {
148 static_assert(sizeof(item) <= 255, "Each member of a SoA component can be at most 255 B long!");
149 soaSizes[i] = (uint8_t)sizeof(item);
150 ++i;
151 });
152 GAIA_ASSERT(i <= meta::StructToTupleMaxTypes);
153 return i;
154 } else {
155 return 0U;
156 }
157 }

214 static constexpr auto func_cmp() {
215 if constexpr (mem::is_soa_layout_v<U>) {
216 return []([[maybe_unused]] const void* left, [[maybe_unused]] const void* right) {
217 GAIA_ASSERT(false && "func_cmp for SoA not implemented yet");
218 return false;
219 };
220 } else {
221 constexpr bool hasGlobalCmp = core::has_ffunc_equals<U>::value;
222 constexpr bool hasMemberCmp = core::has_func_equals<U>::value;
223 if constexpr (hasGlobalCmp || hasMemberCmp) {
224 return [](const void* left, const void* right) {
225 const auto* l = (const U*)left;
226 const auto* r = (const U*)right;
227 return *l == *r;
228 };
229 } else {
230 // fallback comparison function
231 return [](const void* left, const void* right) {
232 const auto* l = (const U*)left;
233 const auto* r = (const U*)right;
234 return memcmp(l, r, sizeof(U)) == 0;
235 };
236 }
237 }

240 static constexpr auto func_save() {
241 return [](ser::serializer& s, const void* pSrc, uint32_t from, uint32_t to, uint32_t cap) {
242 const auto* pComponent = (const U*)pSrc;
243
244#if GAIA_ASSERT_ENABLED
245 // Check if save and load match. Testing with one item is enough.
246 s.check(*pComponent);
247#endif
248
249 if constexpr (mem::is_soa_layout_v<U>) {
250 auto view = mem::auto_view_policy_get<U>{std::span{(const uint8_t*)pSrc, cap}};
251 GAIA_FOR2(from, to) {
252 auto val = view[i];
253 s.save(val);
254 }
255 } else {
256 pComponent += from;
257 GAIA_FOR2(from, to) {
258 s.save(*pComponent);
259 ++pComponent;
260 }
261 }
262 };
263 }

265 static constexpr auto func_load() {
266 return [](ser::serializer& s, void* pDst, uint32_t from, uint32_t to, uint32_t cap) {
267 if constexpr (mem::is_soa_layout_v<U>) {
268 auto view = mem::auto_view_policy_set<U>{std::span{(uint8_t*)pDst, cap}};
269 GAIA_FOR2(from, to) {
270 U val;
271 s.load(val);
272 view[i] = val;
273 }
274 } else {
275 auto* pComponent = (U*)pDst + from;
276 GAIA_FOR2(from, to) {
277 s.load(*pComponent);
278 ++pComponent;
279 }
280 }
