201 static uint32_t soa(std::span<uint8_t, meta::StructToTupleMaxTypes> soaSizes) {
202 if constexpr (mem::is_soa_layout_v<U>) {
203 uint32_t i = 0;
204 using TTuple = decltype(meta::struct_to_tuple(std::declval<U>()));
205 // is_soa_layout_v is always false for empty types so we know there is at least one element in the tuple
206 constexpr auto TTupleSize = std::tuple_size_v<TTuple>;
207 static_assert(TTupleSize > 0);
208 static_assert(TTupleSize <= meta::StructToTupleMaxTypes);
209 core::each_tuple<TTuple>([&](auto&& item) {
210 static_assert(sizeof(item) <= 255, "Each member of a SoA component can be at most 255 B long!");
211 soaSizes[i] = (uint8_t)sizeof(item);
212 ++i;
213 });
214 GAIA_ASSERT(i <= meta::StructToTupleMaxTypes);
215 return i;
216 } else {
217 return 0U;
218 }
219 }

292 static constexpr auto func_cmp() {
293 if constexpr (mem::is_soa_layout_v<U>) {
294 return []([[maybe_unused]] const void* left, [[maybe_unused]] const void* right) {
295 GAIA_ASSERT(false && "func_cmp for SoA not implemented yet");
296 return false;
297 };
298 } else {
299 constexpr bool hasGlobalCmp = core::has_ffunc_equals<U>::value;
300 constexpr bool hasMemberCmp = core::has_func_equals<U>::value;
301 if constexpr (hasGlobalCmp || hasMemberCmp) {
302 return [](const void* left, const void* right) {
303 const auto* l = (const U*)left;
304 const auto* r = (const U*)right;
305 return *l == *r;
306 };
307 } else {
308 // fallback comparison function
309 return [](const void* left, const void* right) {
310 const auto* l = (const U*)left;
311 const auto* r = (const U*)right;
312 return memcmp(l, r, sizeof(U)) == 0;
313 };
314 }
315 }

320 static constexpr auto func_save() {
321 return [](ser::serializer& s, const void* pSrc, uint32_t from, uint32_t to, uint32_t cap) {
322 const auto* pComponent = (const U*)pSrc;
323
324#if GAIA_ASSERT_ENABLED
325 // Check if save and load match. Testing with one item is enough.
326 s.check(*pComponent);
327#endif
328
329 if constexpr (mem::is_soa_layout_v<U>) {
330 auto view = mem::auto_view_policy_get<U>{std::span{(const uint8_t*)pSrc, cap}};
331 GAIA_FOR2(from, to) {
332 auto val = view[i];
333 s.save(val);
334 }
335 } else {
336 pComponent += from;
337 GAIA_FOR2(from, to) {
338 s.save(*pComponent);
339 ++pComponent;
340 }
341 }
342 };
343 }
344
347 static constexpr auto func_load() {
348 return [](ser::serializer& s, void* pDst, uint32_t from, uint32_t to, uint32_t cap) {
349 if constexpr (mem::is_soa_layout_v<U>) {
350 auto view = mem::auto_view_policy_set<U>{std::span{(uint8_t*)pDst, cap}};
351 GAIA_FOR2(from, to) {
352 U val;
353 s.load(val);
354 view[i] = val;
355 }
356 } else {
357 auto* pComponent = (U*)pDst + from;
358 GAIA_FOR2(from, to) {
359 s.load(*pComponent);
360 ++pComponent;
361 }
362 }
363 };

372 static Entity auto_field_type() noexcept {
373 using FU = core::raw_t<Field>;
374 if constexpr (std::is_same_v<FU, bool>)
375 return Bool;

387 static uint32_t auto_fields(std::span<RuntimeFieldDesc, meta::StructToTupleMaxTypes> fields) {
388 using Raw = core::raw_t<T>;
389 if constexpr (std::is_empty_v<Raw> || mem::is_soa_layout_v<Raw>) {
390 return 0;
391 } else if constexpr (!std::is_class_v<Raw>) {
392 const auto fieldType = auto_field_type<Raw>();
393 if (fieldType == EntityBad)
394 return 0;
395 fields[0] = {util::str_view("value", 5), fieldType, 0, 0};
396 return 1;
397 } else if constexpr (!std::is_aggregate_v<Raw>) {
398 return 0;
399 } else {
400 Raw tmp{};
401 const auto* pBase = reinterpret_cast<const uint8_t*>(&tmp);
402 static constexpr const char* FieldNames[meta::StructToTupleMaxTypes] = {
403 "f0", "f1", "f2", "f3", "f4", "f5", "f6", "f7", "f8", "f9", "f10", "f11", "f12", "f13", "f14"};
404 uint32_t fieldCount = 0;
405 meta::each_member(tmp, [&](auto&... members) {
406 auto add_field = [&](auto& member) {
407 using Field = core::raw_t<decltype(member)>;
408 const auto fieldType = auto_field_type<Field>();
409 if (fieldType == EntityBad)
410 return;
411 GAIA_ASSERT(fieldCount < meta::StructToTupleMaxTypes);
412 const auto* pField = reinterpret_cast<const uint8_t*>(&member);
413 const auto offset = (uint32_t)(pField - pBase);
414 const auto* fieldName = FieldNames[fieldCount];
415 const auto fieldNameLen = (uint32_t)GAIA_STRLEN(fieldName, 4);
416 fields[fieldCount] = {util::str_view(fieldName, fieldNameLen), fieldType, offset, 0};
417 ++fieldCount;
418 };
419 (add_field(members), ...);
420 });
421 return fieldCount;
422 }
423 }
