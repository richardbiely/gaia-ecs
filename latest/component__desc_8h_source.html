148 static uint32_t soa(std::span<uint8_t, meta::StructToTupleMaxTypes> soaSizes) {
149 if constexpr (mem::is_soa_layout_v<U>) {
150 uint32_t i = 0;
151 using TTuple = decltype(meta::struct_to_tuple(std::declval<U>()));
152 // is_soa_layout_v is always false for empty types so we know there is at least one element in the tuple
153 constexpr auto TTupleSize = std::tuple_size_v<TTuple>;
154 static_assert(TTupleSize > 0);
155 static_assert(TTupleSize <= meta::StructToTupleMaxTypes);
156 core::each_tuple<TTuple>([&](auto&& item) {
157 static_assert(sizeof(item) <= 255, "Each member of a SoA component can be at most 255 B long!");
158 soaSizes[i] = (uint8_t)sizeof(item);
159 ++i;
160 });
161 GAIA_ASSERT(i <= meta::StructToTupleMaxTypes);
162 return i;
163 } else {
164 return 0U;
165 }

223 static constexpr auto func_cmp() {
224 if constexpr (mem::is_soa_layout_v<U>) {
225 return []([[maybe_unused]] const void* left, [[maybe_unused]] const void* right) {
226 GAIA_ASSERT(false && "func_cmp for SoA not implemented yet");
227 return false;
228 };
229 } else {
230 constexpr bool hasGlobalCmp = core::has_ffunc_equals<U>::value;
231 constexpr bool hasMemberCmp = core::has_func_equals<U>::value;
232 if constexpr (hasGlobalCmp || hasMemberCmp) {
233 return [](const void* left, const void* right) {
234 const auto* l = (const U*)left;
235 const auto* r = (const U*)right;
236 return *l == *r;
237 };
238 } else {
239 // fallback comparison function
240 return [](const void* left, const void* right) {
241 const auto* l = (const U*)left;
242 const auto* r = (const U*)right;
243 return memcmp(l, r, sizeof(U)) == 0;
244 };
245 }
246 }

249 static constexpr auto func_save() {
250 return [](ser::serializer& s, const void* pSrc, uint32_t from, uint32_t to, uint32_t cap) {
251 const auto* pComponent = (const U*)pSrc;
252
253#if GAIA_ASSERT_ENABLED
254 // Check if save and load match. Testing with one item is enough.
255 s.check(*pComponent);
256#endif
257
258 if constexpr (mem::is_soa_layout_v<U>) {
259 auto view = mem::auto_view_policy_get<U>{std::span{(const uint8_t*)pSrc, cap}};
260 GAIA_FOR2(from, to) {
261 auto val = view[i];
262 s.save(val);
263 }
264 } else {
265 pComponent += from;
266 GAIA_FOR2(from, to) {
267 s.save(*pComponent);
268 ++pComponent;
269 }
270 }
271 };
272 }

274 static constexpr auto func_load() {
275 return [](ser::serializer& s, void* pDst, uint32_t from, uint32_t to, uint32_t cap) {
276 if constexpr (mem::is_soa_layout_v<U>) {
277 auto view = mem::auto_view_policy_set<U>{std::span{(uint8_t*)pDst, cap}};
278 GAIA_FOR2(from, to) {
279 U val;
280 s.load(val);
281 view[i] = val;
282 }
283 } else {
284 auto* pComponent = (U*)pDst + from;
285 GAIA_FOR2(from, to) {
286 s.load(*pComponent);
287 ++pComponent;
288 }
289 }

297 template <typename Field>
298 static Entity auto_field_type() noexcept {
299 using FU = core::raw_t<Field>;
300 if constexpr (std::is_same_v<FU, bool>)
301 return Bool;
302 else if constexpr (std::is_same_v<FU, char>)
303 return Char8;
304 else if constexpr (std::is_arithmetic_v<FU>)
305 return runtime_primitive_type_entity(ser::type_id<FU>());
306 else
307 return EntityBad;

313 static uint32_t auto_fields(std::span<RuntimeFieldDesc, meta::StructToTupleMaxTypes> fields) {
314 using Raw = core::raw_t<T>;
315 if constexpr (std::is_empty_v<Raw> || mem::is_soa_layout_v<Raw>) {
316 return 0;
317 } else if constexpr (!std::is_class_v<Raw>) {
318 const auto fieldType = auto_field_type<Raw>();
319 if (fieldType == EntityBad)
320 return 0;
321 fields[0] = {util::str_view("value", 5), fieldType, 0, 0};
322 return 1;
323 } else if constexpr (!std::is_aggregate_v<Raw>) {
324 return 0;
325 } else {
326 Raw tmp{};
327 const auto* pBase = reinterpret_cast<const uint8_t*>(&tmp);
328 static constexpr const char* FieldNames[meta::StructToTupleMaxTypes] = {
329 "f0", "f1", "f2", "f3", "f4", "f5", "f6", "f7", "f8", "f9", "f10", "f11", "f12", "f13", "f14"};
330 uint32_t fieldCount = 0;
331 meta::each_member(tmp, [&](auto&... members) {
332 auto add_field = [&](auto& member) {
333 using Field = core::raw_t<decltype(member)>;
334 const auto fieldType = auto_field_type<Field>();
335 if (fieldType == EntityBad)
336 return;
337 GAIA_ASSERT(fieldCount < meta::StructToTupleMaxTypes);
338 const auto* pField = reinterpret_cast<const uint8_t*>(&member);
339 const auto offset = (uint32_t)(pField - pBase);
340 const auto* fieldName = FieldNames[fieldCount];
341 const auto fieldNameLen = (uint32_t)GAIA_STRLEN(fieldName, 4);
342 fields[fieldCount] = {util::str_view(fieldName, fieldNameLen), fieldType, offset, 0};
343 ++fieldCount;
344 };
345 (add_field(members), ...);
346 });
347 return fieldCount;
348 }
349 }
