283 static uint32_t soa(std::span<uint8_t, meta::StructToTupleMaxTypes> soaSizes) {
284 if constexpr (mem::is_soa_layout_v<U>) {
285 uint32_t i = 0;
286 using TTuple = decltype(meta::struct_to_tuple(std::declval<U>()));
287 // is_soa_layout_v is always false for empty types so we know there is at least one element in the tuple
288 constexpr auto TTupleSize = std::tuple_size_v<TTuple>;
289 static_assert(TTupleSize > 0);
290 static_assert(TTupleSize <= meta::StructToTupleMaxTypes);
291 core::each_tuple<TTuple>([&](auto&& item) {
292 static_assert(sizeof(item) <= 255, "Each member of a SoA component can be at most 255 B long!");
293 soaSizes[i] = (uint8_t)sizeof(item);
294 ++i;
295 });
296 GAIA_ASSERT(i <= meta::StructToTupleMaxTypes);
297 return i;
298 } else {
299 return 0U;
300 }
301 }

374 static constexpr auto func_cmp() {
375 if constexpr (mem::is_soa_layout_v<U>) {
376 return []([[maybe_unused]] const void* left, [[maybe_unused]] const void* right) {
377 GAIA_ASSERT(false && "func_cmp for SoA not implemented yet");
378 return false;
379 };
380 } else {
381 constexpr bool hasGlobalCmp = core::has_ffunc_equals<U>::value;
382 constexpr bool hasMemberCmp = core::has_func_equals<U>::value;
383 if constexpr (hasGlobalCmp || hasMemberCmp) {
384 return [](const void* left, const void* right) {
385 const auto* l = (const U*)left;
386 const auto* r = (const U*)right;
387 return *l == *r;
388 };
389 } else {
390 // fallback comparison function
391 return [](const void* left, const void* right) {
392 const auto* l = (const U*)left;
393 const auto* r = (const U*)right;
394 return memcmp(l, r, sizeof(U)) == 0;
395 };
396 }
397 }

402 static constexpr auto func_save() {
403 return [](ser::serializer& s, const void* pSrc, uint32_t from, uint32_t to, uint32_t cap) {
404 const auto* pComponent = (const U*)pSrc;
405
406#if GAIA_ASSERT_ENABLED
407 // Check if save and load match. Testing with one item is enough.
408 s.check(*pComponent);
409#endif
410
411 if constexpr (mem::is_soa_layout_v<U>) {
412 auto view = mem::auto_view_policy_get<U>{std::span{(const uint8_t*)pSrc, cap}};
413 GAIA_FOR2(from, to) {
414 auto val = view[i];
415 s.save(val);
416 }
417 } else {
418 pComponent += from;
419 GAIA_FOR2(from, to) {
420 s.save(*pComponent);
421 ++pComponent;
422 }
423 }
424 };
425 }
426
429 static constexpr auto func_load() {
430 return [](ser::serializer& s, void* pDst, uint32_t from, uint32_t to, uint32_t cap) {
431 if constexpr (mem::is_soa_layout_v<U>) {
432 auto view = mem::auto_view_policy_set<U>{std::span{(uint8_t*)pDst, cap}};
433 GAIA_FOR2(from, to) {
434 U val;
435 s.load(val);
436 view[i] = val;
437 }
438 } else {
439 auto* pComponent = (U*)pDst + from;
440 GAIA_FOR2(from, to) {
441 s.load(*pComponent);
442 ++pComponent;
443 }
444 }
445 };

453 template <typename Field>
454 static Entity auto_field_type() noexcept {
455 using FU = core::raw_t<Field>;
456 if constexpr (std::is_same_v<FU, bool>)
457 return Bool;
458 else if constexpr (std::is_same_v<FU, char>)
459 return Char8;

461 return runtime_primitive_type_entity(ser::type_id<FU>());
462 else
463 return EntityBad;
464 }

469 static uint32_t auto_fields(std::span<RuntimeFieldDesc, meta::StructToTupleMaxTypes> fields) {
470 using Raw = core::raw_t<T>;
471 if constexpr (std::is_empty_v<Raw> || mem::is_soa_layout_v<Raw>) {
472 return 0;
473 } else if constexpr (!std::is_class_v<Raw>) {
474 const auto fieldType = auto_field_type<Raw>();
475 if (fieldType == EntityBad)
476 return 0;
477 fields[0] = {util::str_view("value", 5), fieldType, 0, 0};
478 return 1;
479 } else if constexpr (!std::is_aggregate_v<Raw>) {
480 return 0;
481 } else {
482 Raw tmp{};
483 const auto* pBase = reinterpret_cast<const uint8_t*>(&tmp);
484 static constexpr const char* FieldNames[meta::StructToTupleMaxTypes] = {
485 "f0", "f1", "f2", "f3", "f4", "f5", "f6", "f7", "f8", "f9", "f10", "f11", "f12", "f13", "f14"};
486 uint32_t fieldCount = 0;
487 meta::each_member(tmp, [&](auto&... members) {
488 auto add_field = [&](auto& member) {
489 using Field = core::raw_t<decltype(member)>;
490 const auto fieldType = auto_field_type<Field>();
491 if (fieldType == EntityBad)
492 return;
493 GAIA_ASSERT(fieldCount < meta::StructToTupleMaxTypes);
494 const auto* pField = reinterpret_cast<const uint8_t*>(&member);
495 const auto offset = (uint32_t)(pField - pBase);
496 const auto* fieldName = FieldNames[fieldCount];
497 const auto fieldNameLen = (uint32_t)GAIA_STRLEN(fieldName, 4);
498 fields[fieldCount] = {util::str_view(fieldName, fieldNameLen), fieldType, offset, 0};
499 ++fieldCount;
500 };
501 (add_field(members), ...);
502 });
503 return fieldCount;
504 }
505 }
