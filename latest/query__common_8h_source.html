399 constexpr bool query_term_less_for_lookup(const QueryTerm& lhs, const QueryTerm& rhs) {
400 if (lhs.op != rhs.op)
401 return lhs.op < rhs.op;
402
403 if (lhs.id != rhs.id)
404 return SortComponentCond()(lhs.id, rhs.id);
405
406 if (lhs.src != rhs.src)
407 return SortComponentCond()(lhs.src, rhs.src);
408
409 if (lhs.entTrav != rhs.entTrav)
410 return SortComponentCond()(lhs.entTrav, rhs.entTrav);
411
412 if (lhs.travKind != rhs.travKind)
413 return (uint8_t)lhs.travKind < (uint8_t)rhs.travKind;
414
415 if (lhs.travDepth != rhs.travDepth)
416 return lhs.travDepth < rhs.travDepth;
417
418 return (uint8_t)lhs.matchKind < (uint8_t)rhs.matchKind;
419 }

421 inline void canonicalize_lookup_terms(std::span<QueryTerm> terms) {
422 const auto idsCnt = (uint32_t)terms.size();
423 if (idsCnt > 0) {
424 uint32_t orCnt = 0;
425 uint32_t orIdx = BadIndex;
426 GAIA_FOR(idsCnt) {
427 if (terms[i].op != QueryOpKind::Or)
428 continue;
429 ++orCnt;
430 orIdx = i;
431 if (orCnt > 1)
432 break;
433 }
434
435 if (orCnt == 1)
436 terms[orIdx].op = QueryOpKind::All;
437 }
438
439 core::sort(terms.data(), terms.data() + idsCnt, [](const QueryTerm& left, const QueryTerm& right) {
440 return query_term_less_for_lookup(left, right);
441 });
442 }

1020 else

1037 ? CreateArchetypeMatchKind::DirectStructuralTerms

1069 canDirectEntitySeedEvalShape_old != data.canDirectEntitySeedEvalShape ||

1163 inline void sort(QueryCtx& ctx) {
1164 const uint32_t idsCnt = ctx.data.idsCnt;
1165 const uint32_t changedCnt = ctx.data.changedCnt;
1166
1167 auto& ctxData = ctx.data;
1168 // Canonicalize degenerate OR queries: a single OR term has AND semantics.
1169 // Rewriting it here keeps ordering/hash behavior identical to an explicit ALL term.
1170 if (idsCnt > 0) {
1171 uint32_t orCnt = 0;
1172 uint32_t orIdx = BadIndex;
1173 GAIA_FOR(idsCnt) {
1174 if (ctxData.terms[i].op != QueryOpKind::Or)
1175 continue;
1176 ++orCnt;
1177 orIdx = i;
1178 if (orCnt > 1)
1179 break;
1180 }
1181
1182 if (orCnt == 1)
1183 ctxData.terms[orIdx].op = QueryOpKind::All;
1184 }
1185
1186 // Sort data. Necessary for correct hash calculation.
1187 // Without sorting query.all<XXX, YYY> would be different than query.all<YYY, XXX>.
1188 // Also makes sure data is in optimal order for query processing.
1189 core::sort(
1190 ctxData.terms.data(), ctxData.terms.data() + ctxData.idsCnt, query_sort_cond{}, //
1191 [&](uint32_t left, uint32_t right) {
1192 core::swap(ctxData.ids[left], ctxData.ids[right]);
1193 core::swap(ctxData.terms[left], ctxData.terms[right]);
1194
1195 // Make sure masks remains correct after sorting
1196 core::swap_bits(ctxData.readWriteMask, left, right);
1197 core::swap_bits(ctxData.as_mask_0, left, right);
1198 core::swap_bits(ctxData.as_mask_1, left, right);
1199 });
1200
1201 if (idsCnt > 0) {
1202 uint32_t i = 0;
1203 while (i < idsCnt && ctxData.terms[i].op == QueryOpKind::All)
1204 ++i;
1205 ctxData.firstOr = (uint8_t)i;
1206 while (i < idsCnt && ctxData.terms[i].op == QueryOpKind::Or)
1207 ++i;
1208 ctxData.firstNot = (uint8_t)i;
1209 while (i < idsCnt && ctxData.terms[i].op == QueryOpKind::Not)
1210 ++i;
1211 ctxData.firstAny = (uint8_t)i;
1212 } else
1213 ctxData.firstOr = ctxData.firstNot = ctxData.firstAny = 0;
1214
1215 // Canonicalize filter order. This enables monotonic component lookup in filter matching
1216 // and keeps cache keys stable regardless of changed() call order.
1217 if (changedCnt > 1) {
1218 core::sort(ctxData.changed.data(), ctxData.changed.data() + changedCnt, SortComponentCond{});
1219 }
1220 }

1242 inline void calc_lookup_hash(QueryCtx& ctx) {
1243 GAIA_ASSERT(ctx.cc != nullptr);
1244 // Make sure we don't calculate the hash twice
1245 GAIA_ASSERT(ctx.hashLookup.hash == 0);
1246
1247 QueryLookupHash::Type hashLookup = 0;
1248
1249 const auto& ctxData = ctx.data;
1250
1251 // Ids & ops
1252 {
1253 QueryLookupHash::Type hash = 0;
1254
1255 for (uint32_t i = 0; i < ctxData.idsCnt; ++i) {
1256 const auto& pair = ctxData.lookupTerms[i];
1257 hash = core::hash_combine(hash, (QueryLookupHash::Type)pair.op);
1258 hash = core::hash_combine(hash, (QueryLookupHash::Type)pair.id.value());
1259 hash = core::hash_combine(hash, (QueryLookupHash::Type)pair.src.value());
1260 hash = core::hash_combine(hash, (QueryLookupHash::Type)pair.entTrav.value());
1261 hash = core::hash_combine(hash, (QueryLookupHash::Type)(uint8_t)pair.travKind);
1262 hash = core::hash_combine(hash, (QueryLookupHash::Type)pair.travDepth);
1263 hash = core::hash_combine(hash, (QueryLookupHash::Type)(uint8_t)pair.matchKind);
1264 }
1265 hash = core::hash_combine(hash, (QueryLookupHash::Type)ctxData.idsCnt);
1266 hash = core::hash_combine(hash, (QueryLookupHash::Type)ctxData.readWriteMask);
1267 hash = core::hash_combine(hash, (QueryLookupHash::Type)ctxData.cacheSrcTrav);
1268
1269 const bool matchPrefab = (ctxData.flags & QueryCtx::QueryFlags::MatchPrefab) != 0;
1270 hash = core::hash_combine(hash, (QueryLookupHash::Type)matchPrefab);
1271
1272 hashLookup = hash;
1273 }
1274
1275 // Filters
1276 {
1277 QueryLookupHash::Type hash = 0;
1278
1279 for (uint32_t i = 0; i < ctxData.changedCnt; ++i)
1280 hash = core::hash_combine(hash, (QueryLookupHash::Type)ctxData.changedLookup[i].value());
1281 hash = core::hash_combine(hash, (QueryLookupHash::Type)ctxData.changedCnt);
1282
1283 hashLookup = core::hash_combine(hashLookup, hash);
1284 }
1285
1286 // Explicit grouping dependencies
1287 {
1288 QueryLookupHash::Type hash = 0;
1289
1290 for (uint32_t i = 0; i < ctxData.groupDepCnt; ++i)
1291 hash = core::hash_combine(hash, (QueryLookupHash::Type)ctxData.groupDepsLookup[i].value());
1292 hash = core::hash_combine(hash, (QueryLookupHash::Type)ctxData.groupDepCnt);
1293
1294 hashLookup = core::hash_combine(hashLookup, hash);
1295 }
1296
1297 // Grouping
1298 {
1299 QueryLookupHash::Type hash = 0;
1300
1301 hash = core::hash_combine(hash, (QueryLookupHash::Type)ctxData.groupBy.value());
1302 hash = core::hash_combine(hash, (QueryLookupHash::Type)ctxData.groupByFunc);
1303
1304 hashLookup = core::hash_combine(hashLookup, hash);
1305 }
1306
1307 ctx.hashLookup = {core::calculate_hash64(hashLookup)};
1308 }

1316 template <uint32_t MAX_COMPONENTS>
1317 GAIA_NODISCARD inline uint32_t comp_idx(const QueryTerm* pTerms, Entity entity, Entity src) {
1318 // We let the compiler know the upper iteration bound at compile-time.
1319 // This way it can optimize better (e.g. loop unrolling, vectorization).
