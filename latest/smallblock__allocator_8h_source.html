123 void write_block_idx(uint32_t blockIdx, uint32_t value) {
124 const uint32_t bitPosition = blockIdx * NBlocks_Bits;
125
126 GAIA_ASSERT(bitPosition < NBlocks * NBlocks_Bits);
127 GAIA_ASSERT(value <= InvalidBlockId);
128
129 BitView{{(uint8_t*)m_blocks.data(), BlockArrayBytes}}.set(bitPosition, (uint8_t)value);
130 }

132 GAIA_NODISCARD uint8_t read_block_idx(uint32_t blockIdx) const {
133 const uint32_t bitPosition = blockIdx * NBlocks_Bits;
134
135 GAIA_ASSERT(bitPosition < NBlocks * NBlocks_Bits);
136
137 return BitView{{(uint8_t*)m_blocks.data(), BlockArrayBytes}}.get(bitPosition);

140#if GAIA_DEBUG
144 GAIA_NODISCARD void* alloc_block(uint32_t bytesWanted){
145#else
148 GAIA_NODISCARD void* alloc_block() {
149#endif
150 auto store_block_address = [&](uint32_t index) {
151 auto* pMemoryBlock = (uint8_t*)m_data + (index * block_stride());
152 GAIA_ASSERT((uintptr_t)pMemoryBlock % SmallBlockAlignment == 0);
153 auto& header = block_header(pMemoryBlock);
154 header.m_pageAddr = (uintptr_t)this;
155#if GAIA_DEBUG
156 header.m_requestedBytes = bytesWanted;
157 header.m_reserved = 0;
158#else
159 header.m_reserved = 0;
160#endif
161 auto* pData = pMemoryBlock + SmallBlockUsableOffset;
162 GAIA_ASSERT((uintptr_t)pData % SmallBlockAlignment == 0);
163 return (void*)pData;
164 };
165
166 GAIA_ASSERT(!full() && "Trying to allocate too many blocks!");
167
168 uint32_t index = 0;
169 if (m_freeBlocks == 0U) {
170 index = m_blockCnt;
171 ++m_usedBlocks;
172 ++m_blockCnt;
173 write_block_idx(index, index);
174 } else {
175 GAIA_ASSERT(m_nextFreeBlock < m_blockCnt && "Block allocator recycle list broken!");
176
177 ++m_usedBlocks;
178 --m_freeBlocks;
179
180 index = m_nextFreeBlock;
181 m_nextFreeBlock = read_block_idx(m_nextFreeBlock);
182 }
183
184#if GAIA_ASSERT_ENABLED
185 GAIA_ASSERT((m_usedMask & (uint64_t(1) << index)) == 0 && "Block already marked as live");
186 m_usedMask |= uint64_t(1) << index;
187#endif
188
189 return store_block_address(index);
190 }
191
194 void free_block(void* pBlock) {
195 GAIA_ASSERT(pBlock != nullptr);
196 GAIA_ASSERT(m_usedBlocks > 0);
197 GAIA_ASSERT(m_freeBlocks <= NBlocks);
198
199 const auto* pMemoryBlock = (uint8_t*)pBlock - SmallBlockUsableOffset;
200 const auto blckAddr = (uintptr_t)pMemoryBlock;
201 GAIA_ASSERT(blckAddr % SmallBlockAlignment == 0);
202 const auto dataAddr = (uintptr_t)m_data;
203 const auto blockStride = (uintptr_t)block_stride();
204 const auto pageSize = blockStride * NBlocks;
205 GAIA_ASSERT(blckAddr >= dataAddr);
206 GAIA_ASSERT(blckAddr < dataAddr + pageSize);
207 GAIA_ASSERT((blckAddr - dataAddr) % blockStride == 0);
208 const auto blockIdx = (uint32_t)((blckAddr - dataAddr) / blockStride);
209 GAIA_ASSERT(blockIdx < m_blockCnt);
210
211#if GAIA_DEBUG
212 auto& header = block_header((void*)pMemoryBlock);
213 GAIA_ASSERT(header.m_requestedBytes > 0);
214#endif
215#if GAIA_ASSERT_ENABLED
216 GAIA_ASSERT((m_usedMask & (uint64_t(1) << blockIdx)) != 0 && "Double free or corrupted block state");
217 m_usedMask &= ~(uint64_t(1) << blockIdx);
218#endif
219
220#if GAIA_DEBUG
221 header.m_requestedBytes = 0;
222 std::memset(pBlock, FreedBlockPattern, small_block_size(m_sizeType));
223#endif
224
225 if (m_freeBlocks == 0U)
226 write_block_idx(blockIdx, InvalidBlockId);
227 else
228 write_block_idx(blockIdx, m_nextFreeBlock);
229 m_nextFreeBlock = blockIdx;
230
231 ++m_freeBlocks;
232 --m_usedBlocks;
233 }

251 void verify() const {
252#if GAIA_ASSERT_ENABLED
253 GAIA_ASSERT(m_sizeType < SmallBlockSizeTypeCount);
254 GAIA_ASSERT(m_blockCnt <= NBlocks);
255 GAIA_ASSERT(m_usedBlocks <= m_blockCnt);
256 GAIA_ASSERT(m_freeBlocks <= m_blockCnt);
257 GAIA_ASSERT(m_usedBlocks + m_freeBlocks == m_blockCnt);
258 GAIA_ASSERT(((uintptr_t)m_data % SmallBlockAlignment) == 0);
259
260 [[maybe_unused]] uint64_t freeMask = 0;
261
262 if (m_freeBlocks != 0) {
263 uint32_t next = m_nextFreeBlock;
264 GAIA_FOR(m_freeBlocks) {
265 GAIA_ASSERT(next < m_blockCnt);
266 #if GAIA_DEBUG
267 const auto bit = uint64_t(1) << next;
268 GAIA_ASSERT((freeMask & bit) == 0 && "Free list contains a cycle");
269 freeMask |= bit;
270 #endif
271 next = read_block_idx(next);
272 }
273
274 GAIA_ASSERT(next == InvalidBlockId);
275 }
276
277 GAIA_FOR(m_blockCnt) {
278 const auto* pMemoryBlock = (const uint8_t*)m_data + (i * block_stride());
279 const auto& header = block_header(pMemoryBlock);
280 GAIA_ASSERT(header.m_pageAddr == (uintptr_t)this);
281 GAIA_ASSERT(((uintptr_t)pMemoryBlock % SmallBlockAlignment) == 0);
282
283 #if GAIA_DEBUG
284 const bool isFree = (freeMask & (uint64_t(1) << i)) != 0;
285 GAIA_ASSERT((header.m_requestedBytes == 0) == isFree);
286 #endif
287 }
288
289 #if GAIA_DEBUG
290 GAIA_ASSERT((m_usedMask & freeMask) == 0);
291 const auto liveMask = m_blockCnt == 64 ? ~uint64_t(0) : ((uint64_t(1) << m_blockCnt) - 1);
292 GAIA_ASSERT((m_usedMask | freeMask) == liveMask);
293 #endif
294#endif
295 }
296
297#if GAIA_DEBUG
299 GAIA_NODISCARD uint64_t requested_bytes() const {
300 if (m_usedBlocks == 0)
301 return 0;
302
303 uint64_t freeMask = 0;
304 uint32_t next = m_nextFreeBlock;
305 GAIA_FOR(m_freeBlocks) {
306 GAIA_ASSERT(next < m_blockCnt);
307 const auto bit = uint64_t(1) << next;
308 GAIA_ASSERT((freeMask & bit) == 0 && "Free list contains a cycle");
309 freeMask |= bit;
310 next = read_block_idx(next);
311 }
312
313 uint64_t requested = 0;
314 GAIA_FOR(m_blockCnt) {
315 if ((freeMask & (uint64_t(1) << i)) != 0)
316 continue;
317
318 const auto* pMemoryBlock = (const uint8_t*)m_data + (i * block_stride());
319 requested += block_header(pMemoryBlock).m_requestedBytes;
320 }
321
322 return requested;
323 }
324#endif

380 GAIA_NODISCARD void* alloc(uint32_t bytesWanted) {
381 GAIA_ASSERT(bytesWanted > 0);
382 GAIA_ASSERT(bytesWanted <= MAX_SIZE);
383 if (bytesWanted == 0 || bytesWanted > MAX_SIZE)
384 return nullptr;
385
386 const auto sizeType = small_block_size_type(bytesWanted);
387 auto& container = m_pages[sizeType];
388
389 SmallBlockPageState prevState = SmallBlockPageState::Partial;
390 auto* pPage = container.pagesPartial.first;
391 if (pPage == nullptr) {
392 prevState = SmallBlockPageState::Empty;
393 pPage = container.pagesEmpty.first;
394 if (pPage == nullptr) {
395 prevState = SmallBlockPageState::Detached;
396 pPage = alloc_page(sizeType);
397 }
398 }
399
400#if GAIA_DEBUG
401 void* pBlock = pPage->alloc_block(bytesWanted);
402#else
403 void* pBlock = pPage->alloc_block();
404#endif
405 GAIA_PROF_ALLOC(pBlock, bytesWanted);
406 move_page(container, pPage, prevState, state_for(*pPage));
407 verify();
408 return pBlock;
409 }
410
413 void free(void* pBlock) {
414 GAIA_ASSERT(pBlock != nullptr);
415 if (pBlock == nullptr)
416 return;
417
418 const auto& header = *(const SmallBlockHeader*)((uint8_t*)pBlock - SmallBlockUsableOffset);
419 const auto pageAddr = header.m_pageAddr;
420 GAIA_ASSERT(pageAddr % sizeof(uintptr_t) == 0);
421#if GAIA_DEBUG
422 GAIA_ASSERT(header.m_requestedBytes > 0);
423#endif
424 auto* pPage = (SmallBlockPage*)pageAddr;
425 const auto prevState = state_for(*pPage);
426 auto& container = m_pages[pPage->m_sizeType];
427
428 GAIA_PROF_FREE(pBlock);
429 pPage->free_block(pBlock);
430 move_page(container, pPage, prevState, state_for(*pPage));
431 verify();
432
433 if (m_isDone) {
434 if (pPage->empty()) {
435 container.pagesEmpty.unlink(pPage);
436 free_page(pPage);
437 }
438
439 try_delete_this();
440 }
441 }

555 static void move_page(

590 GAIA_NODISCARD SmallBlockAllocatorPageStats page_stats(uint32_t sizeType) const {
591 SmallBlockAllocatorPageStats stats{};
592 const auto& container = m_pages[sizeType];
593 const auto blockStride = (uint64_t)small_block_stride(sizeType);
594 const auto pageSize = blockStride * SmallBlockPage::NBlocks;
595
596 stats.num_pages = (uint32_t)container.pagesEmpty.size() + (uint32_t)container.pagesPartial.size() +
597 (uint32_t)container.pagesFull.size();
598 stats.num_pages_free = (uint32_t)container.pagesEmpty.size() + (uint32_t)container.pagesPartial.size();
599 stats.mem_total = stats.num_pages * pageSize;
600 stats.mem_used = container.pagesFull.size() * pageSize;
601
602#if GAIA_DEBUG
603 stats.num_pages_empty = (uint32_t)container.pagesEmpty.size();
604
605 for (const auto& page: container.pagesFull)
606 stats.mem_requested += page.requested_bytes();
607
608 for (const auto& page: container.pagesPartial) {
609 stats.mem_used += page.used_blocks_cnt() * blockStride;
610 stats.mem_requested += page.requested_bytes();
611 }
612#else
613 for (const auto& page: container.pagesPartial)
614 stats.mem_used += page.used_blocks_cnt() * blockStride;
615#endif
616
617 return stats;
618 }

620 static void flush_pages(SmallBlockPageContainer& container, bool releaseAll) {
621 const bool keepWarmPage = !releaseAll && warm_pages_to_keep() != 0;
622 bool keptWarmPage = false;
623
624 for (auto it = container.pagesEmpty.begin(); it != container.pagesEmpty.end();) {
625 auto* pPage = &(*it);
626 ++it;
627
628 if (!pPage->empty())
629 continue;
630
631 if (keepWarmPage && !keptWarmPage) {
632 keptWarmPage = true;
633 continue;
634 }
635
636 container.pagesEmpty.unlink(pPage);
637 free_page(pPage);
638 }
