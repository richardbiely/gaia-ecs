99 void remove_target_source(Entity target, Entity source) {

101 if (target.id() >= tgtToSrc.size())

104 auto& sources = tgtToSrc[target.id()];

106 GAIA_ASSERT(idx != BadIndex && idx < sources.size());

111 if (idx != lastIdx) {
