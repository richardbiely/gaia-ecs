109 template <typename T, span_size_type Extent>
110 class span {
111 static_assert(
112 std::is_object<T>::value, "A span's T must be an object type (not a "
113 "reference type or void)");
114 static_assert(
115 detail::is_complete<T>::value, "A span's T must be a complete type (not a forward "
116 "declaration)");
117 static_assert(!std::is_abstract<T>::value, "A span's T cannot be an abstract class type");
118
119 using m_datatype = detail::span_storage<T, Extent>;
120
121 public:
123 using element_kind = T;
125 using value_type = typename std::remove_cv<T>::type;
127 using size_type = span_size_type;
129 using difference_type = span_diff_type;
131 using pointer = element_kind*;
133 using const_pointer = const element_kind*;
135 using reference = element_kind&;
137 using const_reference = const element_kind&;
138
140 using iterator = pointer;
142 using const_iterator = const_pointer;
144 using iterator_type = core::random_access_iterator_tag;
145
147 static constexpr size_type extent = Extent;
148
149 private:
150 m_datatype m_data{};
151
152 public:
155 template <span_size_type E = Extent, typename std::enable_if<(E == DynamicSpanExtent || E <= 0), int>::type = 0>
156 constexpr span() noexcept {}
157
161 constexpr span(pointer ptr, size_type count): m_data(ptr, count) {
162 GAIA_ASSERT(extent == DynamicSpanExtent || extent == count);
163 }
164
168 constexpr span(pointer begin, pointer end): m_data(begin, end - begin) {
169 GAIA_ASSERT(extent == DynamicSpanExtent || ((uintptr_t)(end - begin) == (uintptr_t)extent));
170 }
171
176 template <
177 span_size_type N, span_size_type E = Extent,
178 typename std::enable_if<
179 (E == DynamicSpanExtent || N == E) &&
180 detail::is_container_element_kind_compatible<element_kind (&)[N], T>::value,
181 int>::type = 0>
182 constexpr span(element_kind (&arr)[N]) noexcept: m_data(arr, N) {}
183
188 template <
189 typename Container, span_size_type E = Extent,
190 typename std::enable_if<
191 E == DynamicSpanExtent && detail::is_container<Container>::value &&
192 detail::is_container_element_kind_compatible<Container&, T>::value,
193 int>::type = 0>
194 constexpr span(Container& cont): m_data(detail::data(cont), detail::size(cont)) {}
195
200 template <
201 typename Container, span_size_type E = Extent,
202 typename std::enable_if<
203 E == DynamicSpanExtent && detail::is_container<Container>::value &&
204 detail::is_container_element_kind_compatible<const Container&, T>::value,
205 int>::type = 0>
206 constexpr span(const Container& cont): m_data(detail::data(cont), detail::size(cont)) {}
207
208 constexpr span(const span&) noexcept = default;
209 constexpr span& operator=(const span&) noexcept = default;

215 template <
216 typename T2, span_size_type Extent2,
217 typename std::enable_if<
218 (Extent == DynamicSpanExtent || Extent2 == DynamicSpanExtent || Extent == Extent2) &&
219 std::is_convertible<T2 (*)[], T (*)[]>::value,
220 int>::type = 0>
221 constexpr span(const span<T2, Extent2>& other) noexcept: m_data(other.data(), other.size()) {}

346 template <span_size_type Offset, span_size_type Count = DynamicSpanExtent>
347 GAIA_NODISCARD constexpr subspan_return_t<Offset, Count> subspan() const {
348 GAIA_ASSERT(Offset <= size() && (Count == DynamicSpanExtent || Offset + Count <= size()));
349 return {m_data.beg + Offset, Count != DynamicSpanExtent ? Count : size() - Offset};
