8 inline TypedQueryExecState build_typed_query_exec_state(
9 QueryImpl& query, World& world, const QueryInfo& queryInfo, const TypedQueryArgMeta* pMetas,
10 uint32_t argCount) {
11 TypedQueryExecState state{};
12 QueryImpl::DirectChunkArgEvalDesc directChunkDescs[MAX_ITEMS_IN_QUERY]{};
13 state.argCount = argCount;
14 GAIA_FOR(argCount) {
15 state.argIds[i] = pMetas[i].termId;
16 state.writeFlags[i] = pMetas[i].isWrite;
17 state.hasWriteArgs = state.hasWriteArgs || pMetas[i].isWrite;
18 state.needsInheritedArgIds = state.needsInheritedArgIds || !pMetas[i].isEntity;
19 directChunkDescs[i] = {.id = pMetas[i].termId, .isEntity = pMetas[i].isEntity, .isPair = pMetas[i].isPair};
20 }
21
22 state.canUseDirectChunkEval =
23 QueryImpl::can_use_direct_chunk_term_eval_descs(world, queryInfo, directChunkDescs, argCount);
24 if (state.needsInheritedArgIds)
25 state.hasInheritedTerms = queryInfo.has_potential_inherited_id_terms();
26 return state;
27 }

205 }
206
207 it.clear_touched_writes();

308 template <typename InvokeRow, typename OnRowDone>
309 inline void run_typed_query_rows_runtime(
310 const QueryInfo* pQueryInfo, Iter& it, InvokeRow&& invokeRow, OnRowDone&& onRowDone) {
311 const auto cnt = it.size();
312 const bool hasEntityFilters = pQueryInfo != nullptr && pQueryInfo->has_entity_filter_terms();
313
314 if (!hasEntityFilters) {
315 GAIA_FOR(cnt) {
316 invokeRow((uint32_t)i);
317 onRowDone((uint32_t)i);
318 }
319 return;
320 }
321
322 const auto entities = it.template view<Entity>();
323 GAIA_FOR(cnt) {
324 if (!QueryImpl::match_entity_filters(*pQueryInfo->world(), entities[i], *pQueryInfo))
325 continue;
326 invokeRow((uint32_t)i);
327 onRowDone((uint32_t)i);
328 }
329 }

408 template <bool UseFilters, typename ContainerOut>
409 inline void run_typed_arr_rows(
410 QueryImpl& query, const QueryInfo& queryInfo, Iter& it, ContainerOut& outArray, uint32_t changedWorldVersion,
411 uint32_t archetypeIdx, const Archetype* pArchetype, Chunk* pChunk, uint16_t from, uint16_t to,
412 bool needsBarrierCache, bool canUseDirectChunkEval) {
413 using ContainerItemType = typename ContainerOut::value_type;
414 const bool barrierPasses = !needsBarrierCache || queryInfo.barrier_passes(archetypeIdx);
415 if GAIA_UNLIKELY (!query.can_process_archetype_inter(queryInfo, *pArchetype, it.constraints(), barrierPasses))
416 return;
417
418 GAIA_PROF_SCOPE(query::arr);
419
420 it.set_archetype(pArchetype);
421 it.set_chunk(pChunk, from, to);
422 it.set_group_id(0);
423
424 const auto cnt = it.size();
425 if (cnt == 0)
426 return;
427
428 if constexpr (UseFilters) {
429 if (!QueryImpl::match_filters(*pChunk, queryInfo, changedWorldVersion))
430 return;
431 }
432
433 const bool hasEntityFilters = queryInfo.has_entity_filter_terms();
434 if (canUseDirectChunkEval) {
435 const auto dataView = it.template sview_auto<ContainerItemType>();
436 GAIA_FOR(cnt) {
437 auto tmp = dataView[i];
438 outArray.push_back(tmp);
439 }
440 return;
441 }
442
443 const auto dataView = it.template view<ContainerItemType>();
444 if (!hasEntityFilters) {
445 GAIA_FOR(cnt) {
446 const auto idx = it.template acc_index<ContainerItemType>(i);
447 auto tmp = dataView[idx];
448 outArray.push_back(tmp);
449 }
450 return;
451 }
452
453 const auto entities = it.template view<Entity>();
454 GAIA_FOR(cnt) {
455 if (!QueryImpl::match_entity_filters(*queryInfo.world(), entities[i], queryInfo))
456 continue;
457 const auto idx = it.template acc_index<ContainerItemType>(i);
458 auto tmp = dataView[idx];
459 outArray.push_back(tmp);
460 }
461 }

463 inline void QueryImpl::finish_typed_chunk_writes_runtime(
464 World& world, Chunk* pChunk, uint16_t from, uint16_t to, const Entity* pArgIds, const bool* pWriteFlags,
465 uint32_t argCnt) {
466 if (from >= to)
467 return;
468
469 Entity seenTerms[ChunkHeader::MAX_COMPONENTS]{};
470 uint32_t seenCnt = 0;
471 const auto entities = pChunk->entity_view();
472 const auto finish_term = [&](Entity term) {
473 GAIA_FOR(seenCnt) {
474 if (seenTerms[i] == term)
475 return;
476 }
477
478 seenTerms[seenCnt++] = term;
479 if (!world_is_out_of_line_component(world, term)) {
480 const auto compIdx = core::get_index(pChunk->ids_view(), term);
481 if (compIdx != BadIndex) {
482 pChunk->finish_write(compIdx, from, to);
483 return;
484 }
485 }
486
487 for (uint16_t row = from; row < to; ++row)
488 world_finish_write(world, term, entities[row]);
489 };
490
491 GAIA_FOR(argCnt) {
492 if (!pWriteFlags[i])
493 continue;
494 const auto term = pArgIds[i];
495 if (term != EntityBad)
496 finish_term(term);
497 }
498 }

513 inline void QueryImpl::finish_typed_iter_writes_runtime(
514 Iter& it, const Entity* pArgIds, const bool* pWriteFlags, uint32_t argCnt) {
515 auto* pChunk = const_cast<Chunk*>(it.chunk());
516 if (pChunk == nullptr || it.row_begin() >= it.row_end())
517 return;
518
519 auto& world = *it.world();
520 auto entities = it.entity_rows();
521 Entity seenTerms[ChunkHeader::MAX_COMPONENTS]{};
522 uint32_t seenCnt = 0;
523 const auto finish_term = [&](Entity term, uint32_t fieldIdx) {
524 GAIA_FOR(seenCnt) {
525 if (seenTerms[i] == term)
526 return;
527 }
528
529 seenTerms[seenCnt++] = term;
530 const bool isOutOfLine = world_is_out_of_line_component(world, term);
531 auto compIdx = uint8_t(0xFF);
532 if (it.comp_indices() != nullptr && fieldIdx < ChunkHeader::MAX_COMPONENTS)
533 compIdx = it.comp_indices()[fieldIdx];
534 else if (!isOutOfLine)
535 compIdx = (uint8_t)pChunk->comp_idx(term);
536
537 if (compIdx != 0xFF && !isOutOfLine) {
538 pChunk->finish_write(compIdx, it.row_begin(), it.row_end());
539 return;
540 }
541
542 GAIA_FOR(entities.size()) {

561 inline void QueryImpl::run_query_on_chunks_direct(
562 QueryInfo& queryInfo, const TypedQueryExecState& state, void* pFunc,
563 void (*runChunk)(QueryImpl&, Iter& it, void*, const TypedQueryExecState&)) {
564 auto& world = *queryInfo.world();
565 if (state.hasWriteArgs)
566 ::gaia::ecs::update_version(*m_worldVersion);
567
568 const bool hasFilters = queryInfo.has_filters();
569 auto cacheView = queryInfo.cache_archetype_view();
570 if (cacheView.empty())
571 return;
572
573 uint32_t idxFrom = 0;
574 uint32_t idxTo = (uint32_t)cacheView.size();
575 if (queryInfo.ctx().data.groupBy != EntityBad && m_groupIdSet != 0) {
576 const auto* pGroupData = queryInfo.selected_group_data(m_groupIdSet);
577 if (pGroupData == nullptr)
578 return;
579 idxFrom = pGroupData->idxFirst;
580 idxTo = pGroupData->idxLast + 1;
581 }
582
583 lock(*m_storage.world());
584 Iter it;
585 it.set_world(queryInfo.world());
586 const Archetype* pLastArchetype = nullptr;
587
588 for (uint32_t i = idxFrom; i < idxTo; ++i) {
589 const auto* pArchetype = cacheView[i];
590 if GAIA_UNLIKELY (!can_process_archetype_inter(queryInfo, *pArchetype, Constraints::EnabledOnly))
591 continue;
592
593 const auto& chunks = pArchetype->chunks();
594 for (auto* pChunk: chunks) {
595 const auto from = Iter::start_index(pChunk);
596 const auto to = Iter::end_index(pChunk);
597 if GAIA_UNLIKELY (from == to)
598 continue;
599
600 if (hasFilters) {
601 if GAIA_UNLIKELY (!match_filters(*pChunk, queryInfo, m_changedWorldVersion))
602 continue;
603 }
604
605 GAIA_PROF_SCOPE(query_func);
606 if (pArchetype != pLastArchetype) {
607 it.set_archetype(pArchetype);
608 pLastArchetype = pArchetype;
609 }
610 it.set_chunk(pChunk, from, to);
611 it.set_group_id(0);
612 runChunk(*this, it, pFunc, state);
613 finish_typed_chunk_state(*this, world, pChunk, from, to, state);
614 }
615 }
616
617 unlock(*m_storage.world());
618 commit_cmd_buffer_st(*m_storage.world());
619 commit_cmd_buffer_mt(*m_storage.world());
620 m_changedWorldVersion = *m_worldVersion;
621 }

774 inline void QueryImpl::each_direct_inter(
775 QueryInfo& queryInfo, Constraints constraints, void* pFunc, const TypedQueryExecState& state,
776 void (*runDirectChunk)(QueryImpl&, Iter&, void*, const TypedQueryExecState&), bool needsInheritedArgIds,
777 void (*invokeInherited)(World&, Entity, const Entity*, void*)) {
778 auto& world = *queryInfo.world();
779 const auto plan = direct_entity_seed_plan(world, queryInfo);
780 const bool hasWriteTerms = queryInfo.ctx().data.readWriteMask != 0;
781
782 auto exec_direct_entity = [&](Entity entity) {
783 uint8_t indices[ChunkHeader::MAX_COMPONENTS];
784 Entity termIds[ChunkHeader::MAX_COMPONENTS];
785 Iter it;
786 it.set_constraints(constraints);
787 init_direct_entity_iter(queryInfo, world, entity, it, indices, termIds);
788 runDirectChunk(*this, it, pFunc, state);
789 };
790
791 auto exec_entity = [&](Entity entity) {
792 if (needsInheritedArgIds && state.hasInheritedTerms) {
793 invokeInherited(world, entity, state.argIds, pFunc);
794 finish_typed_query_args_by_id(world, entity, state.argIds, state.writeFlags, state.argCount);
795 return;
796 }
797
798 exec_direct_entity(entity);
799 };
800
801 if (!hasWriteTerms && !plan.preferOrSeed) {
802 const auto* pSeedTerm = find_direct_all_seed_term(queryInfo, plan);
803 if (pSeedTerm != nullptr && can_use_direct_seed_run_cache(world, queryInfo, *pSeedTerm)) {
804 DirectEntitySeedInfo seedInfo{};
805 seedInfo.seededAllTerm = pSeedTerm->id;
806 seedInfo.seededAllMatchKind = pSeedTerm->matchKind;
807 seedInfo.seededFromAll = true;
808 if (!state.hasInheritedTerms) {
809 const auto runs = cached_direct_seed_runs(queryInfo, *pSeedTerm, seedInfo, constraints);
810 if (state.canUseDirectChunkEval) {
811 Iter it;
812 it.set_world(&world);
813 it.set_constraints(constraints);
814 const Archetype* pLastArchetype = nullptr;
815 for (const auto& run: runs) {
816 if (run.pArchetype != pLastArchetype) {
817 it.set_archetype(run.pArchetype);
818 pLastArchetype = run.pArchetype;
819 }
820 it.set_chunk(run.pChunk, run.from, run.to);
821 it.set_group_id(0);
822 runDirectChunk(*this, it, pFunc, state);
823 }
824 } else {
825 Iter it;
826 it.set_world(&world);
827 it.set_constraints(constraints);
828 const Archetype* pLastArchetype = nullptr;
829 uint8_t indices[ChunkHeader::MAX_COMPONENTS];
830 Entity termIds[ChunkHeader::MAX_COMPONENTS];
831 for (const auto& run: runs) {
832 const auto& ec = ::gaia::ecs::fetch(world, run.pChunk->entity_view()[run.from]);
833 init_direct_entity_iter(queryInfo, world, ec, it, indices, termIds, pLastArchetype);
834 it.set_chunk(run.pChunk, run.from, run.to);
835 it.set_group_id(0);
836 runDirectChunk(*this, it, pFunc, state);
837 }
838 }
839 } else {
840 const auto entities = cached_direct_seed_chunk_entities(queryInfo, *pSeedTerm, seedInfo, constraints);
841 for (const auto entity: entities)
842 exec_entity(entity);
843 }
844 return;
845 }
846 }
847
848 auto walk_entities = [&](auto&& execEntity) {
849 if (hasWriteTerms) {
850 auto& scratch = direct_query_scratch();
851 const auto seedInfo = build_direct_entity_seed(world, queryInfo, scratch.entities);
852 for (const auto entity: scratch.entities) {
853 if (!match_direct_entity_constraints(world, queryInfo, entity, constraints))
854 continue;
855 if (!match_direct_entity_terms(world, entity, queryInfo, seedInfo))
856 continue;
857 execEntity(entity);
858 }
859 return;
860 }
861
862 if (plan.preferOrSeed) {
863 for_each_direct_or_union(world, queryInfo, constraints, [&](Entity entity) {
864 execEntity(entity);
865 return true;
866 });
867 return;
868 }
869
870 (void)for_each_direct_all_seed(world, queryInfo, plan, constraints, [&](Entity entity) {
871 execEntity(entity);
872 return true;
873 });
874 };
875
876 walk_entities(exec_entity);
877 }
