792 template <typename TType>
793 GAIA_NODISCARD bool has_inter([[maybe_unused]] QueryOpKind op, bool isReadWrite) const {
794 using T = core::raw_t<TType>;
795
796 if constexpr (std::is_same_v<T, Entity>) {
797 // Entities are read-only.
798 GAIA_ASSERT(!isReadWrite);
799 // Skip Entity input args. Entities are always there.
800 return true;
801 } else {
802 Entity id;
803
804 if constexpr (is_pair<T>::value) {
805 const auto rel = m_plan.ctx.cc->get<typename T::rel>().entity;
806 const auto tgt = m_plan.ctx.cc->get<typename T::tgt>().entity;
807 id = (Entity)Pair(rel, tgt);
808 } else {
809 id = m_plan.ctx.cc->get<T>().entity;
810 }
811
812 const auto& ctxData = m_plan.ctx.data;
813 const auto compIdx = comp_idx<MAX_ITEMS_IN_QUERY>(ctxData.terms.data(), id, EntityBad);
814
815 if (op != ctxData.terms[compIdx].op)
816 return false;
817
818 // Read-write mask must match
819 const uint32_t maskRW = (uint32_t)ctxData.readWriteMask & (1U << compIdx);
820 const uint32_t maskXX = (uint32_t)isReadWrite << compIdx;
821 return maskRW == maskXX;
822 }
823 }

1390 matchedOrTerm = true;

1400 }
1401 if (term.op == QueryOpKind::Any)
1402 continue;

1405 if (!matched)
1406 return false;
1407 }

1409 return false;
1410 if (hadMatchBefore)
1411 return false;

1424 const auto keyLookup = EntityLookupKey(key);

1426 core::find_if(singleArchetypeLookup.begin(), singleArchetypeLookup.end(), [&](const auto& item) {
1427 return item.matches(keyLookup);

1431 entry.matchCount = (uint16_t)(entry.matchCount + matchCount);
1432 if (compIdx != ComponentIndexBad)
1433 entry.compIdx = compIdx;

1437 SingleArchetypeLookupItem{
1438 keyLookup, ComponentIndexEntry{const_cast<Archetype*>(&archetype), compIdx, matchCount}});

1441 const auto cntIds = (uint32_t)archetypeIds.size();

1443 const auto entity = archetypeIds[i];

1454 const auto rel = Entity((EntityId)entity.id(), 0, false, false, relKind);
1455 const auto tgt = Entity((EntityId)entity.gen(), 0, false, false, entity.kind());

1457 addLookupUnique(Pair(rel, All), ComponentIndexBad, 1);

1466 GAIA_ASSERT(ctxData.lastMatchedArchetypeIdx_Not.empty());

1474 ctxData.lastMatchedArchetypeIdx_Or = GAIA_MOV(lastMatchedArchetypeIdx_Or);
1475 ctxData.lastMatchedArchetypeIdx_Not = GAIA_MOV(lastMatchedArchetypeIdx_Not);

1489 void calculate_sort_data() {
1490 GAIA_PROF_SCOPE(queryinfo::calc_sort_data);
1491
1492 m_state.nonTrivial.archetypeSortData.clear();
1493
1494 // The function doesn't do any moves and expects that all chunks have their data sorted already.
1495 // We use a min-heap / priority queue - like structure during query iteration to merge sorted tables:
1496 // - we hold a cursor into each sorted chunk
1497 // - we compare the next entity from each chunk using your sorting function
1498 // - we then pick the smallest one (like k-way merge sort), and advance that cursor
1499 // This is esentially what this function does:
1500 // while (any_chunk_has_entities) {
1501 // find_chunk_with_smallest_next_element();
1502 // yield(entity_from_that_chunk);
1503 // adv_cursor_for_that_chunk();
1504 // }
1505 // This produces a globally sorted view without modifying actual data. It's a balance between
1506 // performance and memory usage. We could also sort the data in-place across all chunks, but that
1507 // would generated too many data moves (entities + all of their components).
1508
1509 struct Cursor {
1510 uint32_t chunkIdx = 0;
1511 uint16_t row = 0;
1512 };
1513
1514 auto& archetypes = m_state.archetypeCache;
1515
1516 // Initialize cursors. We will need as many as there are archetypes.
1517 cnt::sarray_ext<Cursor, 128> cursors(archetypes.size());
1518
1519 uint32_t currArchetypeIdx = (uint32_t)-1;
1520 Chunk* pCurrentChunk = nullptr;
1521 uint16_t currentStartRow = 0;
1522 uint16_t currentRow = 0;
1523
1524 const void* pDataMin = nullptr;
1525 const void* pDataCurr = nullptr;
1526
1527 while (true) {
1528 uint32_t minArchetypeIdx = (uint32_t)-1;
1529 Entity minEntity = EntityBad;
1530
1531 // Find the next entity across all tables/chunks
1532 for (uint32_t t = 0; t < archetypes.size(); ++t) {
1533 const auto* pArchetype = archetypes[t];
1534 const auto& chunks = pArchetype->chunks();
1535 auto& cur = cursors[t];
1536
1537 while (cur.chunkIdx < chunks.size() && cur.row >= chunks[cur.chunkIdx]->size()) {
1538 ++cur.chunkIdx;
1539 cur.row = 0;
1540 }
1541
1542 if (cur.chunkIdx >= chunks.size())
1543 continue;
1544
1545 const auto* pChunk = pArchetype->chunks()[cur.chunkIdx];
1546 auto entity = pChunk->entity_view()[cur.row];
1547
1548 if (m_plan.ctx.data.sortBy != ecs::EntityBad) {
1549 auto compIdx = world_component_index_comp_idx(*m_plan.ctx.w, *pArchetype, m_plan.ctx.data.sortBy);
1550 if (compIdx == BadIndex)
1551 compIdx = pChunk->comp_idx(m_plan.ctx.data.sortBy);
1552 pDataCurr = pChunk->comp_ptr(compIdx, cur.row);
1553 } else
1554 pDataCurr = &pChunk->entity_view()[cur.row];
1555
1556 if (minEntity == EntityBad) {
1557 minEntity = entity;
1558 minArchetypeIdx = t;
1559 pDataMin = pDataCurr;
1560 continue;
1561 }
1562
1563 if (m_plan.ctx.data.sortByFunc(*m_plan.ctx.w, pDataCurr, pDataMin) < 0) {
1564 minEntity = entity;
1565 minArchetypeIdx = t;
1566 }
1567 }
1568
1569 // No more results found, we can stop
1570 if (minArchetypeIdx == (uint32_t)-1)
1571 break;
1572
1573 auto& cur = cursors[minArchetypeIdx];
1574 const auto& chunks = archetypes[minArchetypeIdx]->chunks();
1575 Chunk* pChunk = chunks[cur.chunkIdx];
1576
1577 if (minArchetypeIdx == currArchetypeIdx && pChunk == pCurrentChunk) {
1578 // Current slice
1579 } else {
1580 // End previous slice
1581 if (pCurrentChunk != nullptr) {
1582 m_state.nonTrivial.archetypeSortData.push_back(
1583 {pCurrentChunk, currArchetypeIdx, currentStartRow, (uint16_t)(currentRow - currentStartRow)});
1584 }
1585
1586 // Start a new slice
1587 currArchetypeIdx = minArchetypeIdx;
1588 pCurrentChunk = pChunk;
1589 currentStartRow = cur.row;
1590 }
1591
1592 ++cur.row;
1593 currentRow = cur.row;
1594 }
1595
1596 if (pCurrentChunk != nullptr) {
1597 m_state.nonTrivial.archetypeSortData.push_back(
1598 {pCurrentChunk, currArchetypeIdx, currentStartRow, (uint16_t)(currentRow - currentStartRow)});
1599 }
1600 }

1787 ArchetypeCompIndices cacheData{};

1791 GAIA_FOR(cnt) {
