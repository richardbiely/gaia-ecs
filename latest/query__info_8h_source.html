481 const auto maxItems = (uint32_t)m_plan.ctx.data.cacheSrcTrav;

569 if (!can_reuse_dyn_cache())
570 return;
571
572 const auto relations = m_plan.ctx.data.deps.relations_view();
573 const auto cnt = (uint32_t)relations.size();
574 GAIA_FOR(cnt)

578 if (uses_direct_src_version_tracking()) {

596 m_state.srcTravSnapshotOverflowed = false;

600 m_state.varBindingMask = runtimeVarBindingMask;

603 template <typename TType>
604 GAIA_NODISCARD bool has_inter([[maybe_unused]] QueryOpKind op, bool isReadWrite) const {
605 using T = core::raw_t<TType>;
606
607 if constexpr (std::is_same_v<T, Entity>) {
608 // Entities are read-only.
609 GAIA_ASSERT(!isReadWrite);
610 // Skip Entity input args. Entities are always there.
611 return true;
612 } else {
613 Entity id;
614
615 if constexpr (is_pair<T>::value) {
616 const auto rel = m_plan.ctx.cc->get<typename T::rel>().entity;
617 const auto tgt = m_plan.ctx.cc->get<typename T::tgt>().entity;
618 id = (Entity)Pair(rel, tgt);
619 } else {
620 id = m_plan.ctx.cc->get<T>().entity;
621 }
622
623 const auto& ctxData = m_plan.ctx.data;
624 const auto compIdx = comp_idx<MAX_ITEMS_IN_QUERY>(ctxData.terms.data(), id, EntityBad);
625
626 if (op != ctxData.terms[compIdx].op)
627 return false;
628
629 // Read-write mask must match
630 const uint32_t maskRW = (uint32_t)ctxData.readWriteMask & (1U << compIdx);
631 const uint32_t maskXX = (uint32_t)isReadWrite << compIdx;
632 return maskRW == maskXX;
633 }
634 }

1094 bool register_archetype(const Archetype& archetype, Entity matchedSelector = EntityBad, bool assumeNew = false) {
1095 auto& ctxData = m_plan.ctx.data;
1096
1097 // Recompile if necessary.
1098 if ((ctxData.flags & QueryCtx::QueryFlags::Recompile) != 0)
1099 recompile();
1100
1101 if (!can_update_with_new_archetype())
1102 return false;
1103
1104 const bool hadMatchBefore = !assumeNew && m_state.archetypeSet.contains(&archetype);
1105 if (can_use_direct_create_archetype_match()) {
1106 const bool usesIs = direct_create_archetype_match_uses_is();
1107 bool hasOrTerms = false;
1108 bool matchedOrTerm = false;
1109 for (const auto& term: ctxData.terms_view()) {
1110 if (term.id == matchedSelector) {
1111 if (term.op == QueryOpKind::Or)
1112 matchedOrTerm = true;
1113 continue;
1114 }
1115
1116 const bool present = usesIs ? vm::detail::match_single_id_on_archetype(*world(), archetype, term.id)
1117 : world_component_index_match_count(*world(), archetype, term.id) != 0;
1118 if (term.op == QueryOpKind::Or) {
1119 hasOrTerms = true;
1120 matchedOrTerm |= present;
1121 continue;
1122 }
1123 if (term.op == QueryOpKind::Any)
1124 continue;
1125
1126 const bool matched = term.op == QueryOpKind::Not ? !present : present;
1127 if (!matched)
1128 return false;
1129 }
1130 if (hasOrTerms && !matchedOrTerm)
1131 return false;
1132 if (hadMatchBefore)
1133 return false;
1134
1135 if (assumeNew)
1136 add_new_archetype_to_immediate_caches(&archetype, true);
1137 else {
1138 add_archetype_to_seed_cache(&archetype, false);
1139 add_archetype_to_cache(&archetype, true, false);
1140 }
1141 return true;
1142 }

1144 SingleArchetypeLookup singleArchetypeLookup;
1145 auto addLookupUnique = [&](Entity key, uint16_t compIdx, uint16_t matchCount) {
1146 const auto keyLookup = EntityLookupKey(key);
1147 const auto itLookup =
1148 core::find_if(singleArchetypeLookup.begin(), singleArchetypeLookup.end(), [&](const auto& item) {
1149 return item.matches(keyLookup);
1150 });
1151 if (itLookup != singleArchetypeLookup.end()) {
1152 auto& entry = itLookup->entry;
1153 entry.matchCount = (uint16_t)(entry.matchCount + matchCount);
1154 if (compIdx != ComponentIndexBad)
1155 entry.compIdx = compIdx;
1156 return;
1157 }
1158 singleArchetypeLookup.push_back(
1159 SingleArchetypeLookupItem{
1160 keyLookup, ComponentIndexEntry{const_cast<Archetype*>(&archetype), compIdx, matchCount}});
1161 };
1162 auto archetypeIds = archetype.ids_view();
1163 const auto cntIds = (uint32_t)archetypeIds.size();
1164 GAIA_FOR(cntIds) {
1165 const auto entity = archetypeIds[i];
1166 singleArchetypeLookup.push_back(
1167 SingleArchetypeLookupItem{
1168 EntityLookupKey(entity), ComponentIndexEntry{const_cast<Archetype*>(&archetype), (uint16_t)i, 1}});
1169
1170 if (!entity.pair())
1171 continue;
1172
1173 // Wildcard pair lookups use the same special records as the world-level
1174 // entity-to-archetype map, so incremental matching can reuse the normal VM path.
1175 const auto relKind = entity.entity() ? EntityKind::EK_Uni : EntityKind::EK_Gen;
1176 const auto rel = Entity((EntityId)entity.id(), 0, false, false, relKind);
1177 const auto tgt = Entity((EntityId)entity.gen(), 0, false, false, entity.kind());
1178 addLookupUnique(Pair(All, tgt), ComponentIndexBad, 1);
1179 addLookupUnique(Pair(rel, All), ComponentIndexBad, 1);
1180 addLookupUnique(Pair(All, All), ComponentIndexBad, 1);
1181 }
1182
1183 auto lastMatchedArchetypeIdx_All = GAIA_MOV(ctxData.lastMatchedArchetypeIdx_All);
1184 auto lastMatchedArchetypeIdx_Or = GAIA_MOV(ctxData.lastMatchedArchetypeIdx_Or);
1185 auto lastMatchedArchetypeIdx_Not = GAIA_MOV(ctxData.lastMatchedArchetypeIdx_Not);
1186 GAIA_ASSERT(ctxData.lastMatchedArchetypeIdx_All.empty());
1187 GAIA_ASSERT(ctxData.lastMatchedArchetypeIdx_Or.empty());
1188 GAIA_ASSERT(ctxData.lastMatchedArchetypeIdx_Not.empty());
1189
1190 const auto* pArchetype = &archetype;
1191 const cnt::sarray<Entity, MaxVarCnt> noRuntimeVarBindings{};
1192 match(
1193 singleArchetypeLookup, std::span((const Archetype**)&pArchetype, 1), archetype.id(), noRuntimeVarBindings,
1194 0);
1195 ctxData.lastMatchedArchetypeIdx_All = GAIA_MOV(lastMatchedArchetypeIdx_All);
1196 ctxData.lastMatchedArchetypeIdx_Or = GAIA_MOV(lastMatchedArchetypeIdx_Or);
1197 ctxData.lastMatchedArchetypeIdx_Not = GAIA_MOV(lastMatchedArchetypeIdx_Not);
1198 const bool matched = assumeNew ? m_state.archetypeSet.contains(&archetype)
1199 : !hadMatchBefore && m_state.archetypeSet.contains(&archetype);
1200
1201 if (!matched)
1202 return false;
1203
1204 sort_entities();
1205 sort_cache_groups();
1206 return true;
1207 }

1211 void calculate_sort_data() {
1212 GAIA_PROF_SCOPE(queryinfo::calc_sort_data);
1213
1214 m_state.nonTrivial.archetypeSortData.clear();
1215
1216 // The function doesn't do any moves and expects that all chunks have their data sorted already.
1217 // We use a min-heap / priority queue - like structure during query iteration to merge sorted tables:
1218 // - we hold a cursor into each sorted chunk
1219 // - we compare the next entity from each chunk using your sorting function
1220 // - we then pick the smallest one (like k-way merge sort), and advance that cursor
1221 // This is esentially what this function does:
1222 // while (any_chunk_has_entities) {
1223 // find_chunk_with_smallest_next_element();
1224 // yield(entity_from_that_chunk);
1225 // adv_cursor_for_that_chunk();
1226 // }
1227 // This produces a globally sorted view without modifying actual data. It's a balance between
1228 // performance and memory usage. We could also sort the data in-place across all chunks, but that
1229 // would generated too many data moves (entities + all of their components).
1230
1231 struct Cursor {
1232 uint32_t chunkIdx = 0;
1233 uint16_t row = 0;
1234 };
1235
1236 auto& archetypes = m_state.archetypeCache;
1237
1238 // Initialize cursors. We will need as many as there are archetypes.
1239 cnt::sarray_ext<Cursor, 128> cursors(archetypes.size());
1240
1241 uint32_t currArchetypeIdx = (uint32_t)-1;
1242 Chunk* pCurrentChunk = nullptr;
1243 uint16_t currentStartRow = 0;
1244 uint16_t currentRow = 0;
1245
1246 const void* pDataMin = nullptr;
1247 const void* pDataCurr = nullptr;
1248
1249 while (true) {
1250 uint32_t minArchetypeIdx = (uint32_t)-1;
1251 Entity minEntity = EntityBad;
1252
1253 // Find the next entity across all tables/chunks
1254 for (uint32_t t = 0; t < archetypes.size(); ++t) {
1255 const auto* pArchetype = archetypes[t];
1256 const auto& chunks = pArchetype->chunks();
1257 auto& cur = cursors[t];
1258
1259 while (cur.chunkIdx < chunks.size() && cur.row >= chunks[cur.chunkIdx]->size()) {
1260 ++cur.chunkIdx;
1261 cur.row = 0;
1262 }
1263
1264 if (cur.chunkIdx >= chunks.size())
1265 continue;
1266
1267 const auto* pChunk = pArchetype->chunks()[cur.chunkIdx];
1268 auto entity = pChunk->entity_view()[cur.row];
1269
1270 if (m_plan.ctx.data.sortBy != ecs::EntityBad) {
1271 auto compIdx = world_component_index_comp_idx(*m_plan.ctx.w, *pArchetype, m_plan.ctx.data.sortBy);
1272 if (compIdx == BadIndex)
1273 compIdx = pChunk->comp_idx(m_plan.ctx.data.sortBy);
1274 pDataCurr = pChunk->comp_ptr(compIdx, cur.row);
1275 } else
1276 pDataCurr = &pChunk->entity_view()[cur.row];
1277
1278 if (minEntity == EntityBad) {
1279 minEntity = entity;
1280 minArchetypeIdx = t;
1281 pDataMin = pDataCurr;
1282 continue;
1283 }
1284
1285 if (m_plan.ctx.data.sortByFunc(*m_plan.ctx.w, pDataCurr, pDataMin) < 0) {
1286 minEntity = entity;
1287 minArchetypeIdx = t;
1288 }
1289 }
1290
1291 // No more results found, we can stop
1292 if (minArchetypeIdx == (uint32_t)-1)
1293 break;
1294
1295 auto& cur = cursors[minArchetypeIdx];
1296 const auto& chunks = archetypes[minArchetypeIdx]->chunks();
1297 Chunk* pChunk = chunks[cur.chunkIdx];
1298
1299 if (minArchetypeIdx == currArchetypeIdx && pChunk == pCurrentChunk) {
1300 // Current slice
1301 } else {
1302 // End previous slice
1303 if (pCurrentChunk != nullptr) {
1304 m_state.nonTrivial.archetypeSortData.push_back(
1305 {pCurrentChunk, currArchetypeIdx, currentStartRow, (uint16_t)(currentRow - currentStartRow)});
1306 }
1307
1308 // Start a new slice
1309 currArchetypeIdx = minArchetypeIdx;
1310 pCurrentChunk = pChunk;
1311 currentStartRow = cur.row;
1312 }
1313
1314 ++cur.row;
1315 currentRow = cur.row;
1316 }
1317
1318 if (pCurrentChunk != nullptr) {
1319 m_state.nonTrivial.archetypeSortData.push_back(
1320 {pCurrentChunk, currArchetypeIdx, currentStartRow, (uint16_t)(currentRow - currentStartRow)});
1321 }
1322 }

1414 void ensure_group_data() {
1415 if (m_plan.ctx.data.groupBy == EntityBad || !m_state.grouped.dataPending)
1416 return;
1417
1418 struct sort_cond {
1419 bool operator()(GroupId a, GroupId b) const {
1420 return a <= b;
1421 }
1422 };
1423
1424 core::sort(m_state.grouped.archetypeGroupIds, sort_cond{}, [&](uint32_t left, uint32_t right) {
1425 swap_archetype_cache_entry(left, right);
1426 });
1427
1428 m_state.grouped.archetypeGroupData.clear();
1429 m_state.grouped.selectedGroupDataValid = false;
1430
1431 if (m_state.grouped.archetypeGroupIds.empty()) {
1432 m_state.grouped.dataPending = false;
1433 return;
1434 }
1435
1436 GroupId groupId = m_state.grouped.archetypeGroupIds[0];
1437 uint32_t idxFirst = 0;
1438 const auto cnt = (uint32_t)m_state.grouped.archetypeGroupIds.size();
1439 for (uint32_t i = 1; i < cnt; ++i) {
1440 if (m_state.grouped.archetypeGroupIds[i] == groupId)
1441 continue;
1442
1443 m_state.grouped.archetypeGroupData.push_back({groupId, idxFirst, i - 1, false});
1444 groupId = m_state.grouped.archetypeGroupIds[i];
1445 idxFirst = i;
1446 }
1447
1448 m_state.grouped.archetypeGroupData.push_back({groupId, idxFirst, cnt - 1, false});
1449 m_state.grouped.dataPending = false;
1450 }

1452 void ensure_depth_order_hierarchy_barrier_cache_inter() {
1453 if (!world_depth_order_prunes_disabled_subtrees(*world(), m_plan.ctx.data.groupBy))
1454 return;
1455
1456 ensure_group_data();
1457
1458 const auto currRelationVersion = world_rel_version(*world(), m_plan.ctx.data.groupBy);
1459 const auto currEnabledVersion = world_enabled_hierarchy_version(*world());
1460 if (m_state.nonTrivial.barrierRelVersion == currRelationVersion &&
1461 m_state.nonTrivial.barrierEnabledVersion == currEnabledVersion)
1462 return;
1463
1464 m_state.nonTrivial.archetypeBarrierPasses.resize(m_state.archetypeCache.size(), 1);
1465
1466 const auto relation = m_plan.ctx.data.groupBy;
1467 for (uint32_t i = 0; i < m_state.archetypeCache.size(); ++i) {
1468 const auto* pArchetype = m_state.archetypeCache[i];
1469 auto& barrierPasses = m_state.nonTrivial.archetypeBarrierPasses[i];
1470 barrierPasses = 1;
1471
1472 auto ids = pArchetype->ids_view();
1473 for (auto idsIdx: pArchetype->pair_rel_indices(relation)) {
1474 const auto pair = ids[idsIdx];
1475 const auto parent = world_pair_target_if_alive(*world(), pair);
1476 if (parent == EntityBad || !world_entity_enabled_hierarchy(*world(), parent, relation)) {
1477 barrierPasses = 0;
1478 break;
1479 }
1480 }
1481 }
1482
1483 m_state.nonTrivial.barrierRelVersion = currRelationVersion;
1484 m_state.nonTrivial.barrierEnabledVersion = currEnabledVersion;
1485 }

1487 ArchetypeCompIndices create_comp_indices(const Archetype* pArchetype) {
1488 ArchetypeCompIndices cacheData{};
1489 core::fill(cacheData.indices, cacheData.indices + ChunkHeader::MAX_COMPONENTS, (uint8_t)0xFF);
1490 const auto terms = ctx().data.terms_view();
1491 const auto cnt = (uint32_t)terms.size();
1492 GAIA_FOR(cnt) {
1493 const auto& term = terms[i];
1494 const auto fieldIdx = term.fieldIndex;
1495 const auto queryId = term.id;
1496 if (!queryId.pair() && world_is_out_of_line_component(*world(), queryId)) {
1497 const auto compIdx = core::get_index_unsafe(pArchetype->ids_view(), queryId);
1498 GAIA_ASSERT(compIdx != BadIndex);
1499 cacheData.indices[fieldIdx] = 0xFF;
1500 continue;
1501 }
1502
1503 auto compIdx = world_component_index_comp_idx(*world(), *pArchetype, queryId);
1504 if (compIdx == BadIndex) {
1505 // Wildcard/semantic terms are not represented by an exact component index entry.
1506 // Fall back to the archetype-local scan in those cases.
1507 compIdx = core::get_index_unsafe(pArchetype->ids_view(), queryId);
1508 }
1509 GAIA_ASSERT(compIdx != BadIndex);
1510
1511 cacheData.indices[fieldIdx] = (uint8_t)compIdx;
1512 }
1513 return cacheData;
1514 }
