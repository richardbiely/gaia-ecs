866 template <typename TType>
867 GAIA_NODISCARD bool has_inter([[maybe_unused]] QueryOpKind op, bool isReadWrite) const {
868 using T = core::raw_t<TType>;
869
870 if constexpr (std::is_same_v<T, Entity>) {
871 // Entities are read-only.
872 GAIA_ASSERT(!isReadWrite);
873 // Skip Entity input args. Entities are always there.
874 return true;
875 } else {
876 Entity id;
877
878 if constexpr (is_pair<T>::value) {
879 const auto rel = m_plan.ctx.cc->get<typename T::rel>().entity;
880 const auto tgt = m_plan.ctx.cc->get<typename T::tgt>().entity;
881 id = (Entity)Pair(rel, tgt);
882 } else {
883 id = m_plan.ctx.cc->get<T>().entity;
884 }
885
886 const auto& ctxData = m_plan.ctx.data;
887 const auto compIdx = comp_idx<MAX_ITEMS_IN_QUERY>(ctxData.terms.data(), id, EntityBad);
888
889 if (op != ctxData.terms[compIdx].op)
890 return false;
891
892 // Read-write mask must match
893 const uint32_t maskRW = (uint32_t)ctxData.readWriteMask & (1U << compIdx);
894 const uint32_t maskXX = (uint32_t)isReadWrite << compIdx;
895 return maskRW == maskXX;
896 }

1147 if (!m_plan.vm.is_compiled())

1157
1158 if (refreshDynamicCache) {

1160 if (compareDynamicMembership) {

1167 } else if (m_state.seed_dirty()) {

1169 } else if (m_state.result_dirty()) {

1180 GAIA_ASSERT(archetypeLastId >= m_state.lastArchetypeId);
1181 if (!m_state.needs_refresh() && m_state.lastArchetypeId == archetypeLastId &&
1182 (!hasDynamicTerms || canReuseDynamicCache)) {

1207 }

1210 ctx.pMatchesStampByArchetypeId = &matchScratch.matchStamps;

1219 ctx.varBindings = runtimeVarBindings;
1220 ctx.varBindingMask = runtimeVarBindingMask;
1221

1223#if GAIA_ECS_TEST_HOOKS
1224 ++m_testMatchPassCount;
1225#endif
1226 m_plan.vm.exec(ctx);
1227
1228 // Write found matches to cache
1229 const bool trackMembershipChangeOnAdd = !compareDynamicMembership;
1230 for (const auto* pArchetype: *ctx.pMatchesArr) {
1231 if (hasDynamicTerms) {
1232 add_archetype_to_cache(pArchetype, trackMembershipChangeOnAdd, false);
1233 } else {
1234 add_archetype_to_seed_cache(pArchetype);
1235 add_archetype_to_cache(pArchetype, true, false);
1236 }
1237 }
1238
1239 if (compareDynamicMembership) {

1242 }
1243
1244 // Sort entities if necessary
1245 sort_entities();

1247 sort_cache_groups();
1248 snapshot_dyn_inputs(runtimeVarBindings, runtimeVarBindingMask);
1249 m_state.clear_dirty();

1448
1449 // Recompile if necessary.
1450 if ((ctxData.flags & QueryCtx::QueryFlags::Recompile) != 0)
1451 recompile();
1452
1453 if (!can_update_with_new_archetype())
1454 return false;
1455
1456 const bool hadMatchBefore = !assumeNew && m_state.archetypeSet.contains(&archetype);
1457 if (can_use_direct_create_archetype_match()) {
1458 const bool usesIs = direct_create_archetype_match_uses_is();
1459 bool hasOrTerms = false;
1460 bool matchedOrTerm = false;
1461 for (const auto& term: ctxData.terms_view()) {
1462 if (term.id == matchedSelector) {
1463 if (term.op == QueryOpKind::Or)
1464 matchedOrTerm = true;
1465 continue;
1466 }
1467
1468 const bool present = usesIs ? vm::detail::match_single_id_on_archetype(*world(), archetype, term.id)
1469 : world_component_index_match_count(*world(), archetype, term.id) != 0;
1470 if (term.op == QueryOpKind::Or) {
1471 hasOrTerms = true;
1472 matchedOrTerm |= present;
1473 continue;
1474 }
1475 if (term.op == QueryOpKind::Any)
1476 continue;
1477
1478 const bool matched = term.op == QueryOpKind::Not ? !present : present;
1479 if (!matched)
1480 return false;
1481 }
1482 if (hasOrTerms && !matchedOrTerm)
1483 return false;
1484 if (hadMatchBefore)
1485 return false;
1486
1487 if (assumeNew)
1488 add_new_archetype_to_immediate_caches(&archetype, true);
1489 else {
1490 add_archetype_to_seed_cache(&archetype, false);
1491 add_archetype_to_cache(&archetype, true, false);

1496 SingleArchetypeLookup singleArchetypeLookup;
1497 auto addLookupUnique = [&](Entity key, uint16_t compIdx, uint16_t matchCount) {
1498 const auto keyLookup = EntityLookupKey(key);
1499 const auto itLookup =
1500 core::find_if(singleArchetypeLookup.begin(), singleArchetypeLookup.end(), [&](const auto& item) {
1501 return item.matches(keyLookup);
1502 });
1503 if (itLookup != singleArchetypeLookup.end()) {
1504 auto& entry = itLookup->entry;
1505 entry.matchCount = (uint16_t)(entry.matchCount + matchCount);
1506 if (compIdx != ComponentIndexBad)
1507 entry.compIdx = compIdx;
1508 return;
1509 }
1510 singleArchetypeLookup.push_back(
1511 SingleArchetypeLookupItem{
1512 keyLookup, ComponentIndexEntry{const_cast<Archetype*>(&archetype), compIdx, matchCount}});
1513 };
1514 auto archetypeIds = archetype.ids_view();
1515 const auto cntIds = (uint32_t)archetypeIds.size();
1516 GAIA_FOR(cntIds) {
1517 const auto entity = archetypeIds[i];
1518 singleArchetypeLookup.push_back(
1519 SingleArchetypeLookupItem{
1520 EntityLookupKey(entity), ComponentIndexEntry{const_cast<Archetype*>(&archetype), (uint16_t)i, 1}});
1521
1522 if (!entity.pair())
1523 continue;
1524
1525 // Wildcard pair lookups use the same special records as the world-level
1526 // entity-to-archetype map, so incremental matching can reuse the normal VM path.
1527 const auto relKind = entity.entity() ? EntityKind::EK_Uni : EntityKind::EK_Gen;
1528 const auto rel = Entity((EntityId)entity.id(), 0, false, false, relKind);
1529 const auto tgt = Entity((EntityId)entity.gen(), 0, false, false, entity.kind());
1530 addLookupUnique(Pair(All, tgt), ComponentIndexBad, 1);
1531 addLookupUnique(Pair(rel, All), ComponentIndexBad, 1);
1532 addLookupUnique(Pair(All, All), ComponentIndexBad, 1);
1533 }
1534
1535 auto lastMatchedArchetypeIdx_All = GAIA_MOV(ctxData.lastMatchedArchetypeIdx_All);
1536 auto lastMatchedArchetypeIdx_Or = GAIA_MOV(ctxData.lastMatchedArchetypeIdx_Or);
1537 auto lastMatchedArchetypeIdx_Not = GAIA_MOV(ctxData.lastMatchedArchetypeIdx_Not);
1538 GAIA_ASSERT(ctxData.lastMatchedArchetypeIdx_All.empty());
1539 GAIA_ASSERT(ctxData.lastMatchedArchetypeIdx_Or.empty());
1540 GAIA_ASSERT(ctxData.lastMatchedArchetypeIdx_Not.empty());
1541
1542 const auto* pArchetype = &archetype;
1543 const cnt::sarray<Entity, MaxVarCnt> noRuntimeVarBindings{};
1544 match(

1563 void calculate_sort_data() {
1564 GAIA_PROF_SCOPE(queryinfo::calc_sort_data);
1565
1566 m_state.nonTrivial.archetypeSortData.clear();
1567
1568 // The function doesn't do any moves and expects that all chunks have their data sorted already.
1569 // We use a min-heap / priority queue - like structure during query iteration to merge sorted tables:
1570 // - we hold a cursor into each sorted chunk
1571 // - we compare the next entity from each chunk using your sorting function
1572 // - we then pick the smallest one (like k-way merge sort), and advance that cursor
1573 // This is esentially what this function does:
1574 // while (any_chunk_has_entities) {
1575 // find_chunk_with_smallest_next_element();
1576 // yield(entity_from_that_chunk);
1577 // adv_cursor_for_that_chunk();
1578 // }
1579 // This produces a globally sorted view without modifying actual data. It's a balance between
1580 // performance and memory usage. We could also sort the data in-place across all chunks, but that
1581 // would generated too many data moves (entities + all of their components).
1582
1583 struct Cursor {
1584 uint32_t chunkIdx = 0;
1585 uint16_t row = 0;
1586 };
1587
1588 auto& archetypes = m_state.archetypeCache;
1589
1590 // Initialize cursors. We will need as many as there are archetypes.
1591 cnt::sarray_ext<Cursor, 128> cursors(archetypes.size());
1592
1593 uint32_t currArchetypeIdx = (uint32_t)-1;
1594 Chunk* pCurrentChunk = nullptr;
1595 uint16_t currentStartRow = 0;
1596 uint16_t currentRow = 0;
1597
1598 const void* pDataMin = nullptr;
1599 const void* pDataCurr = nullptr;
1600
1601 while (true) {
1602 uint32_t minArchetypeIdx = (uint32_t)-1;
1603 Entity minEntity = EntityBad;
1604
1605 // Find the next entity across all tables/chunks
1606 for (uint32_t t = 0; t < archetypes.size(); ++t) {
1607 const auto* pArchetype = archetypes[t];
1608 const auto& chunks = pArchetype->chunks();
1609 auto& cur = cursors[t];
1610
1611 while (cur.chunkIdx < chunks.size() && cur.row >= chunks[cur.chunkIdx]->size()) {
1612 ++cur.chunkIdx;
1613 cur.row = 0;
1614 }
1615
1616 if (cur.chunkIdx >= chunks.size())
1617 continue;
1618
1619 const auto* pChunk = pArchetype->chunks()[cur.chunkIdx];
1620 auto entity = pChunk->entity_view()[cur.row];
1621
1622 if (m_plan.ctx.data.sortBy != ecs::EntityBad) {
1623 auto compIdx = world_component_index_comp_idx(*m_plan.ctx.w, *pArchetype, m_plan.ctx.data.sortBy);
1624 if (compIdx == BadIndex)
1625 compIdx = pChunk->comp_idx(m_plan.ctx.data.sortBy);
1626 pDataCurr = pChunk->comp_ptr(compIdx, cur.row);
1627 } else
1628 pDataCurr = &pChunk->entity_view()[cur.row];
1629
1630 if (minEntity == EntityBad) {
1631 minEntity = entity;
1632 minArchetypeIdx = t;
1633 pDataMin = pDataCurr;
1634 continue;
1635 }
1636
1637 if (m_plan.ctx.data.sortByFunc(*m_plan.ctx.w, pDataCurr, pDataMin) < 0) {
1638 minEntity = entity;
1639 minArchetypeIdx = t;
1640 }
1641 }
1642
1643 // No more results found, we can stop
1644 if (minArchetypeIdx == (uint32_t)-1)
1645 break;
1646
1647 auto& cur = cursors[minArchetypeIdx];
1648 const auto& chunks = archetypes[minArchetypeIdx]->chunks();
1649 Chunk* pChunk = chunks[cur.chunkIdx];
1650
1651 if (minArchetypeIdx == currArchetypeIdx && pChunk == pCurrentChunk) {
1652 // Current slice
1653 } else {
1654 // End previous slice
1655 if (pCurrentChunk != nullptr) {
1656 m_state.nonTrivial.archetypeSortData.push_back(
1657 {pCurrentChunk, currArchetypeIdx, currentStartRow, (uint16_t)(currentRow - currentStartRow)});
1658 }
1659
1660 // Start a new slice
1661 currArchetypeIdx = minArchetypeIdx;
1662 pCurrentChunk = pChunk;
1663 currentStartRow = cur.row;
1664 }
1665
1666 ++cur.row;
1667 currentRow = cur.row;
1668 }
1669
1670 if (pCurrentChunk != nullptr) {
1671 m_state.nonTrivial.archetypeSortData.push_back(
1672 {pCurrentChunk, currArchetypeIdx, currentStartRow, (uint16_t)(currentRow - currentStartRow)});
1673 }
1674 }

1879 void ensure_group_data(bool orderGroups) {

1881 return;

1884 return;
1885
1886 const auto groupCnt = (uint32_t)m_state.grouped.archetypeGroupIds.size();
1887 // Include the initial cache index so equal groups keep deterministic order under a strict comparator.
1888 cnt::darray_ext<uint64_t, 32> sortKeys;
1889 sortKeys.resize(groupCnt);
1890 GAIA_FOR(groupCnt) {
1891 sortKeys[i] = ((uint64_t)m_state.grouped.archetypeGroupIds[i] << 32U) | i;
1892 }
1893
1894 struct sort_cond {
1895 bool operator()(uint64_t a, uint64_t b) const {
1896 return a < b;
1897 }
1898 };
1899
1900 core::sort(sortKeys, sort_cond{}, [&](uint32_t left, uint32_t right) {
1901 const auto tmp = sortKeys[left];
1902 sortKeys[left] = sortKeys[right];
1903 sortKeys[right] = tmp;
1904 swap_archetype_cache_entry(left, right);
1905 });
1906
1907 m_state.grouped.archetypeGroupData.clear();
1908 m_state.grouped.selectedGroupDataValid = false;
1909
1910 if (m_state.grouped.archetypeGroupIds.empty()) {
1911 m_state.grouped.dataPending = false;
1912 return;
1913 }
1914
1915 GroupId groupId = m_state.grouped.archetypeGroupIds[0];
1916 uint32_t idxFirst = 0;
1917 const auto cnt = (uint32_t)m_state.grouped.archetypeGroupIds.size();

1919 if (m_state.grouped.archetypeGroupIds[i] == groupId)
1920 continue;
1921
1922 m_state.grouped.archetypeGroupData.push_back({groupId, idxFirst, i - 1, false});

1924 idxFirst = i;
1925 }
