53 GAIA_NODISCARD inline bool resolve_runtime_json_field_layout(
54 const ComponentCache* pCache, const RuntimeField& field, RuntimeJsonFieldLayout& out) noexcept {
55 out = {};
56 const auto* pFieldType = find_runtime_json_type(pCache, field.type);
57 out.type = field.type;
58 out.pType = pFieldType;
59 out.elemCount = ComponentCacheItem::field_element_count(field);
60
61 if (pFieldType != nullptr && pFieldType->typeKind == RuntimeTypeKind::Array) {
62 if (field.count != 0)
63 return false;
64 out.type = pFieldType->element_type();
65 out.elemCount = pFieldType->element_count();
66 out.pType = find_runtime_json_type(pCache, out.type);
67 }
68
69 return out.elemCount != 0 && runtime_json_type_size(out.pType, out.type, out.elemSize);
70 }

72 GAIA_NODISCARD inline ser::json_str
73 make_runtime_json_child_path(ser::json_str_view parent, ser::json_str_view child) {
74 if (parent.empty())
75 return ser::json_str(child);
76 if (child.empty())
77 return ser::json_str(parent);
78
79 ser::json_str path;
80 path.reserve(parent.size() + 1 + child.size());
81 path.append(parent.data(), parent.size());
82 path.append('.');
83 path.append(child.data(), child.size());
84 return path;
85 }

87 GAIA_NODISCARD inline ser::json_str make_runtime_json_element_path(ser::json_str_view parent, uint32_t index) {
88 ser::json_str path(parent);
89 path.append('[');
90 char idx[16]{};
91 const auto len = (uint32_t)snprintf(idx, sizeof(idx), "%u", index);
92 path.append(idx, len);
93 path.append(']');
94 return path;
95 }

97 GAIA_NODISCARD inline bool count_runtime_json_array_elements(ser::ser_json& reader, uint32_t& outCount) {
98 outCount = 0;
99 reader.ws();
100 const auto* start = reader.pos();
101 const auto* end = reader.end();
102 if (start == nullptr || end == nullptr || start > end)
103 return false;
104
105 ser::ser_json counter(start, (uint32_t)(end - start));
106 if (!counter.expect('['))
107 return false;
108 counter.ws();
109 if (counter.consume(']'))
110 return true;
111
112 while (true) {
113 if (!counter.skip_value())
114 return false;
115 ++outCount;
116 if (counter.consume(','))
117 continue;
118 return counter.consume(']');
119 }
120 }

126 inline bool write_runtime_json_field(
127 const ComponentCache* pCache, const ComponentCacheItem& owner, const RuntimeField& field,
128 const uint8_t* pBase, ser::ser_json& writer, uint32_t depth) {
129 RuntimeJsonFieldLayout layout{};
130 if (!resolve_runtime_json_field_layout(pCache, field, layout)) {
131 writer.value_null();
132 return false;
133 }
134
135 const auto fieldSize64 = (uint64_t)layout.elemSize * (uint64_t)layout.elemCount;
136 const auto end = (uint64_t)field.offset + fieldSize64;
137 if (layout.elemSize == 0 || fieldSize64 > UINT32_MAX || end > owner.comp.size()) {
138 writer.value_null();
139 return false;
140 }
141
142 const auto* pFieldData = pBase + field.offset;
143 if (layout.elemCount == 1 || runtime_json_is_char8_type(layout.pType, layout.type))
144 return write_runtime_json_value(
145 pCache, layout.pType, layout.type, pFieldData, (uint32_t)fieldSize64, writer, depth + 1);
146
147 bool ok = true;
148 writer.begin_array();
149 GAIA_FOR(layout.elemCount) {
150 const auto* pElemData = pFieldData + (uintptr_t)layout.elemSize * i;
151 ok = write_runtime_json_value(
152 pCache, layout.pType, layout.type, pElemData, layout.elemSize, writer, depth + 1) &&
153 ok;
154 }
155 writer.end_array();
156 return ok;
157 }

159 inline bool write_runtime_json_struct(
160 const ComponentCache* pCache, const ComponentCacheItem& item, const uint8_t* pData, ser::ser_json& writer,
161 uint32_t depth) {
162 if (depth >= RuntimeJsonMaxDepth) {
163 writer.value_null();
164 return false;
165 }
166
167 bool ok = true;
168 writer.begin_object();
169 GAIA_FOR(item.field_count()) {
170 const auto* pField = item.field(i);
171 GAIA_ASSERT(pField != nullptr);
172 const auto& field = *pField;
173 writer.key(field.name);
174 ok = write_runtime_json_field(pCache, item, field, pData, writer, depth + 1) && ok;
175 }
176 writer.end_object();
177 return ok;
178 }

180 inline bool write_runtime_json_value(
181 const ComponentCache* pCache, const ComponentCacheItem* pType, Entity typeEntity, const uint8_t* pData,
182 uint32_t valueSize, ser::ser_json& writer, uint32_t depth) {
183 if (depth >= RuntimeJsonMaxDepth) {
184 writer.value_null();
185 return false;
186 }
187
188 if (pType != nullptr && pType->typeKind == RuntimeTypeKind::Array) {
189 const auto elemCount = pType->element_count();
190 const auto elementType = pType->element_type();
191 const auto* pElementType = find_runtime_json_type(pCache, elementType);
192 uint32_t elemSize = 0;
193 if (elemCount == 0 || !runtime_json_type_size(pElementType, elementType, elemSize) ||
194 (uint64_t)elemSize * elemCount != valueSize) {
195 writer.value_null();
196 return false;
197 }
198
199 bool ok = true;
200 writer.begin_array();
201 GAIA_FOR(elemCount) {
202 const auto* pElemData = pData + (uintptr_t)elemSize * i;
203 ok = write_runtime_json_value(pCache, pElementType, elementType, pElemData, elemSize, writer, depth + 1) &&
204 ok;
205 }
206 writer.end_array();
207 return ok;
208 }
209
210 if (pType != nullptr && pType->typeKind == RuntimeTypeKind::Vector) {
211 const auto* adapter = pType->sequence_adapter();
212 const auto elementType = pType->element_type();
213 const auto* pElementType = find_runtime_json_type(pCache, elementType);
214 if (adapter == nullptr || adapter->count == nullptr || adapter->element == nullptr) {
215 writer.value_null();
216 return false;
217 }
218 RuntimeSequenceScope sequence{typeEntity, pData, nullptr, valueSize};
219 uint32_t elemCount = 0;
220 if (!adapter->count(adapter->ctx, sequence, elemCount)) {
221 writer.value_null();
222 return false;
223 }
224
225 bool ok = true;
226 writer.begin_array();
227 GAIA_FOR(elemCount) {
228 RuntimeSequenceElement element{};
229 element.type = elementType;
230 if (!adapter->element(adapter->ctx, sequence, i, element)) {
231 writer.value_null();
232 ok = false;
233 continue;
234 }
235 if (element.type == EntityBad)
236 element.type = elementType;
237 ok = write_runtime_json_value(
238 pCache, pElementType, element.type, (const uint8_t*)element.data, element.size, writer,
239 depth + 1) &&
240 ok;
241 }
242 writer.end_array();
243 return ok;
244 }
245
246 if (pType != nullptr && pType->typeKind == RuntimeTypeKind::Opaque) {
247 const auto* adapter = pType->opaque_adapter();
248 const auto semanticType = pType->opaque_as_type();
249 const auto* pSemanticType = find_runtime_json_type(pCache, semanticType);
250 if (adapter == nullptr || adapter->project == nullptr || pSemanticType == nullptr) {
251 writer.value_null();
252 return false;
253 }
254 RuntimeOpaqueScope opaque{typeEntity, pData, nullptr, valueSize};
255 RuntimeOpaqueValue projected{};
256 projected.type = semanticType;
257 if (!adapter->project(adapter->ctx, opaque, projected)) {
258 writer.value_null();
259 return false;
260 }
261 if (projected.type == EntityBad)
262 projected.type = semanticType;
263 if (projected.type != semanticType || projected.data == nullptr) {
264 writer.value_null();
265 return false;
266 }
267 return write_runtime_json_value(
268 pCache, pSemanticType, projected.type, (const uint8_t*)projected.data, projected.size, writer, depth + 1);
269 }
270
271 if (pType != nullptr && pType->typeKind == RuntimeTypeKind::Struct)
272 return write_runtime_json_struct(pCache, *pType, pData, writer, depth + 1);
273
274 ser::serialization_type_id type = ser::serialization_type_id::ignore;
275 if (pType != nullptr) {
276 if (!runtime_type_json_type(*pType, type)) {
277 writer.value_null();
278 return false;
279 }
280 } else if (!runtime_primitive_serialization_type(typeEntity, type)) {
281 writer.value_null();
282 return false;
283 }
284
285 return ser::detail::write_runtime_field_json(writer, pData, type, valueSize);
286 }

298 inline bool read_runtime_json_field(
299 const ComponentCache* pCache, const ComponentCacheItem& owner, const RuntimeField& field, uint8_t* pBase,
300 ser::ser_json& reader, ser::JsonDiagnostics& diagnostics, ser::json_str_view path, uint32_t depth, bool& ok) {
301 RuntimeJsonFieldLayout layout{};
302 if (!resolve_runtime_json_field_layout(pCache, field, layout)) {
303 ok = false;
304 warn_runtime_json(
305 diagnostics, ser::JsonDiagReason::FieldOutOfBounds, path,
306 "Runtime field uses an unknown reflected type.");
307 return reader.skip_value();
308 }
309
310 const auto fieldSize64 = (uint64_t)layout.elemSize * (uint64_t)layout.elemCount;
311 const auto end = (uint64_t)field.offset + fieldSize64;
312 if (layout.elemSize == 0 || fieldSize64 > UINT32_MAX || end > owner.comp.size()) {
313 ok = false;
314 warn_runtime_json(
315 diagnostics, ser::JsonDiagReason::FieldOutOfBounds, path,
316 "Runtime field points outside component size or uses an unsupported type.");
317 return reader.skip_value();
318 }
319
320 auto* pFieldData = pBase + field.offset;
321 if (layout.elemCount == 1 || runtime_json_is_char8_type(layout.pType, layout.type))
322 return read_runtime_json_value(
323 pCache, layout.pType, layout.type, pFieldData, (uint32_t)fieldSize64, reader, diagnostics, path,
324 depth + 1, ok);
325
326 if (!reader.expect('['))
327 return false;
328
329 GAIA_FOR(layout.elemCount) {
330 if (i > 0 && !reader.expect(','))
331 return false;
332 const auto elemPath = make_runtime_json_element_path(path, i);
333 auto* pElemData = pFieldData + (uintptr_t)layout.elemSize * i;
334 if (!read_runtime_json_value(
335 pCache, layout.pType, layout.type, pElemData, layout.elemSize, reader, diagnostics, elemPath,
336 depth + 1, ok))
337 return false;
338 }
339 return reader.expect(']');
340 }

342 inline bool read_runtime_json_struct(
343 const ComponentCache* pCache, const ComponentCacheItem& item, uint8_t* pData, ser::ser_json& reader,
344 ser::JsonDiagnostics& diagnostics, ser::json_str_view path, uint32_t depth, bool& ok) {
345 if (depth >= RuntimeJsonMaxDepth) {
346 ok = false;
347 warn_runtime_json(
348 diagnostics, ser::JsonDiagReason::FieldOutOfBounds, path, "Runtime JSON nesting is too deep.");
349 return reader.skip_value();
350 }
351
352 if (reader.parse_null()) {
353 ok = false;
354 warn_runtime_json(
355 diagnostics, ser::JsonDiagReason::NullComponentPayload, path, "Runtime object payload is null.");
356 return true;
357 }
358
359 if (!reader.expect('{'))
360 return false;
361
362 reader.ws();
363 if (reader.consume('}'))
364 return true;
365
366 while (true) {
367 ser::json_str_view key;
368 bool keyFromScratch = false;
369 if (!reader.parse_string_view(key, &keyFromScratch))
370 return false;
371 ser::json_str keyStorage;
372 if (keyFromScratch) {
373 keyStorage.assign(key.data(), key.size());
374 key = keyStorage;
375 }
376 if (!reader.expect(':'))
377 return false;
378
379 const auto* pField = item.field(util::str_view(key.data(), (uint32_t)key.size()));
380 const auto fieldPath = make_runtime_json_child_path(path, key);
381 if (pField == nullptr) {
382 ok = false;
383 warn_runtime_json(diagnostics, ser::JsonDiagReason::UnknownField, fieldPath, "Unknown runtime field.");
384 if (!reader.skip_value())
385 return false;
386 } else if (!read_runtime_json_field(
387 pCache, item, *pField, pData, reader, diagnostics, fieldPath, depth + 1, ok))
388 return false;
389
390 reader.ws();
391 if (reader.consume(','))
392 continue;
393 if (reader.consume('}'))
394 break;
395 return false;
396 }
397
398 return true;
399 }

401 inline bool read_runtime_json_value(
402 const ComponentCache* pCache, const ComponentCacheItem* pType, Entity typeEntity, uint8_t* pData,
403 uint32_t valueSize, ser::ser_json& reader, ser::JsonDiagnostics& diagnostics, ser::json_str_view path,
404 uint32_t depth, bool& ok) {
405 if (depth >= RuntimeJsonMaxDepth) {
406 ok = false;
407 warn_runtime_json(
408 diagnostics, ser::JsonDiagReason::FieldOutOfBounds, path, "Runtime JSON nesting is too deep.");
409 return reader.skip_value();
410 }
411
412 if (pType != nullptr && pType->typeKind == RuntimeTypeKind::Array) {
413 const auto elemCount = pType->element_count();
414 const auto elementType = pType->element_type();
415 const auto* pElementType = find_runtime_json_type(pCache, elementType);
416 uint32_t elemSize = 0;
417 if (elemCount == 0 || !runtime_json_type_size(pElementType, elementType, elemSize) ||
418 (uint64_t)elemSize * elemCount != valueSize) {
419 ok = false;
420 warn_runtime_json(
421 diagnostics, ser::JsonDiagReason::FieldOutOfBounds, path,
422 "Runtime array payload uses an invalid reflected element type.");
423 return reader.skip_value();
424 }
425
426 if (!reader.expect('['))
427 return false;
428
429 GAIA_FOR(elemCount) {
430 if (i > 0 && !reader.expect(','))
431 return false;
432 const auto elemPath = make_runtime_json_element_path(path, i);
433 auto* pElemData = pData + (uintptr_t)elemSize * i;
434 if (!read_runtime_json_value(
435 pCache, pElementType, elementType, pElemData, elemSize, reader, diagnostics, elemPath, depth + 1,
436 ok))
437 return false;
438 }
439 return reader.expect(']');
440 }
441
442 if (pType != nullptr && pType->typeKind == RuntimeTypeKind::Vector) {
443 const auto* adapter = pType->sequence_adapter();
444 const auto elementType = pType->element_type();
445 const auto* pElementType = find_runtime_json_type(pCache, elementType);
446 uint32_t elemCount = 0;
447 if (adapter == nullptr || adapter->resize == nullptr || adapter->element == nullptr ||
448 !count_runtime_json_array_elements(reader, elemCount)) {
449 ok = false;
450 warn_runtime_json(
451 diagnostics, ser::JsonDiagReason::FieldOutOfBounds, path,
452 "Runtime vector payload cannot be resized or traversed.");
453 return reader.skip_value();
454 }
455
456 RuntimeSequenceScope sequence{typeEntity, pData, pData, valueSize};
457 if (!adapter->resize(adapter->ctx, sequence, elemCount)) {
458 ok = false;
459 warn_runtime_json(
460 diagnostics, ser::JsonDiagReason::FieldOutOfBounds, path,
461 "Runtime vector adapter rejected the requested element count.");
462 return reader.skip_value();
463 }
464
465 if (!reader.expect('['))
466 return false;
467 GAIA_FOR(elemCount) {
468 if (i > 0 && !reader.expect(','))
469 return false;
470 RuntimeSequenceElement element{};
471 element.type = elementType;
472 if (!adapter->element(adapter->ctx, sequence, i, element) || element.mutData == nullptr) {
473 ok = false;
474 warn_runtime_json(
475 diagnostics, ser::JsonDiagReason::FieldOutOfBounds, path,
476 "Runtime vector adapter rejected an element.");
477 return reader.skip_value();
478 }
479 if (element.type == EntityBad)
480 element.type = elementType;
481 const auto elemPath = make_runtime_json_element_path(path, i);
482 if (!read_runtime_json_value(
483 pCache, pElementType, element.type, (uint8_t*)element.mutData, element.size, reader, diagnostics,
484 elemPath, depth + 1, ok))
485 return false;
486 }
487 return reader.expect(']');
488 }
489
490 if (pType != nullptr && pType->typeKind == RuntimeTypeKind::Opaque) {
491 const auto* adapter = pType->opaque_adapter();
492 const auto semanticType = pType->opaque_as_type();
493 const auto* pSemanticType = find_runtime_json_type(pCache, semanticType);
494 if (adapter == nullptr || adapter->project == nullptr || pSemanticType == nullptr) {
495 ok = false;
496 warn_runtime_json(
497 diagnostics, ser::JsonDiagReason::FieldOutOfBounds, path,
498 "Runtime opaque payload cannot be projected.");
499 return reader.skip_value();
500 }
501 RuntimeOpaqueScope opaque{typeEntity, pData, pData, valueSize};
502 RuntimeOpaqueValue projected{};
503 projected.type = semanticType;
504 if (!adapter->project(adapter->ctx, opaque, projected) || projected.mutData == nullptr) {
505 ok = false;
506 warn_runtime_json(
507 diagnostics, ser::JsonDiagReason::FieldOutOfBounds, path,
508 "Runtime opaque adapter rejected projection.");
509 return reader.skip_value();
510 }
511 if (projected.type == EntityBad)
512 projected.type = semanticType;
513 if (projected.type != semanticType) {
514 ok = false;
515 warn_runtime_json(
516 diagnostics, ser::JsonDiagReason::FieldOutOfBounds, path,
517 "Runtime opaque adapter projected an unexpected semantic type.");
518 return reader.skip_value();
519 }
520 const bool parsed = read_runtime_json_value(
521 pCache, pSemanticType, projected.type, (uint8_t*)projected.mutData, projected.size, reader, diagnostics,
522 path, depth + 1, ok);
523 if (parsed && adapter->commit != nullptr && ok) {
524 if (!adapter->commit(adapter->ctx, opaque, projected)) {
525 ok = false;
526 warn_runtime_json(
527 diagnostics, ser::JsonDiagReason::FieldOutOfBounds, path, "Runtime opaque adapter rejected commit.");
528 }
529 }
530 return parsed;
531 }
532
533 if (pType != nullptr && pType->typeKind == RuntimeTypeKind::Struct)
534 return read_runtime_json_struct(pCache, *pType, pData, reader, diagnostics, path, depth + 1, ok);
535
536 ser::serialization_type_id type = ser::serialization_type_id::ignore;
537 if (pType != nullptr) {
538 if (!runtime_type_json_type(*pType, type)) {
539 ok = false;
540 warn_runtime_json(
541 diagnostics, ser::JsonDiagReason::FieldOutOfBounds, path,
542 "Runtime field uses an unsupported reflected type.");
543 return reader.skip_value();
544 }
545 } else if (!runtime_primitive_serialization_type(typeEntity, type)) {
546 ok = false;
547 warn_runtime_json(diagnostics, ser::JsonDiagReason::FieldOutOfBounds, path, "Runtime field type is unknown.");
548 return reader.skip_value();
549 }
550
551 bool fieldOk = true;
552 if (!ser::detail::read_runtime_field_json(reader, pData, type, valueSize, fieldOk))
553 return false;
554 if (!fieldOk) {
555 ok = false;
556 warn_runtime_json(
557 diagnostics, ser::JsonDiagReason::FieldValueAdjusted, path,
558 "Field value was lossy, truncated, or unsupported for the target runtime field type.");
559 }
560 return true;
561 }

591 inline bool json_to_component(
592 const ComponentCacheItem& item, void* pComponentData, ser::ser_json& reader, ser::JsonDiagnostics& diagnostics,
593 ser::json_str_view componentPath = {}) {
594 GAIA_ASSERT(pComponentData != nullptr);
595 if (pComponentData == nullptr)
596 return false;
597
598 if (reader.parse_null()) {
599 detail::warn_runtime_json(
600 diagnostics, ser::JsonDiagReason::NullComponentPayload, componentPath, "Component payload is null.");
601 return true;
602 }
603
604 if (item.typeKind == RuntimeTypeKind::Opaque && item.opaque_adapter() != nullptr) {
605 bool ok = true;
606 return detail::read_runtime_json_value(
607 item.owner_cache(), &item, item.entity, reinterpret_cast<uint8_t*>(pComponentData), item.comp.size(),
608 reader, diagnostics, componentPath, 0, ok);
609 }
610
611 bool rawFound = false;
612 bool fieldFound = false;
613 bool ok = true;
614 ser::ser_buffer_binary rawPayload;
615 auto* pBase = reinterpret_cast<uint8_t*>(pComponentData);
616
617 if (!reader.expect('{'))
618 return false;
619
620 reader.ws();
621 if (reader.consume('}')) {
622 detail::warn_runtime_json(
623 diagnostics, ser::JsonDiagReason::MissingRuntimeFieldsOrRawPayload, componentPath,
624 "Component object is empty and contains no runtime fields or $raw payload.");
625 return true;
626 }
627
628 while (true) {
629 ser::json_str_view key;
630 bool keyFromScratch = false;
631 if (!reader.parse_string_view(key, &keyFromScratch))
632 return false;
633 ser::json_str keyStorage;
634 if (keyFromScratch) {
635 keyStorage.assign(key.data(), key.size());
636 key = keyStorage;
637 }
638 if (!reader.expect(':'))
639 return false;
640
641 const auto fieldPath = detail::make_runtime_json_child_path(componentPath, key);
642 if (key == "$raw") {
643 rawFound = true;
644 if (!ser::detail::parse_json_byte_array(reader, rawPayload))
645 return false;
646 } else if (item.field_count() != 0 && item.comp.soa() == 0) {
647 const auto* pField = item.field(util::str_view(key.data(), (uint32_t)key.size()));
648 if (pField == nullptr) {
649 ok = false;
650 detail::warn_runtime_json(
651 diagnostics, ser::JsonDiagReason::UnknownField, fieldPath, "Unknown runtime field.");
652 if (!reader.skip_value())
653 return false;
654 } else {
655 fieldFound = true;
656 if (!detail::read_runtime_json_field(
657 item.owner_cache(), item, *pField, pBase, reader, diagnostics, fieldPath, 0, ok))
658 return false;
659 }
660 } else {
661 ok = false;
662 detail::warn_runtime_json(
663 diagnostics, ser::JsonDiagReason::MissingRuntimeFieldsOrRawPayload, fieldPath,
664 "Runtime field metadata is unavailable for keyed field payloads.");
665 if (!reader.skip_value())
666 return false;
667 }
668
669 reader.ws();
670 if (reader.consume(','))
671 continue;
672 if (reader.consume('}'))
673 break;
674 return false;
675 }
676
677 if (rawFound) {
678 if (item.comp.soa() != 0) {

681 detail::make_runtime_json_child_path(componentPath, "$raw"),
682 "$raw payload is not supported for SoA components.");
683 return true;

685
686 auto s = ser::make_serializer(rawPayload);
687 s.seek(0);
688 item.load(s, pBase, 0, 1, 1);
689 }
690
691 if (!rawFound && !fieldFound)
692 detail::warn_runtime_json(
693 diagnostics, ser::JsonDiagReason::MissingRuntimeFieldsOrRawPayload, componentPath,
694 "Component payload contains neither recognized runtime fields nor $raw data.");
695
696 (void)ok;
697 return true;

713 inline bool World::save_json(ser::ser_json& writer, ser::JsonSaveFlags flags) const {
714 auto write_raw_component = [&](const ComponentCacheItem& item, const uint8_t* pData, uint32_t from, uint32_t to,
715 uint32_t cap) {
716 ser::ser_buffer_binary raw;
717 auto s = ser::make_serializer(raw);
718 item.save(s, pData, from, to, cap);
719
720 writer.begin_object();
721 writer.key("$raw");
722 writer.begin_array();
723 const auto* pRaw = raw.data();
724 GAIA_FOR(raw.bytes()) writer.value_int(pRaw[i]);
725 writer.end_array();
726 writer.end_object();
727 };
728
729 bool ok = true;
730 const bool includeBinarySnapshot = (flags & ser::JsonSaveFlags::BinarySnapshot) != 0;
731 const bool allowRawFallback = (flags & ser::JsonSaveFlags::RawFallback) != 0;
732 ser::bin_stream binarySnapshot;
733 if (includeBinarySnapshot) {
734 auto s = ser::make_serializer(binarySnapshot);
735 s.reset();
736 save_to(s);
737 }
738
739 writer.clear();
740 writer.begin_object();
741 writer.key("format");
742 writer.value_int(WorldSerializerJSONVersion);
743 writer.key("worldVersion");
744 writer.value_int(m_worldVersion);
745 if (includeBinarySnapshot) {
746 writer.key("binary");
747 writer.begin_array();
748 {
749 const auto* pData = (const uint8_t*)binarySnapshot.data();
750 GAIA_FOR(binarySnapshot.bytes()) writer.value_int(pData[i]);
751 }
752 writer.end_array();
753 }
754 writer.key("archetypes");
755 writer.begin_array();
756
757 for (const auto* pArchetype: m_archetypes) {
758 if (pArchetype == nullptr || pArchetype->chunks().empty())
759 continue;
760
761 writer.begin_object();
762 writer.key("id");
763 writer.value_int((uint32_t)pArchetype->id());
764 writer.key("hash");
765 writer.value_int((uint64_t)pArchetype->lookup_hash().hash);
766
767 writer.key("components");
768 writer.begin_array();
769 {
770 for (const auto entity: pArchetype->ids_view()) {
771 const auto itemName = name(entity);
772 if (!itemName.empty())
773 writer.value_string(itemName.data(), itemName.size());
774 else
775 writer.value_string("<unnamed>");
776 }
777 }
778 writer.end_array();
779
780 writer.key("entities");
781 writer.begin_array();
782 {
783 for (const auto* pChunk: pArchetype->chunks()) {
784 if (pChunk == nullptr || pChunk->empty())
785 continue;
786
787 const auto ents = pChunk->entity_view();
788 const auto recs = pChunk->comp_rec_view();
789 GAIA_FOR((uint32_t)ents.size()) {
790 const auto entity = ents[i];
791
792 writer.begin_object();
793 {
794 writer.key("entity");
795 {
796 writer.begin_object();
797 writer.key("id");
798 writer.value_int(entity.id());
799 writer.key("gen");
800 writer.value_int(entity.gen());
801 writer.key("pair");
802 writer.value_bool(entity.pair());
803 writer.key("kind");
804 writer.value_string(EntityKindString[entity.kind()]);
805 const auto entityName = name(entity);
806 if (!entityName.empty()) {
807 writer.key("name");
808 writer.value_string(entityName.data(), entityName.size());
809 }
810 writer.end_object();
811 }
812
813 writer.key("components");
814 writer.begin_object();
815 {
816 GAIA_FOR_((uint32_t)recs.size(), j) {
817 const auto& rec = recs[j];
818 const auto& item = *rec.pItem;
819 const auto name = symbol(item.entity);
820 writer.key(name.data(), name.size());
821
822 // Tags have no associated payload.
823 if (rec.comp.size() == 0) {
824 writer.value_bool(true);
825 continue;
826 }
827
828 const auto row = item.entity.kind() == EntityKind::EK_Uni ? 0U : i;
829
830 if (item.field_count() != 0 && rec.comp.soa() == 0) {
831 const auto* pCompData = pChunk->comp_ptr(j, row);
832 ok = ecs::component_to_json(item, pCompData, writer) && ok;
833 } else {
834 if (allowRawFallback)
835 write_raw_component(item, rec.pData, row, row + 1, pChunk->capacity());
836 else {
837 writer.value_null();
838 ok = false;
839 }
840 }
841 }
842 writer.end_object();
843 }
844 }
845 writer.end_object();
846 }
847 }
848 }
849
850 writer.end_array();
851 writer.end_object();
852 }
853
854 writer.end_array();
855 writer.end_object();
856 return ok;
857 }

867 inline bool World::load_json(const char* json, uint32_t len, ser::JsonDiagnostics& diagnostics) {
868 diagnostics.clear();
869 if (json == nullptr)
870 return false;
871 if (len == 0) {
872 diagnostics.add(
873 ser::JsonDiagSeverity::Error, ser::JsonDiagReason::InvalidJson, "$",
874 "Input JSON length must be provided and non-zero.");
875 return false;
876 }
877
878 const auto dataLen = len;
879 const auto* p = json;
880 const auto* end = json + dataLen;
881 auto warn = [&](ser::JsonDiagReason reason, ser::json_str_view path, const char* message) {
882 diagnostics.add(ser::JsonDiagSeverity::Warning, reason, path, message);
883 };
884 auto error = [&](ser::JsonDiagReason reason, ser::json_str_view path, const char* message) {
885 diagnostics.add(ser::JsonDiagSeverity::Error, reason, path, message);
886 };
887
888 // Validate top-level format version first.
889 {
890 ser::ser_json header(json, dataLen);
891 if (!header.expect('{')) {
892 error(ser::JsonDiagReason::InvalidJson, "$", "Root JSON value must be an object.");
893 return false;
894 }
895
896 bool hasFormat = false;
897 uint32_t formatValue = 0;
898
899 header.ws();
900 if (!header.consume('}')) {
901 while (true) {
902 ser::json_str_view key;
903 if (!header.parse_string_view(key))
904 return false;
905 if (!header.expect(':'))
906 return false;
907
908 if (key == "format") {
909 double d = 0.0;
910 if (!header.parse_number(d))
911 return false;
912 if (d < 0.0 || d > 4294967295.0)
913 return false;
914 const auto v = (uint32_t)d;
915 if ((double)v != d)
916 return false;
917 formatValue = v;
918 hasFormat = true;
919 } else {
920 if (!header.skip_value())
921 return false;
922 }
923
924 header.ws();
925 if (header.consume(','))
926 continue;
927 if (header.consume('}'))
928 break;
929 return false;
930 }
931 }
932
933 header.ws();
934 if (!header.eof())
935 return false;
936
937 if (!hasFormat) {
938 error(ser::JsonDiagReason::MissingFormatField, "$.format", "Missing required 'format' field.");
939 return false;
940 }
941
942 if (formatValue != WorldSerializerJSONVersion) {
943 error(
944 ser::JsonDiagReason::UnsupportedFormatVersion, "$.format",
945 "Unsupported format version. Expected numeric value 1.");
946 return false;
947 }
948 }
949
950 // Prefer fast-path: binary snapshot payload.
951 {
952 const char key[] = "\"binary\"";
953 const uint32_t keyLen = (uint32_t)(sizeof(key) - 1);
954 const char* keyPos = nullptr;
955 for (const char* it = p; it + keyLen <= end; ++it) {
956 if (memcmp(it, key, keyLen) == 0) {
957 keyPos = it;
958 break;
959 }
960 }
961 if (keyPos != nullptr) {
962 const char* arr = nullptr;
963 for (const char* it = keyPos + keyLen; it < end; ++it) {
964 if (*it == '[') {
965 arr = it;
966 break;
967 }
968 }
969 if (arr != nullptr) {
970 ser::bin_stream serializer;
971 ser::ser_json binaryReader(arr, (uint32_t)(end - arr));
972 if (!ser::detail::parse_json_byte_array(binaryReader, serializer))
973 return false;
974
975 return load(serializer);
976 }
977 }
978 }

981 ser::ser_json jp(json, dataLen);

1014 auto parse_and_apply_component_value = [&](Entity entity, const ComponentCacheItem& item,
1015 ser::json_str_view compPath) -> bool {
1016 jp.ws();
1017 if (jp.eof())
1018 return false;
1019
1020 // Tag values are currently ignored by semantic loader.
1021 const char next = jp.peek();
1022 if (next == 't' || next == 'f') {
1023 bool tagValue = false;
1024 if (!jp.parse_bool(tagValue))
1025 return false;
1026 warn(
1027 ser::JsonDiagReason::TagValueIgnored, compPath,
1028 "Tag-like boolean component payload is ignored in semantic mode.");
1029 return true;
1030 }
1031
1032 if (jp.parse_null()) {
1033 warn(
1034 ser::JsonDiagReason::NullComponentPayload, compPath,
1035 "Null component payload is ignored in semantic mode.");
1036 return true;
1037 }
1038
1039 if (!has_direct_component(entity, item.entity))
1040 add(entity, item.entity);
1041
1042 auto loc = locate_component_data(entity, item);
1043 if (!loc.valid) {
1044 warn(
1045 ser::JsonDiagReason::MissingComponentStorage, compPath,
1046 "Component storage is unavailable on the target entity.");
1047 return jp.skip_value();
1048 }
1049
1050 auto* pRowData = loc.pBase + (uintptr_t)item.comp.size() * loc.row;
1051 if (!ecs::json_to_component(item, pRowData, jp, diagnostics, compPath))
1052 return false;
1053 return true;
1054 };
