113 void diag(const World& world) const {
114 auto diagEdge = [&](const auto& edges) {
115 for (const auto& edge: edges) {
116 const auto entity = edge.first.entity();
117 if (entity.pair()) {

120 GAIA_LOG_N(
121 " pair [%u:%u], %.*s -> %.*s, aid:%u",
122 //
123 entity.id(), entity.gen(), (int)name0.size(), name0.empty() ? "" : name0.data(), (int)name1.size(),
124 name1.empty() ? "" : name1.data(), edge.second.id);
125 } else {
126 const auto name = entity_name(world, entity);
127 GAIA_LOG_N(
128 " ent [%u:%u], %.*s [%s], aid:%u",
129 //
130 entity.id(), entity.gen(), (int)name.size(), name.empty() ? "" : name.data(),
131 EntityKindString[entity.kind()], edge.second.id);
132 }
133 }
134 };

136 // Add edges (movement towards the leafs)
137 if (!m_edgesAdd.empty()) {
138 GAIA_LOG_N(" Add edges - count:%u", (uint32_t)m_edgesAdd.size());
139 diagEdge(m_edgesAdd);
