49 class JobArgsFunc {
50 static constexpr uint32_t BufferSize = 24;
51
52 enum class Op : uint8_t { Invoke, Destroy, Move };
53 using OpFn = void (*)(Op op, JobArgsFunc* dst, JobArgsFunc* src, const JobArgs* pArgs);
54
55 OpFn m_func = nullptr;
56 alignas(std::max_align_t) uint8_t m_storage[BufferSize];
57
58 void destroy() {
59 if (m_func != nullptr) {
60 m_func(Op::Destroy, this, nullptr, nullptr);
61 m_func = nullptr;
62 }
63 }

65 template <typename F>
66 void init(F&& f) {
67 using Fn = std::decay_t<F>;
68 static_assert(std::is_invocable_r_v<void, Fn&, const JobArgs&>, "JobArgsFunc requires a compatible callable");
69 static_assert(std::is_move_constructible_v<Fn>, "Callable must be move-constructible");
70 static_assert(
71 alignof(Fn) <= alignof(std::max_align_t), "Over-aligned callables are not supported for JobArgsFunc");
72
73 if constexpr (sizeof(Fn) <= BufferSize) {
74 new (m_storage) Fn(GAIA_FWD(f));
75
76 m_func = [](Op op, JobArgsFunc* dst, JobArgsFunc* src, const JobArgs* pArgs) {
77 auto* pFn = reinterpret_cast<Fn*>(dst->m_storage);
78 switch (op) {
79 case Op::Invoke:
80 GAIA_ASSERT(pArgs != nullptr);
81 (*pFn)(*pArgs);
82 break;
83 case Op::Destroy:
84 if constexpr (!std::is_trivially_destructible_v<Fn>)
85 pFn->~Fn();
86 break;
87 case Op::Move: {
88 GAIA_ASSERT(src != nullptr);
89 auto* pSrcFn = reinterpret_cast<Fn*>(src->m_storage);
90 new (dst->m_storage) Fn(GAIA_MOV(*pSrcFn));
91 if constexpr (!std::is_trivially_destructible_v<Fn>)
92 pSrcFn->~Fn();
93 dst->m_func = src->m_func;
94 src->m_func = nullptr;
95 break;
96 }
97 }
98 };
99 } else {
100 auto* pStorage = mem::AllocHelper::alloc<Fn>();
101 GAIA_ASSERT((uintptr_t)pStorage % alignof(Fn) == 0);
102 auto* pFunc = new (pStorage) Fn(GAIA_FWD(f));
103 *reinterpret_cast<Fn**>(m_storage) = pFunc;
104
105 m_func = [](Op op, JobArgsFunc* dst, JobArgsFunc* src, const JobArgs* pArgs) {
106 auto*& pFn = *reinterpret_cast<Fn**>(dst->m_storage);
107 switch (op) {
108 case Op::Invoke:
109 GAIA_ASSERT(pArgs != nullptr);
110 GAIA_ASSERT(pFn != nullptr);
111 (*pFn)(*pArgs);
112 break;
113 case Op::Destroy:
114 GAIA_ASSERT(pFn != nullptr);
115 if constexpr (!std::is_trivially_destructible_v<Fn>)
116 pFn->~Fn();
117 mem::AllocHelper::free(pFn);
118 pFn = nullptr;
119 break;
120 case Op::Move:
121 GAIA_ASSERT(src != nullptr);
122 *reinterpret_cast<Fn**>(dst->m_storage) = *reinterpret_cast<Fn**>(src->m_storage);
123 dst->m_func = src->m_func;
124 *reinterpret_cast<Fn**>(src->m_storage) = nullptr;
125 src->m_func = nullptr;
126 break;
127 }
128 };
129 }
130 }
