498 GAIA_NODISCARD uint32_t calc_new_mem_offset(uint32_t addr, size_t cnt) const noexcept {
499 if (comp.soa() == 0) {
500 addr = (uint32_t)mem::detail::get_aligned_byte_offset(addr, comp.alig(), comp.size(), cnt);
501 } else {
502 GAIA_FOR(comp.soa()) {
503 addr = (uint32_t)mem::detail::get_aligned_byte_offset(addr, comp.alig(), soaSizes[i], cnt);
504 }
505 // TODO: Magic offset. Otherwise, SoA data might leak past the chunk boundary when accessing
506 // the last element. By faking the memory offset we can bypass this is issue for now.
507 // Obviously, this needs fixing at some point.
508 addr += comp.soa() * 12;
509 }

518 template <typename T>
519 GAIA_NODISCARD static uint32_t init_type_name(char (&nameTmp)[MaxNameLength]) {
520 auto ct_name = detail::ComponentDesc<T>::name();
521
522 // Allocate enough memory for the name string + the null-terminating character (
523 // the compile time string returned by ComponentDesc<T>::name is not null-terminated).
524 // Different compilers will give a bit different strings, e.g.:
525 // Clang/GCC: gaia::ecs::uni<Position>
526 // MSVC : gaia::ecs::uni<struct Position>
527 // Therefore, we first copy the compile-time string and then tweak it so it is
528 // the same on all supported compilers.
529 auto nameTmpLen = (uint32_t)ct_name.size();
530 GAIA_ASSERT(nameTmpLen < MaxNameLength);
531 memcpy((void*)nameTmp, (const void*)ct_name.data(), nameTmpLen);
532 nameTmp[ct_name.size()] = 0;
533
534 auto strip_prefix = [&](const char* prefix, uint32_t prefixLen) {
535 if (nameTmpLen <= prefixLen || strncmp(nameTmp, prefix, prefixLen) != 0)
536 return;
537
538 memmove(nameTmp, nameTmp + prefixLen, nameTmpLen - prefixLen + 1);
539 nameTmpLen -= prefixLen;
540 };
541
542 strip_prefix("const ", 6);
543
544 const uint32_t NSubstrings = 3;
545 const char* to_remove[NSubstrings] = {"class ", "struct ", "enum "};
546 const uint32_t to_remove_len[NSubstrings] = {6, 7, 5};
547 GAIA_FOR(NSubstrings) {
548 strip_prefix(to_remove[i], to_remove_len[i]);
549 }
550
551 while (nameTmpLen > 0) {
552 const auto ch = nameTmp[nameTmpLen - 1];
553 if (ch != ' ' && ch != '&' && ch != '*')
554 break;
555
556 nameTmp[--nameTmpLen] = 0;
557 }
558
559 if (nameTmpLen > 6 && strncmp(nameTmp + nameTmpLen - 6, " const", 6) == 0) {
560 nameTmpLen -= 6;
561 nameTmp[nameTmpLen] = 0;
562 }
563
564 // Normalization template names by removing keywords when they appear as template argument
565 // prefixes instead of as part of a longer identifier.
566 GAIA_FOR(NSubstrings) {
567 const auto* str = to_remove[i];
568 const auto len = to_remove_len[i];
569
570 auto* pos = nameTmp;
571 while ((pos = strstr(pos, str)) != nullptr) {
572 const bool isBoundary = pos == nameTmp || pos[-1] == '<' || pos[-1] == ',' || pos[-1] == ' ';
573 if (!isBoundary) {
574 ++pos;
575 continue;
576 }
577
578 const auto tailMaxLen = (size_t)(MaxNameLength - (uint32_t)(pos + len - nameTmp));
579 memmove(pos, pos + len, GAIA_STRLEN(pos + len, tailMaxLen) + 1);
580 nameTmpLen -= len;
581 }
582 }
583
584 return nameTmpLen;
585 }

635 GAIA_NODISCARD static ComponentCacheItem* create(Entity entity) {
636 static_assert(core::is_raw_v<T>);
637
638 constexpr auto componentSize = detail::ComponentDesc<T>::size();
639 static_assert(
640 componentSize < Component::MaxComponentSizeInBytes,
641 "Trying to register a component larger than the maximum allowed component size! In the future this "
642 "restriction won't apply to components not stored inside archetype chunks.");
643
644 char nameTmp[MaxNameLength];
645 const auto nameTmpLen = init_type_name<T>(nameTmp);
646
647 uint8_t soaSizes[meta::StructToTupleMaxTypes]{};
648 RuntimeFieldDesc fields[meta::StructToTupleMaxTypes]{};
649 auto desc = detail::ComponentDesc<T>::make(
650 util::str_view(nameTmp, nameTmpLen), std::span<uint8_t, meta::StructToTupleMaxTypes>{soaSizes});
651#if GAIA_ECS_AUTO_COMPONENT_FIELDS
652 desc.fields = fields;
653 desc.fieldCount =
654 detail::ComponentDesc<T>::auto_fields(std::span<RuntimeFieldDesc, meta::StructToTupleMaxTypes>{fields});
655#endif
656 return create(entity, desc);
657 }

663 GAIA_NODISCARD static ComponentCacheItem* create(Entity entity, const ecs::ComponentDesc& desc) {
664 GAIA_ASSERT(!desc.name.empty());
665 GAIA_ASSERT(desc.name.size() < MaxNameLength);
666 GAIA_ASSERT(desc.size < Component::MaxComponentSizeInBytes);
667 GAIA_ASSERT((desc.size == 0 && desc.alig == 0) || (desc.alig > 0 && desc.alig < Component::MaxAlignment));
668 GAIA_ASSERT(desc.soa <= meta::StructToTupleMaxTypes);
669
670 auto* cci = new ComponentCacheItem();
671 cci->entity = entity;
672 cci->comp = Component(entity.id(), desc.soa, desc.size, desc.alig, desc.storageType);
673 cci->hashLookup = desc.hashLookup.hash != 0
674 ? desc.hashLookup
675 : ComponentLookupHash{core::calculate_hash64(desc.name.data(), desc.name.size())};
676
677 if (desc.soa > 0 && desc.pSoaSizes != nullptr) {
678 GAIA_FOR(desc.soa) cci->soaSizes[i] = desc.pSoaSizes[i];
679 }
680
681 init_name(cci->name, desc.name);
682
683 cci->func_ctor = desc.funcCtor;
684 cci->func_move_ctor = desc.funcMoveCtor;
685 cci->func_copy_ctor = desc.funcCopyCtor;
686 cci->func_dtor = desc.funcDtor;
687 cci->func_copy = desc.funcCopy;
688 cci->func_move = desc.funcMove;
689 cci->func_swap = desc.funcSwap;
690 cci->func_cmp = desc.funcCmp;
691 cci->func_save = desc.funcSave;
692 cci->func_load = desc.funcLoad;
693 cci->typeKind = desc.typeKind;
694 cci->underlyingType = desc.underlyingType;
695 cci->elementType = desc.elementType;
696 cci->elementCount = desc.elementCount;
697 cci->opaqueAsType = desc.opaqueAsType;
698
699 if (desc.fieldCount > 0) {
700 GAIA_FOR(desc.fieldCount) {
701 const bool copied = cci->copy_runtime_field(desc.fields[i]);
702 GAIA_ASSERT(copied);
703 }
704 }
705
706 if (desc.constantCount > 0) {
707 GAIA_FOR(desc.constantCount) {
708 const bool copied = cci->copy_runtime_constant(desc.constants[i]);
709 GAIA_ASSERT(copied);
710 }
711 }
712
713 return cci;
714 }
