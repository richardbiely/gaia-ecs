167 void move_soa_element(
168 void* pDst, const void* pSrc, uint32_t idxDst, uint32_t idxSrc, uint32_t capDst,
169 uint32_t capSrc) const noexcept {
170 capDst = soa_capacity(capDst);
171 capSrc = soa_capacity(capSrc);
172 const std::span<const uint8_t> fieldSizes{soaSizes, comp.soa()};
173 GAIA_FOR(comp.soa()) {
174 auto* pD = mem::data_view_policy_soa_erased::set(pDst, comp.alig(), fieldSizes, i, idxDst, capDst);
175 const auto* pS = mem::data_view_policy_soa_erased::get(pSrc, comp.alig(), fieldSizes, i, idxSrc, capSrc);
176 memmove(pD, pS, soaSizes[i]);
177 }
178 }

187 void swap_soa_elements(
188 void* pLeft, void* pRight, uint32_t idxLeft, uint32_t idxRight, uint32_t capLeft,
189 uint32_t capRight) const noexcept {
190 capLeft = soa_capacity(capLeft);
191 capRight = soa_capacity(capRight);
192 const std::span<const uint8_t> fieldSizes{soaSizes, comp.soa()};
193 GAIA_FOR(comp.soa()) {
194 auto* pL = mem::data_view_policy_soa_erased::set(pLeft, comp.alig(), fieldSizes, i, idxLeft, capLeft);
195 auto* pR = mem::data_view_policy_soa_erased::set(pRight, comp.alig(), fieldSizes, i, idxRight, capRight);
196 GAIA_FOR_(soaSizes[i], j) {
197 const auto value = pL[j];
198 pL[j] = pR[j];
199 pR[j] = value;
200 }
201 }

361 l[i] = r[i];
362 r[i] = tmp;

586 GAIA_NODISCARD uint32_t calc_new_mem_offset(uint32_t addr, size_t cnt) const noexcept {
587 if (comp.soa() == 0) {
588 addr = (uint32_t)mem::detail::get_aligned_byte_offset(addr, comp.alig(), comp.size(), cnt);
589 } else {
590 GAIA_FOR(comp.soa()) {
591 addr = (uint32_t)mem::detail::get_aligned_byte_offset(addr, comp.alig(), soaSizes[i], cnt);
592 }
593 // TODO: Magic offset. Otherwise, SoA data might leak past the chunk boundary when accessing
594 // the last element. By faking the memory offset we can bypass this is issue for now.
595 // Obviously, this needs fixing at some point.
596 addr += comp.soa() * 12;
597 }
598 return addr;

606 template <typename T>
607 GAIA_NODISCARD static uint32_t init_type_name(char (&nameTmp)[MaxNameLength]) {
608 auto ct_name = detail::ComponentDesc<T>::name();
609
610 // Allocate enough memory for the name string + the null-terminating character (
611 // the compile time string returned by ComponentDesc<T>::name is not null-terminated).
612 // Different compilers will give a bit different strings, e.g.:
613 // Clang/GCC: gaia::ecs::uni<Position>
614 // MSVC : gaia::ecs::uni<struct Position>
615 // Therefore, we first copy the compile-time string and then tweak it so it is
616 // the same on all supported compilers.
617 auto nameTmpLen = (uint32_t)ct_name.size();
618 GAIA_ASSERT(nameTmpLen < MaxNameLength);
619 memcpy((void*)nameTmp, (const void*)ct_name.data(), nameTmpLen);
620 nameTmp[ct_name.size()] = 0;
621
622 auto strip_prefix = [&](const char* prefix, uint32_t prefixLen) {
623 if (nameTmpLen <= prefixLen || strncmp(nameTmp, prefix, prefixLen) != 0)
624 return;
625
626 memmove(nameTmp, nameTmp + prefixLen, nameTmpLen - prefixLen + 1);
627 nameTmpLen -= prefixLen;
628 };
629
630 strip_prefix("const ", 6);
631
632 const uint32_t NSubstrings = 3;
633 const char* to_remove[NSubstrings] = {"class ", "struct ", "enum "};
634 const uint32_t to_remove_len[NSubstrings] = {6, 7, 5};
635 GAIA_FOR(NSubstrings) {
636 strip_prefix(to_remove[i], to_remove_len[i]);
637 }
638
639 while (nameTmpLen > 0) {
640 const auto ch = nameTmp[nameTmpLen - 1];
641 if (ch != ' ' && ch != '&' && ch != '*')
642 break;
643
644 nameTmp[--nameTmpLen] = 0;
645 }
646
647 if (nameTmpLen > 6 && strncmp(nameTmp + nameTmpLen - 6, " const", 6) == 0) {
648 nameTmpLen -= 6;
649 nameTmp[nameTmpLen] = 0;
650 }
651
652 // Normalization template names by removing keywords when they appear as template argument
653 // prefixes instead of as part of a longer identifier.
654 GAIA_FOR(NSubstrings) {
655 const auto* str = to_remove[i];
656 const auto len = to_remove_len[i];
657
658 auto* pos = nameTmp;
659 while ((pos = strstr(pos, str)) != nullptr) {
660 const bool isBoundary = pos == nameTmp || pos[-1] == '<' || pos[-1] == ',' || pos[-1] == ' ';
661 if (!isBoundary) {
662 ++pos;
663 continue;
664 }
665
666 const auto tailMaxLen = (size_t)(MaxNameLength - (uint32_t)(pos + len - nameTmp));
667 memmove(pos, pos + len, GAIA_STRLEN(pos + len, tailMaxLen) + 1);
668 nameTmpLen -= len;
669 }
670 }
671
672 return nameTmpLen;
673 }

751 GAIA_NODISCARD static ComponentCacheItem* create(Entity entity, const ecs::ComponentDesc& desc) {
752 GAIA_ASSERT(!desc.name.empty());
753 GAIA_ASSERT(desc.name.size() < MaxNameLength);
754 GAIA_ASSERT(desc.size < Component::MaxComponentSizeInBytes);
755 GAIA_ASSERT((desc.size == 0 && desc.alig == 0) || (desc.alig > 0 && desc.alig < Component::MaxAlignment));
756 GAIA_ASSERT(desc.alig == 0 || (desc.alig & (desc.alig - 1)) == 0);
757 GAIA_ASSERT(desc.soa <= meta::StructToTupleMaxTypes);
758 GAIA_ASSERT(desc.soa == 0 || desc.pSoaSizes != nullptr);
759#if GAIA_ASSERT_ENABLED
760 if (desc.soa > 0) {
761 uint32_t soaSize = 0;
762 GAIA_FOR(desc.soa) {
763 GAIA_ASSERT(desc.pSoaSizes[i] != 0);
764 soaSize += desc.pSoaSizes[i];
765 }
766 GAIA_ASSERT(soaSize <= desc.size);
767 }
768#endif
769
770 auto* cci = new ComponentCacheItem();
771 cci->entity = entity;
772 cci->comp = Component(entity.id(), desc.soa, desc.size, desc.alig, desc.storageType);
773 cci->hashLookup = desc.hashLookup.hash != 0
774 ? desc.hashLookup
775 : ComponentLookupHash{core::calculate_hash64(desc.name.data(), desc.name.size())};
776
777 if (desc.soa > 0) {
778 GAIA_FOR(desc.soa) cci->soaSizes[i] = desc.pSoaSizes[i];
779 }
780
781 init_name(cci->name, desc.name);
782
783 cci->func_ctor = desc.funcCtor;
784 cci->func_move_ctor = desc.funcMoveCtor;
785 cci->func_copy_ctor = desc.funcCopyCtor;
786 cci->func_dtor = desc.funcDtor;
787 cci->func_copy = desc.funcCopy;
788 cci->func_move = desc.funcMove;
789 cci->func_swap = desc.funcSwap;
790 cci->func_cmp = desc.funcCmp;
791 cci->func_save = desc.funcSave;
