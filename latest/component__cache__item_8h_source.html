166 void move_soa_element(
167 void* pDst, const void* pSrc, uint32_t idxDst, uint32_t idxSrc, uint32_t capDst,
168 uint32_t capSrc) const noexcept {
169 capDst = soa_capacity(capDst);
170 capSrc = soa_capacity(capSrc);
171 const std::span<const uint8_t> fieldSizes{soaSizes, comp.soa()};
172 GAIA_FOR(comp.soa()) {
173 auto* pD = mem::data_view_policy_soa_erased::set(pDst, comp.alig(), fieldSizes, i, idxDst, capDst);
174 const auto* pS = mem::data_view_policy_soa_erased::get(pSrc, comp.alig(), fieldSizes, i, idxSrc, capSrc);
175 memmove(pD, pS, soaSizes[i]);
176 }
177 }

186 void swap_soa_elements(
187 void* pLeft, void* pRight, uint32_t idxLeft, uint32_t idxRight, uint32_t capLeft,
188 uint32_t capRight) const noexcept {
189 capLeft = soa_capacity(capLeft);
190 capRight = soa_capacity(capRight);
191 const std::span<const uint8_t> fieldSizes{soaSizes, comp.soa()};
192 GAIA_FOR(comp.soa()) {
193 auto* pL = mem::data_view_policy_soa_erased::set(pLeft, comp.alig(), fieldSizes, i, idxLeft, capLeft);
194 auto* pR = mem::data_view_policy_soa_erased::set(pRight, comp.alig(), fieldSizes, i, idxRight, capRight);
195 GAIA_FOR_(soaSizes[i], j) {
196 const auto value = pL[j];
197 pL[j] = pR[j];
198 pR[j] = value;
199 }
200 }

587 GAIA_NODISCARD uint32_t calc_new_mem_offset(uint32_t addr, size_t cnt) const noexcept {
588 if (comp.soa() == 0) {
589 addr = (uint32_t)mem::detail::get_aligned_byte_offset(addr, comp.alig(), comp.size(), cnt);
590 } else {
591 GAIA_FOR(comp.soa()) {
592 addr = (uint32_t)mem::detail::get_aligned_byte_offset(addr, comp.alig(), soaSizes[i], cnt);
593 }
594 // TODO: Magic offset. Otherwise, SoA data might leak past the chunk boundary when accessing
595 // the last element. By faking the memory offset we can bypass this is issue for now.
596 // Obviously, this needs fixing at some point.
597 addr += comp.soa() * 12;
598 }
599 return addr;

607 template <typename T>
608 GAIA_NODISCARD static uint32_t init_type_name(char (&nameTmp)[MaxNameLength]) {
609 auto ct_name = detail::ComponentDesc<T>::name();
610
611 // Allocate enough memory for the name string + the null-terminating character (
612 // the compile time string returned by ComponentDesc<T>::name is not null-terminated).
613 // Different compilers will give a bit different strings, e.g.:
614 // Clang/GCC: gaia::ecs::uni<Position>
615 // MSVC : gaia::ecs::uni<struct Position>
616 // Therefore, we first copy the compile-time string and then tweak it so it is
617 // the same on all supported compilers.
618 auto nameTmpLen = (uint32_t)ct_name.size();
619 GAIA_ASSERT(nameTmpLen < MaxNameLength);
620 memcpy((void*)nameTmp, (const void*)ct_name.data(), nameTmpLen);
621 nameTmp[ct_name.size()] = 0;
622
623 auto strip_prefix = [&](const char* prefix, uint32_t prefixLen) {
624 if (nameTmpLen <= prefixLen || strncmp(nameTmp, prefix, prefixLen) != 0)
625 return;
626
627 memmove(nameTmp, nameTmp + prefixLen, nameTmpLen - prefixLen + 1);
628 nameTmpLen -= prefixLen;
629 };
630
631 strip_prefix("const ", 6);
632
633 const uint32_t NSubstrings = 3;
634 const char* to_remove[NSubstrings] = {"class ", "struct ", "enum "};
635 const uint32_t to_remove_len[NSubstrings] = {6, 7, 5};
636 GAIA_FOR(NSubstrings) {
637 strip_prefix(to_remove[i], to_remove_len[i]);
638 }
639
640 while (nameTmpLen > 0) {
641 const auto ch = nameTmp[nameTmpLen - 1];
642 if (ch != ' ' && ch != '&' && ch != '*')
643 break;
644
645 nameTmp[--nameTmpLen] = 0;
646 }
647
648 if (nameTmpLen > 6 && strncmp(nameTmp + nameTmpLen - 6, " const", 6) == 0) {
649 nameTmpLen -= 6;
650 nameTmp[nameTmpLen] = 0;
651 }
652
653 // Normalization template names by removing keywords when they appear as template argument
654 // prefixes instead of as part of a longer identifier.
655 GAIA_FOR(NSubstrings) {
656 const auto* str = to_remove[i];
657 const auto len = to_remove_len[i];
658
659 auto* pos = nameTmp;
660 while ((pos = strstr(pos, str)) != nullptr) {
661 const bool isBoundary = pos == nameTmp || pos[-1] == '<' || pos[-1] == ',' || pos[-1] == ' ';
662 if (!isBoundary) {
663 ++pos;
664 continue;
665 }
666
667 const auto tailMaxLen = (size_t)(MaxNameLength - (uint32_t)(pos + len - nameTmp));
668 memmove(pos, pos + len, GAIA_STRLEN(pos + len, tailMaxLen) + 1);
669 nameTmpLen -= len;
670 }
671 }
672
673 return nameTmpLen;
674 }

752 GAIA_NODISCARD static ComponentCacheItem* create(Entity entity, const ecs::ComponentDesc& desc) {
753 GAIA_ASSERT(!desc.name.empty());
754 GAIA_ASSERT(desc.name.size() < MaxNameLength);
755 GAIA_ASSERT(desc.size < Component::MaxComponentSizeInBytes);
756 GAIA_ASSERT((desc.size == 0 && desc.alig == 0) || (desc.alig > 0 && desc.alig < Component::MaxAlignment));
757 GAIA_ASSERT(desc.alig == 0 || (desc.alig & (desc.alig - 1)) == 0);
758 GAIA_ASSERT(desc.soa <= meta::StructToTupleMaxTypes);
759 GAIA_ASSERT(desc.soa == 0 || desc.pSoaSizes != nullptr);
760#if GAIA_ASSERT_ENABLED
761 if (desc.soa > 0) {
762 uint32_t soaSize = 0;
763 GAIA_FOR(desc.soa) {
764 GAIA_ASSERT(desc.pSoaSizes[i] != 0);
765 soaSize += desc.pSoaSizes[i];
766 }
767 GAIA_ASSERT(soaSize <= desc.size);
768 }
769#endif
770
771 auto* cci = new ComponentCacheItem();
772 cci->entity = entity;
773 cci->comp = Component(entity.id(), desc.soa, desc.size, desc.alig, desc.storageType);
774 cci->hashLookup = desc.hashLookup.hash != 0
775 ? desc.hashLookup
776 : ComponentLookupHash{core::calculate_hash64(desc.name.data(), desc.name.size())};
777
778 if (desc.soa > 0) {
779 GAIA_FOR(desc.soa) cci->soaSizes[i] = desc.pSoaSizes[i];
780 }
781
782 init_name(cci->name, desc.name);
783
784 cci->func_ctor = desc.funcCtor;
785 cci->func_move_ctor = desc.funcMoveCtor;
786 cci->func_copy_ctor = desc.funcCopyCtor;
787 cci->func_dtor = desc.funcDtor;
788 cci->func_copy = desc.funcCopy;
789 cci->func_move = desc.funcMove;
790 cci->func_swap = desc.funcSwap;
791 cci->func_cmp = desc.funcCmp;
792 cci->func_save = desc.funcSave;
