512 GAIA_NODISCARD uint32_t calc_new_mem_offset(uint32_t addr, size_t cnt) const noexcept {
513 if (comp.soa() == 0) {
514 addr = (uint32_t)mem::detail::get_aligned_byte_offset(addr, comp.alig(), comp.size(), cnt);
515 } else {
516 GAIA_FOR(comp.soa()) {
517 addr = (uint32_t)mem::detail::get_aligned_byte_offset(addr, comp.alig(), soaSizes[i], cnt);
518 }
519 // TODO: Magic offset. Otherwise, SoA data might leak past the chunk boundary when accessing
520 // the last element. By faking the memory offset we can bypass this is issue for now.
521 // Obviously, this needs fixing at some point.
522 addr += comp.soa() * 12;
523 }
524 return addr;

532 template <typename T>
533 GAIA_NODISCARD static uint32_t init_type_name(char (&nameTmp)[MaxNameLength]) {
534 auto ct_name = detail::ComponentDesc<T>::name();
535
536 // Allocate enough memory for the name string + the null-terminating character (
537 // the compile time string returned by ComponentDesc<T>::name is not null-terminated).
538 // Different compilers will give a bit different strings, e.g.:
539 // Clang/GCC: gaia::ecs::uni<Position>
540 // MSVC : gaia::ecs::uni<struct Position>
541 // Therefore, we first copy the compile-time string and then tweak it so it is
542 // the same on all supported compilers.
543 auto nameTmpLen = (uint32_t)ct_name.size();
544 GAIA_ASSERT(nameTmpLen < MaxNameLength);
545 memcpy((void*)nameTmp, (const void*)ct_name.data(), nameTmpLen);
546 nameTmp[ct_name.size()] = 0;
547
548 auto strip_prefix = [&](const char* prefix, uint32_t prefixLen) {
549 if (nameTmpLen <= prefixLen || strncmp(nameTmp, prefix, prefixLen) != 0)
550 return;
551
552 memmove(nameTmp, nameTmp + prefixLen, nameTmpLen - prefixLen + 1);
553 nameTmpLen -= prefixLen;
554 };
555
556 strip_prefix("const ", 6);
557
558 const uint32_t NSubstrings = 3;
559 const char* to_remove[NSubstrings] = {"class ", "struct ", "enum "};
560 const uint32_t to_remove_len[NSubstrings] = {6, 7, 5};
561 GAIA_FOR(NSubstrings) {
562 strip_prefix(to_remove[i], to_remove_len[i]);
563 }
564
565 while (nameTmpLen > 0) {
566 const auto ch = nameTmp[nameTmpLen - 1];
567 if (ch != ' ' && ch != '&' && ch != '*')
568 break;
569
570 nameTmp[--nameTmpLen] = 0;
571 }
572
573 if (nameTmpLen > 6 && strncmp(nameTmp + nameTmpLen - 6, " const", 6) == 0) {
574 nameTmpLen -= 6;
575 nameTmp[nameTmpLen] = 0;
576 }
577
578 // Normalization template names by removing keywords when they appear as template argument
579 // prefixes instead of as part of a longer identifier.
580 GAIA_FOR(NSubstrings) {
581 const auto* str = to_remove[i];
582 const auto len = to_remove_len[i];
583
584 auto* pos = nameTmp;
585 while ((pos = strstr(pos, str)) != nullptr) {
586 const bool isBoundary = pos == nameTmp || pos[-1] == '<' || pos[-1] == ',' || pos[-1] == ' ';
587 if (!isBoundary) {
588 ++pos;
589 continue;
590 }
591
592 const auto tailMaxLen = (size_t)(MaxNameLength - (uint32_t)(pos + len - nameTmp));
593 memmove(pos, pos + len, GAIA_STRLEN(pos + len, tailMaxLen) + 1);
594 nameTmpLen -= len;
595 }
596 }
597
598 return nameTmpLen;
599 }

649 GAIA_NODISCARD static ComponentCacheItem* create(Entity entity) {
650 static_assert(core::is_raw_v<T>);
651
652 constexpr auto componentSize = detail::ComponentDesc<T>::size();
653 static_assert(
654 componentSize < Component::MaxComponentSizeInBytes,
655 "Trying to register a component larger than the maximum allowed component size! In the future this "
656 "restriction won't apply to components not stored inside archetype chunks.");
657
658 char nameTmp[MaxNameLength];
659 const auto nameTmpLen = init_type_name<T>(nameTmp);
660
661 uint8_t soaSizes[meta::StructToTupleMaxTypes]{};
662 RuntimeFieldDesc fields[meta::StructToTupleMaxTypes]{};
663 auto desc = detail::ComponentDesc<T>::make(
664 util::str_view(nameTmp, nameTmpLen), std::span<uint8_t, meta::StructToTupleMaxTypes>{soaSizes});
665#if GAIA_ECS_AUTO_COMPONENT_FIELDS
666 desc.fields = fields;
667 desc.fieldCount =
668 detail::ComponentDesc<T>::auto_fields(std::span<RuntimeFieldDesc, meta::StructToTupleMaxTypes>{fields});
669#endif
670 return create(entity, desc);
671 }

677 GAIA_NODISCARD static ComponentCacheItem* create(Entity entity, const ecs::ComponentDesc& desc) {
678 GAIA_ASSERT(!desc.name.empty());
679 GAIA_ASSERT(desc.name.size() < MaxNameLength);
680 GAIA_ASSERT(desc.size < Component::MaxComponentSizeInBytes);
681 GAIA_ASSERT((desc.size == 0 && desc.alig == 0) || (desc.alig > 0 && desc.alig < Component::MaxAlignment));
682 GAIA_ASSERT(desc.soa <= meta::StructToTupleMaxTypes);
683
684 auto* cci = new ComponentCacheItem();
685 cci->entity = entity;
686 cci->comp = Component(entity.id(), desc.soa, desc.size, desc.alig, desc.storageType);
687 cci->hashLookup = desc.hashLookup.hash != 0
688 ? desc.hashLookup
689 : ComponentLookupHash{core::calculate_hash64(desc.name.data(), desc.name.size())};
690
691 if (desc.soa > 0 && desc.pSoaSizes != nullptr) {
692 GAIA_FOR(desc.soa) cci->soaSizes[i] = desc.pSoaSizes[i];
693 }
694
695 init_name(cci->name, desc.name);
696
697 cci->func_ctor = desc.funcCtor;
698 cci->func_move_ctor = desc.funcMoveCtor;
699 cci->func_copy_ctor = desc.funcCopyCtor;
700 cci->func_dtor = desc.funcDtor;
701 cci->func_copy = desc.funcCopy;
702 cci->func_move = desc.funcMove;
703 cci->func_swap = desc.funcSwap;
704 cci->func_cmp = desc.funcCmp;
705 cci->func_save = desc.funcSave;
706 cci->func_load = desc.funcLoad;
707 cci->typeKind = desc.typeKind;
708 cci->underlyingType = desc.underlyingType;
709 cci->elementType = desc.elementType;
710 cci->elementCount = desc.elementCount;
711 cci->opaqueAsType = desc.opaqueAsType;
712 cci->sequenceAdapter = desc.sequenceAdapter;
713 cci->opaqueAdapter = desc.opaqueAdapter;
714
715 if (desc.fieldCount > 0) {
716 GAIA_FOR(desc.fieldCount) {
717 const bool copied = cci->copy_runtime_field(desc.fields[i]);
718 GAIA_ASSERT(copied);
719 }
720 }
721
722 if (desc.constantCount > 0) {
723 GAIA_FOR(desc.constantCount) {
724 const bool copied = cci->copy_runtime_constant(desc.constants[i]);
725 GAIA_ASSERT(copied);
726 }
727 }
728
729 return cci;
730 }
