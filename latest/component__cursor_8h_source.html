584 GAIA_NODISCARD CursorStatus
585 validate_primitive(Entity expectedType, bool requireWrite, bool requireScalar) const noexcept {
586 if (!m_valid)
587 return CursorStatus::Invalid;
588 const auto& current = m_stack[m_depth];
589 if (current.type != expectedType)
590 return CursorStatus::TypeMismatch;
591 if (requireScalar && current.elemCount != 1)
592 return CursorStatus::TypeMismatch;
593 if (current.size != current.elemSize * current.elemCount)
594 return CursorStatus::OutOfRange;
595 if (current.size == 0)
596 return CursorStatus::TypeMismatch;
597 if (ptr() == nullptr)
598 return CursorStatus::Invalid;
599 if (requireWrite) {
600 if (!current.writable || m_world == nullptr || m_entity == EntityBad || m_rootType == EntityBad)
601 return CursorStatus::ReadOnly;
602 if (mut_ptr() == nullptr)
603 return CursorStatus::Invalid;
604 }
605 return CursorStatus::Ok;
606 }

626 template <typename T>
627 GAIA_NODISCARD CursorResult<void> write_primitive(Entity expectedType, const T& value) noexcept {
628 const auto status = validate_primitive(expectedType, true, true);
629 if (status != CursorStatus::Ok)
630 return {status};
631 if (size() != sizeof(T))
632 return {CursorStatus::TypeMismatch};
633
634 memcpy(mut_ptr(), &value, sizeof(T));

638
639 bool descend(const RuntimeField& field) noexcept {
640 if (!m_valid || m_components == nullptr || m_depth + 1 >= MaxDepth)
641 return false;
642
643 const auto* pType = m_components->find(field.type);
644 if (pType == nullptr)
645 return false;
646
647 const auto elemSize = pType->comp.size();
648 const auto elemCount = ComponentCacheItem::field_element_count(field);
649 const auto fieldSize64 = (uint64_t)elemSize * (uint64_t)elemCount;
650 const auto end = (uint64_t)field.offset + fieldSize64;
651 if (fieldSize64 > UINT32_MAX || end > m_stack[m_depth].size)
652 return false;
653
654 Scope next{};
655 next.type = field.type;
656 next.size = (uint32_t)fieldSize64;
657 next.elemSize = elemSize;
658 next.elemCount = elemCount;
659 next.writable = m_stack[m_depth].writable;
660 if (m_stack[m_depth].data != nullptr)
661 next.data = (const uint8_t*)m_stack[m_depth].data + field.offset;
662 if (m_stack[m_depth].mutData != nullptr)
663 next.mutData = (uint8_t*)m_stack[m_depth].mutData + field.offset;
664
665 m_stack[++m_depth] = next;
666 return true;
667 }
