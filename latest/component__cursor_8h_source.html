327 GAIA_NODISCARD CursorResult<uint32_t> count() const noexcept {
328 CursorResult<uint32_t> result{};
329 if (!m_valid) {
330 result.status = CursorStatus::Invalid;
331 return result;
332 }
333
334 const auto& current = m_stack[m_depth];
335 if (current.elemCount > 1) {
336 result.status = CursorStatus::Ok;
337 result.value = current.elemCount;
338 return result;
339 }
340
341 const auto* pType = current_item();
342 if (pType == nullptr) {
343 result.status = CursorStatus::TypeMismatch;
344 return result;
345 }
346 if (pType->typeKind == RuntimeTypeKind::Array) {
347 result.status = CursorStatus::Ok;
348 result.value = pType->element_count();
349 return result;
350 }
351 if (pType->typeKind == RuntimeTypeKind::Vector) {
352 const auto* adapter = pType->sequence_adapter();
353 if (adapter == nullptr || adapter->count == nullptr) {
354 result.status = CursorStatus::TypeMismatch;
355 return result;
356 }
357 RuntimeSequenceScope sequence{current.type, current.data, current.mutData, current.size};
358 uint32_t value = 0;
359 if (!adapter->count(adapter->ctx, sequence, value)) {
360 result.status = CursorStatus::Invalid;
361 return result;
362 }
363 result.status = CursorStatus::Ok;
364 result.value = value;
365 return result;
366 }
367 result.status = CursorStatus::TypeMismatch;
368 return result;
369 }

415 bool elem(uint32_t index) noexcept {
416 if (!m_valid || m_depth + 1 >= MaxDepth)
417 return false;
418
419 const auto& current = m_stack[m_depth];
420 Scope next{};
421 if (current.elemCount > 1) {
422 if (index >= current.elemCount || current.elemSize == 0)
423 return false;
424 if ((uint64_t)current.elemSize * ((uint64_t)index + 1) > current.size)
425 return false;
426
427 next.type = current.type;
428 next.size = current.elemSize;
429 next.elemSize = current.elemSize;
430 next.elemCount = 1;
431 next.sequenceAdapter = current.sequenceAdapter;
432 next.sequenceDepth = current.sequenceDepth;
433 next.elementDepth = current.elementDepth;
434 next.elementToken = current.elementToken;
435 next.opaqueAdapter = current.opaqueAdapter;
436 next.opaqueDepth = current.opaqueDepth;
437 next.opaqueValueDepth = current.opaqueValueDepth;
438 next.opaqueToken = current.opaqueToken;
439 next.writable = current.writable;
440 if (current.data != nullptr)
441 next.data = (const uint8_t*)current.data + (uint64_t)current.elemSize * index;
442 if (current.mutData != nullptr)
443 next.mutData = (uint8_t*)current.mutData + (uint64_t)current.elemSize * index;
444 } else {
445 const auto* pType = m_components != nullptr ? m_components->find(current.type) : nullptr;
446 if (pType == nullptr)
447 return false;
448
449 if (pType->typeKind == RuntimeTypeKind::Vector) {
450 const auto* adapter = pType->sequence_adapter();
451 if (adapter == nullptr || adapter->count == nullptr || adapter->element == nullptr)
452 return false;
453 RuntimeSequenceScope sequence{current.type, current.data, current.mutData, current.size};
454 uint32_t elemCount = 0;
455 if (!adapter->count(adapter->ctx, sequence, elemCount) || index >= elemCount)
456 return false;
457 RuntimeSequenceElement element{};
458 element.type = pType->element_type();
459 if (!adapter->element(adapter->ctx, sequence, index, element))
460 return false;
461 if (element.type == EntityBad)
462 element.type = pType->element_type();
463 if (element.type == EntityBad || element.size == 0 || element.data == nullptr)
464 return false;
465
466 next.type = element.type;
467 next.data = element.data;
468 next.mutData = element.mutData;
469 next.size = element.size;
470 next.elemSize = element.size;
471 next.elemCount = 1;
472 next.sequenceAdapter = adapter;
473 next.sequenceDepth = m_depth;
474 next.elementDepth = m_depth + 1;
475 next.elementToken = element.token;
476 next.writable = current.writable && element.mutData != nullptr;
477 } else if (pType->typeKind == RuntimeTypeKind::Array) {
478
479 const auto elemCount = pType->element_count();
480 const auto elementType = pType->element_type();
481 const auto* pElementType = m_components->find(elementType);
482 if (pElementType == nullptr || elemCount == 0 || index >= elemCount)
483 return false;
484
485 const auto elemSize = pElementType->comp.size();
486 if (elemSize == 0)
487 return false;
488 if ((uint64_t)elemSize * ((uint64_t)index + 1) > current.size)
489 return false;
490
491 next.type = elementType;
492 next.size = elemSize;
493 next.elemSize = elemSize;
494 next.elemCount = 1;
495 next.writable = current.writable;
496 if (current.data != nullptr)
497 next.data = (const uint8_t*)current.data + (uint64_t)elemSize * index;
498 if (current.mutData != nullptr)
499 next.mutData = (uint8_t*)current.mutData + (uint64_t)elemSize * index;
500 } else {
501 return false;
502 }
503 }
504
505 m_stack[++m_depth] = next;
506 return true;
507 }

510 CursorResult<void> resize(uint32_t count) noexcept {
511 if (!m_valid)
512 return {CursorStatus::Invalid};
513 const auto* pType = current_item();
514 if (pType == nullptr || pType->typeKind != RuntimeTypeKind::Vector)
515 return {CursorStatus::TypeMismatch};
516 const auto* adapter = pType->sequence_adapter();
517 if (adapter == nullptr || adapter->resize == nullptr)
518 return {CursorStatus::TypeMismatch};
519 if (!m_stack[m_depth].writable || m_world == nullptr || m_entity == EntityBad || m_rootType == EntityBad)
520 return {CursorStatus::ReadOnly};
521 RuntimeSequenceScope sequence{
522 m_stack[m_depth].type, m_stack[m_depth].data, m_stack[m_depth].mutData, m_stack[m_depth].size};
523 if (!adapter->resize(adapter->ctx, sequence, count))
524 return {CursorStatus::OutOfRange};
525 world_finish_write(*m_world, m_rootType, m_entity);
526 return {CursorStatus::Ok};
527 }

757 GAIA_NODISCARD CursorResult<uint32_t> get_raw(void* data, uint32_t byteCount) const noexcept {
758 CursorResult<uint32_t> result{};
759 if (!m_valid) {
760 result.status = CursorStatus::Invalid;
761 return result;
762 }
763 if (byteCount < size()) {
764 result.status = CursorStatus::OutOfRange;
765 return result;
766 }
767 if (size() != 0 && (data == nullptr || ptr() == nullptr)) {
768 result.status = CursorStatus::Invalid;
769 return result;
770 }
771 if (size() != 0)
772 memcpy(data, ptr(), size());
773 result.status = CursorStatus::Ok;
774 result.value = size();
775 return result;
776 }

787 CursorResult<void> set_raw(const void* data, uint32_t byteCount) noexcept {
788 if (!m_valid)
789 return {CursorStatus::Invalid};
790 if (!m_stack[m_depth].writable || m_world == nullptr || m_entity == EntityBad || m_rootType == EntityBad)
791 return {CursorStatus::ReadOnly};
792 if (byteCount != size())
793 return {CursorStatus::OutOfRange};
794 if (byteCount == 0)
795 return {CursorStatus::Ok};
796 if (data == nullptr || mut_ptr() == nullptr)
797 return {CursorStatus::Invalid};
798
799 memcpy(mut_ptr(), data, byteCount);
800 if (!commit_current_sequence_element())
801 return {CursorStatus::Invalid};
802 world_finish_write(*m_world, m_rootType, m_entity);
803 return {CursorStatus::Ok};
804 }

859 bool project_opaque_scope(const ComponentCacheItem& item, Scope& out) const noexcept {
860 const auto* adapter = item.opaque_adapter();
861 const auto* pSemantic = opaque_semantic_item(item);
862 if (adapter == nullptr || adapter->project == nullptr || pSemantic == nullptr)
863 return false;
864 RuntimeOpaqueScope opaque = make_opaque_scope(m_stack[m_depth]);
865 RuntimeOpaqueValue value{};
866 value.type = item.opaque_as_type();
867 if (!adapter->project(adapter->ctx, opaque, value))
868 return false;
869 if (value.type == EntityBad)
870 value.type = item.opaque_as_type();
871 if (value.type != item.opaque_as_type() || value.data == nullptr || value.size != pSemantic->comp.size())
872 return false;
873
874 out = {};
875 out.type = value.type;
876 out.data = value.data;
877 out.mutData = value.mutData;
878 out.size = value.size;
879 out.elemSize = value.size;
880 out.elemCount = 1;
881 out.opaqueAdapter = adapter;
882 out.opaqueDepth = m_depth;
883 out.opaqueValueDepth = m_depth + 1;

939 bool commit_current_sequence_element() noexcept {
940 const auto& current = m_stack[m_depth];
941 if (current.sequenceAdapter != nullptr && current.sequenceAdapter->commitElement != nullptr) {
942 if (current.sequenceDepth >= MaxDepth || current.elementDepth >= MaxDepth)
943 return false;
944 RuntimeSequenceScope sequence = make_sequence_scope(m_stack[current.sequenceDepth]);
945 RuntimeSequenceElement element = make_sequence_element(m_stack[current.elementDepth]);
946 if (!current.sequenceAdapter->commitElement(current.sequenceAdapter->ctx, sequence, element))
947 return false;
948 }
949

951 if (current.opaqueDepth >= MaxDepth || current.opaqueValueDepth >= MaxDepth)
952 return false;
953 RuntimeOpaqueScope opaque = make_opaque_scope(m_stack[current.opaqueDepth]);
954 RuntimeOpaqueValue value = make_opaque_value(m_stack[current.opaqueValueDepth]);
955 if (!current.opaqueAdapter->commit(current.opaqueAdapter->ctx, opaque, value))
956 return false;

961 GAIA_NODISCARD CursorStatus
962 validate_primitive(Entity expectedType, bool requireWrite, bool requireScalar) const noexcept {
963 if (!m_valid)
964 return CursorStatus::Invalid;
965 const auto& current = m_stack[m_depth];
966 if (current.type != expectedType) {
967 const auto* pCurrentType = m_components != nullptr ? m_components->find(current.type) : nullptr;
968 if (pCurrentType == nullptr || pCurrentType->primitive_type() != expectedType)
969 return CursorStatus::TypeMismatch;
970 }
971 if (requireScalar && current.elemCount != 1)
972 return CursorStatus::TypeMismatch;
973 if (current.size != current.elemSize * current.elemCount)
974 return CursorStatus::OutOfRange;
975 if (current.size == 0)
976 return CursorStatus::TypeMismatch;
977 if (ptr() == nullptr)
978 return CursorStatus::Invalid;
979 if (requireWrite) {
980 if (!current.writable || m_world == nullptr || m_entity == EntityBad || m_rootType == EntityBad)
981 return CursorStatus::ReadOnly;
982 if (mut_ptr() == nullptr)
983 return CursorStatus::Invalid;
984 }
985 return CursorStatus::Ok;
986 }

988 template <typename T>
989 GAIA_NODISCARD CursorResult<T> read_primitive(Entity expectedType) const noexcept {
990 CursorResult<T> result{};
991 const auto status = validate_primitive(expectedType, false, true);
992 if (status != CursorStatus::Ok) {
993 result.status = status;
994 return result;
995 }
996 if (size() != sizeof(T)) {
997 result.status = CursorStatus::TypeMismatch;
998 return result;
999 }
1000
1001 memcpy(&result.value, ptr(), sizeof(T));
1002 result.status = CursorStatus::Ok;
1003 return result;
1004 }

1006 template <typename T>
1007 GAIA_NODISCARD CursorResult<void> write_primitive(Entity expectedType, const T& value) noexcept {
1008 const auto status = validate_primitive(expectedType, true, true);
1009 if (status != CursorStatus::Ok)
1010 return {status};
1011 if (size() != sizeof(T))
1012 return {CursorStatus::TypeMismatch};
1013
1014 memcpy(mut_ptr(), &value, sizeof(T));
1015 if (!commit_current_sequence_element())
1016 return {CursorStatus::Invalid};
1017 world_finish_write(*m_world, m_rootType, m_entity);
1018 return {CursorStatus::Ok};

1021 bool descend(const RuntimeField& field, uint32_t fieldIdx = UINT32_MAX) noexcept {
1022 if (!m_valid || m_components == nullptr || m_depth + 1 >= MaxDepth)
1023 return false;
1024
1025 const auto* pType = m_components->find(field.type);
1026 if (pType == nullptr)
1027 return false;
1028
1029 Entity scopeType = field.type;
1030 uint32_t elemSize = pType->comp.size();
1031 uint32_t elemCount = ComponentCacheItem::field_element_count(field);
1032 if (pType->typeKind == RuntimeTypeKind::Array) {
1033 if (field.count != 0)
1034 return false;
1035 const auto elementType = pType->element_type();
1036 const auto* pElementType = m_components->find(elementType);
1037 if (pElementType == nullptr || pType->element_count() == 0)
1038 return false;
1039 scopeType = elementType;
1040 elemSize = pElementType->comp.size();
1041 elemCount = pType->element_count();
1042 }
1043 const auto fieldSize64 = (uint64_t)elemSize * (uint64_t)elemCount;
1044 const auto end = (uint64_t)field.offset + fieldSize64;
1045 if (fieldSize64 > UINT32_MAX || end > m_stack[m_depth].size)
1046 return false;
1047
1048 Scope next{};
1049 next.type = scopeType;
1050 next.size = (uint32_t)fieldSize64;
1051 next.elemSize = elemSize;
1052 next.elemCount = elemCount;
1053 next.sequenceAdapter = m_stack[m_depth].sequenceAdapter;
1054 next.sequenceDepth = m_stack[m_depth].sequenceDepth;
1055 next.elementDepth = m_stack[m_depth].elementDepth;
1056 next.elementToken = m_stack[m_depth].elementToken;
1057 next.opaqueAdapter = m_stack[m_depth].opaqueAdapter;
1058 next.opaqueDepth = m_stack[m_depth].opaqueDepth;
1059 next.opaqueValueDepth = m_stack[m_depth].opaqueValueDepth;
1060 next.opaqueToken = m_stack[m_depth].opaqueToken;
1061 next.writable = m_stack[m_depth].writable;
1062 if (m_rootSoa && m_depth == 0) {
1063 if (fieldIdx == UINT32_MAX || m_readWorld == nullptr)
1064 return false;
1065
1066 if (m_world != nullptr) {
1067 const auto view = world_mut_raw_field(*m_world, m_entity, m_rootType, fieldIdx);
1068 if (!view.valid() || view.size != next.size)
1069 return false;
1070 next.data = view.data;
1071 next.mutData = view.data;
1072 } else {
1073 const auto view = world_get_raw_field(*m_readWorld, m_entity, m_rootType, fieldIdx);
1074 if (!view.valid() || view.size != next.size)
1075 return false;
1076 next.data = view.data;
1077 }
1078 } else {
1079 if (m_stack[m_depth].data != nullptr)
1080 next.data = (const uint8_t*)m_stack[m_depth].data + field.offset;
1081 if (m_stack[m_depth].mutData != nullptr)
1082 next.mutData = (uint8_t*)m_stack[m_depth].mutData + field.offset;
1083 }
1084
1085 m_stack[++m_depth] = next;
1086 return true;
1087 }
