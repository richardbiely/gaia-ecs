204 inline mt::JobPriority exec_prio(QueryExecType execType) {
205 // QueryExecType::ParallelEff is encoded as value 3. Keep the scheduler bridge independent
206 // from the enum definition point so this header stays C-like and forward-declarable.
207 return (uint32_t)execType == 3U ? mt::JobPriority::Low : mt::JobPriority::High;

230 inline SchedToken add_one_def([[maybe_unused]] void* pCtx, const SchedTaskDesc* pDesc) {
231 GAIA_ASSERT(pDesc != nullptr);
232 GAIA_ASSERT(pDesc->invoke != nullptr);
233 if (pDesc == nullptr || pDesc->invoke == nullptr)
234 return {};
235
236 auto* pData = new SchedTokenDefData();
237 pData->kind = SchedTokenKind::Single;
238 pData->handles.resize(1);
239
240 mt::Job job;
241 job.priority = exec_prio(pDesc->execType);
242 job.flags = mt::JobCreationFlags::ManualDelete;
243 job.func = [pCtx = pDesc->pCtx, invoke = pDesc->invoke]() {
244 invoke(pCtx);
245 };
246
247 pData->handles[0] = mt::ThreadPool::get().add(GAIA_MOV(job));
248 return make_sched_token(pData);
249 }

251 inline SchedToken sched_one_def([[maybe_unused]] void* pCtx, const SchedTaskDesc* pDesc) {
252 auto token = add_one_def(pCtx, pDesc);
253 auto* pData = sched_token_data(token);
254 if (pData != nullptr) {
255 mt::ThreadPool::get().submit(pData->handles[0]);
256 pData->submitted = true;
257 }
258 return token;
259 }

261 inline SchedToken add_par_def([[maybe_unused]] void* pCtx, const SchedParDesc* pDesc) {
262 GAIA_ASSERT(pDesc != nullptr);
263 GAIA_ASSERT(pDesc->invoke != nullptr);
264 if (pDesc == nullptr || pDesc->invoke == nullptr || pDesc->itemCount == 0)
265 return {};
266
267 auto& tp = mt::ThreadPool::get();
268 const auto prio = exec_prio(pDesc->execType);
269 uint32_t groupSize = pDesc->groupSize;
270 if (groupSize == 0) {
271 const auto workers = core::get_max(1U, tp.workers() + 1U);
272 groupSize = (pDesc->itemCount + workers - 1) / workers;
273 constexpr uint32_t maxUnitsOfWorkPerGroup = 8;
274 groupSize = groupSize / maxUnitsOfWorkPerGroup;
275 if (groupSize == 0)
276 groupSize = 1;
277 }
278
279 const auto jobs = (pDesc->itemCount + groupSize - 1) / groupSize;
280 auto* pData = new SchedTokenDefData();
281 pData->kind = SchedTokenKind::Parallel;
282 pData->handles.resize(jobs + 1);
283
284 for (uint32_t jobIndex = 0; jobIndex < jobs; ++jobIndex) {
285 const uint32_t idxStart = jobIndex * groupSize;
286 const uint32_t idxEnd = core::get_min(idxStart + groupSize, pDesc->itemCount);
287
288 mt::Job job;
289 job.priority = prio;
290 job.flags = mt::JobCreationFlags::ManualDelete;
291 job.func = [desc = *pDesc, idxStart, idxEnd]() {
292 desc.invoke(desc.pCtx, idxStart, idxEnd);
293 };
294 pData->handles[jobIndex] = tp.add(GAIA_MOV(job));
295 }
296 {
297 mt::Job syncJob;
298 syncJob.priority = prio;
299 syncJob.flags = mt::JobCreationFlags::ManualDelete;
300 pData->handles[jobs] = tp.add(GAIA_MOV(syncJob));
301 }
302 tp.dep(std::span(pData->handles.data(), jobs), pData->handles[jobs]);
303 return make_sched_token(pData);
304 }

306 inline SchedToken sched_par_def([[maybe_unused]] void* pCtx, const SchedParDesc* pDesc) {
307 auto token = add_par_def(pCtx, pDesc);
308 auto* pData = sched_token_data(token);
309 if (pData != nullptr) {

311 pData->submitted = true;
312 }

316 inline void sched_submit_def([[maybe_unused]] void* pCtx, SchedToken token) {
317 auto* pData = sched_token_data(token);
318 if (pData == nullptr || pData->submitted)
319 return;

321 tp.submit(std::span(pData->handles.data(), pData->handles.size()));

325 inline void sched_dep_def([[maybe_unused]] void* pCtx, SchedToken tokenFirst, SchedToken tokenSecond) {
326 auto* pFirst = sched_token_data(tokenFirst);
327 auto* pSecond = sched_token_data(tokenSecond);
328 if (pFirst == nullptr || pSecond == nullptr || pFirst->handles.empty() || pSecond->handles.empty())
329 return;
330 auto& tp = mt::ThreadPool::get();
331 const auto firstDone = pFirst->handles.back();
332 if (pSecond->kind == SchedTokenKind::Parallel && pSecond->handles.size() > 1) {
333 const auto childCount = (uint32_t)pSecond->handles.size() - 1;
334 for (uint32_t i = 0; i < childCount; ++i)
335 tp.dep(firstDone, pSecond->handles[i]);
336 return;
337 }
338 tp.dep(firstDone, pSecond->handles.back());
339 }

341 inline void sched_wait_def([[maybe_unused]] void* pCtx, SchedToken token) {
342 auto* pData = sched_token_data(token);
343 if (pData == nullptr || pData->handles.empty())
344 return;
345 mt::ThreadPool::get().wait(pData->handles.back());
346 }

348 inline void sched_del_def([[maybe_unused]] void* pCtx, [[maybe_unused]] SchedToken token) {
349 auto* pData = sched_token_data(token);
350 if (pData == nullptr)
351 return;
352 auto& tp = mt::ThreadPool::get();
353 for (auto handle: pData->handles) {
354 if (handle != mt::JobNull)
355 tp.del(handle);
356 }
357 delete pData;
358 }

526 inline void SchedJob::dep(const SchedJob& jobFirst) {
527 if (!m_valid || !jobFirst.m_valid)
528 return;
529 GAIA_ASSERT(same_sched(m_sched, jobFirst.m_sched));
530 GAIA_ASSERT(!m_submitted && !jobFirst.m_submitted);
531 if (!same_sched(m_sched, jobFirst.m_sched) || m_submitted || jobFirst.m_submitted)
532 return;
533 sched_dep(m_sched, jobFirst.m_token, m_token);
534 }
