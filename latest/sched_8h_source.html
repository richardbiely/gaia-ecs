73 inline mt::JobPriority exec_prio(QueryExecType execType) {
74 // QueryExecType::ParallelEff is encoded as value 3. Keep the scheduler bridge independent
75 // from the enum definition point so this header stays C-like and forward-declarable.
76 return (uint32_t)execType == 3U ? mt::JobPriority::Low : mt::JobPriority::High;
77 }

79 inline SchedToken sched_one_def([[maybe_unused]] void* pCtx, const SchedTaskDesc* pDesc) {
80 GAIA_ASSERT(pDesc != nullptr);
81 GAIA_ASSERT(pDesc->invoke != nullptr);
82 if (pDesc == nullptr || pDesc->invoke == nullptr)
83 return {};
84
85 mt::Job job;
86 job.priority = exec_prio(pDesc->execType);
87 job.func = [pCtx = pDesc->pCtx, invoke = pDesc->invoke]() {
88 invoke(pCtx);
89 };
90
91 const auto handle = mt::ThreadPool::get().sched(GAIA_MOV(job));
92 SchedToken token{};
93 token.value[0] = (uintptr_t)handle.value();
94 return token;
95 }
96
97 inline SchedToken sched_par_def([[maybe_unused]] void* pCtx, const SchedParDesc* pDesc) {
98 GAIA_ASSERT(pDesc != nullptr);
99 GAIA_ASSERT(pDesc->invoke != nullptr);
100 if (pDesc == nullptr || pDesc->invoke == nullptr || pDesc->itemCount == 0)
101 return {};
102
103 mt::JobParallelRef job{};
104 job.pCtx = const_cast<SchedParDesc*>(pDesc);
105 job.priority = exec_prio(pDesc->execType);
106 job.invoke = [](void* pCtx, const mt::JobArgs& args) {
107 auto& desc = *(const SchedParDesc*)pCtx;
108 desc.invoke(desc.pCtx, args.idxStart, args.idxEnd);
109 };
110
111 const auto handle = mt::ThreadPool::get().sched_par(job, pDesc->itemCount, pDesc->groupSize);
112 SchedToken token{};
113 token.value[0] = (uintptr_t)handle.value();
114 return token;
115 }

141 GAIA_NODISCARD inline const Sched& sched_resolve(const Sched& sched) {

151 GAIA_NODISCARD inline SchedToken sched_one(const Sched& sched, const SchedTaskDesc& desc) {
