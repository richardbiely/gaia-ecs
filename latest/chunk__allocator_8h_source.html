30 static constexpr uint32_t MemoryBlockAlignment = 64;

38 constexpr uint16_t mem_block_size(uint32_t sizeType) {
39 constexpr uint16_t sizes[] = {MinMemoryBlockSize, MinMemoryBlockSize * 2, MaxMemoryBlockSize};
40 return sizes[sizeType];

43 constexpr uint8_t mem_block_size_type(uint32_t sizeBytes) {
44 GAIA_ASSERT(sizeBytes > 0);
45 // Ceil division by smallest block size
46 const uint32_t blocks = (sizeBytes + MinMemoryBlockSize - 1) / MinMemoryBlockSize;
47 return blocks > 2 ? 2 : static_cast<uint8_t>(blocks - 1);
48 }

125 void write_block_idx(uint32_t blockIdx, uint32_t value) {
126 const uint32_t bitPosition = blockIdx * NBlocks_Bits;
127
128 GAIA_ASSERT(bitPosition < NBlocks * NBlocks_Bits);
129 GAIA_ASSERT(value <= InvalidBlockId);
130
131 BitView{{(uint8_t*)m_blocks.data(), BlockArrayBytes}}.set(bitPosition, (uint8_t)value);
132 }

135 const uint32_t bitPosition = blockIdx * NBlocks_Bits;
136
137 GAIA_ASSERT(bitPosition < NBlocks * NBlocks_Bits);
138
139 return BitView{{(uint8_t*)m_blocks.data(), BlockArrayBytes}}.get(bitPosition);

142 GAIA_NODISCARD void* alloc_block(
143 #if GAIA_DEBUG
144 uint32_t bytesWanted
145 #endif
146 ) {
147 auto StoreBlockAddress = [&](uint32_t index) {
148 // Encode info about chunk's page in the memory block.
149 // The actual pointer returned is offset by MemoryBlockUsableOffset bytes
150 auto* pMemoryBlock = (uint8_t*)m_data + (index * mem_block_size(m_sizeType));
151 GAIA_ASSERT((uintptr_t)pMemoryBlock % MemoryBlockAlignment == 0);
152 auto& header = block_header(pMemoryBlock);
153 header.m_pageAddr = (uintptr_t)this;
154 header.m_reserved = 0;
155 #if GAIA_DEBUG
156 header.m_requestedBytes = bytesWanted;
157 #endif
158 return (void*)(pMemoryBlock + MemoryBlockUsableOffset);
159 };
160
161 // We don't want to go out of range for new blocks
162 GAIA_ASSERT(!full() && "Trying to allocate too many blocks!");
163
164 uint32_t index = 0;
165 if (m_freeBlocks == 0U) {
166 index = m_blockCnt;
167 ++m_usedBlocks;
168 ++m_blockCnt;
169 write_block_idx(index, index);
170 } else {
171 GAIA_ASSERT(m_nextFreeBlock < m_blockCnt && "Block allocator recycle list broken!");
172
173 ++m_usedBlocks;
174 --m_freeBlocks;
175
176 index = m_nextFreeBlock;
177 m_nextFreeBlock = read_block_idx(m_nextFreeBlock);
178 }
179
180 #if GAIA_ASSERT_ENABLED
181 GAIA_ASSERT((m_usedMask & (uint64_t(1) << index)) == 0 && "Block already marked as live");
182 m_usedMask |= uint64_t(1) << index;
183 #endif
184
185 return StoreBlockAddress(index);
186 }

188 void free_block(void* pBlock) {
189 GAIA_ASSERT(pBlock != nullptr);
190 GAIA_ASSERT(m_usedBlocks > 0);
191 GAIA_ASSERT(m_freeBlocks <= NBlocks);
192
193 // Offset the chunk memory so we get the real block address
194 const auto* pMemoryBlock = (uint8_t*)pBlock - MemoryBlockUsableOffset;
195 const auto blckAddr = (uintptr_t)pMemoryBlock;
196 GAIA_ASSERT(blckAddr % MemoryBlockAlignment == 0);
197 const auto dataAddr = (uintptr_t)m_data;
198 const auto blockSize = (uintptr_t)mem_block_size(m_sizeType);
199 const auto pageSize = blockSize * NBlocks;
200 GAIA_ASSERT(blckAddr >= dataAddr);
201 GAIA_ASSERT(blckAddr < dataAddr + pageSize);
202 GAIA_ASSERT((blckAddr - dataAddr) % blockSize == 0);
203 const auto blockIdx = (uint32_t)((blckAddr - dataAddr) / blockSize);
204 GAIA_ASSERT(blockIdx < m_blockCnt);
205
206 #if GAIA_DEBUG
207 auto& header = block_header((void*)pMemoryBlock);
208 GAIA_ASSERT(header.m_requestedBytes > 0);
209 #endif
210 #if GAIA_ASSERT_ENABLED
211 GAIA_ASSERT((m_usedMask & (uint64_t(1) << blockIdx)) != 0 && "Double free or corrupted block state");
212 m_usedMask &= ~(uint64_t(1) << blockIdx);
213 #endif
214
215 #if GAIA_DEBUG
216 header.m_requestedBytes = 0;
217 std::memset(pBlock, FreedBlockPattern, blockSize - MemoryBlockUsableOffset);
218 #endif
219
220 // Update our implicit list
221 if (m_freeBlocks == 0U)
222 write_block_idx(blockIdx, InvalidBlockId);
223 else
224 write_block_idx(blockIdx, m_nextFreeBlock);
225 m_nextFreeBlock = blockIdx;
226
227 ++m_freeBlocks;
228 --m_usedBlocks;
229 }

243 void verify() const {
244 #if GAIA_ASSERT_ENABLED
245 GAIA_ASSERT(m_sizeType < 3);
246 GAIA_ASSERT(m_blockCnt <= NBlocks);
247 GAIA_ASSERT(m_usedBlocks <= m_blockCnt);
248 GAIA_ASSERT(m_freeBlocks <= m_blockCnt);
249 GAIA_ASSERT(m_usedBlocks + m_freeBlocks == m_blockCnt);
250
251 const auto blockSize = (uintptr_t)mem_block_size(m_sizeType);
252
253 const auto pageAddr = (uintptr_t)m_data;
254 GAIA_ASSERT(pageAddr % MemoryBlockAlignment == 0);
255
256 [[maybe_unused]] uint64_t freeMask = 0;
257
258 if (m_freeBlocks != 0) {
259 uint32_t next = m_nextFreeBlock;
260 GAIA_FOR(m_freeBlocks) {
261 GAIA_ASSERT(next < m_blockCnt);
262 #if GAIA_DEBUG
263 const auto bit = uint64_t(1) << next;
264 GAIA_ASSERT((freeMask & bit) == 0 && "Free list contains a cycle");
265 freeMask |= bit;
266 #endif
267 next = read_block_idx(next);
268 }
269
270 GAIA_ASSERT(next == InvalidBlockId);
271 }
272
273 GAIA_FOR(m_blockCnt) {
274 const auto* pMemoryBlock = (const uint8_t*)m_data + (i * blockSize);
275 const auto& header = block_header(pMemoryBlock);
276 GAIA_ASSERT(header.m_pageAddr == (uintptr_t)this);
277 GAIA_ASSERT(((uintptr_t)pMemoryBlock % MemoryBlockAlignment) == 0);
278
279 #if GAIA_DEBUG
280 const bool isFree = (freeMask & (uint64_t(1) << i)) != 0;
281 GAIA_ASSERT((header.m_requestedBytes == 0) == isFree);
282 #endif
283 }
284
285 GAIA_ASSERT((m_usedMask & freeMask) == 0);
286 const auto liveMask = m_blockCnt == 64 ? ~uint64_t(0) : ((uint64_t(1) << m_blockCnt) - 1);
287 GAIA_ASSERT((m_usedMask | freeMask) == liveMask);
288 #endif
289 }

291 #if GAIA_DEBUG
292 GAIA_NODISCARD uint64_t requested_bytes() const {
293 if (m_usedBlocks == 0)
294 return 0;
295
296 uint64_t freeMask = 0;
297 uint32_t next = m_nextFreeBlock;
298 GAIA_FOR(m_freeBlocks) {
299 GAIA_ASSERT(next < m_blockCnt);
300 const auto bit = uint64_t(1) << next;
301 GAIA_ASSERT((freeMask & bit) == 0 && "Free list contains a cycle");
302 freeMask |= bit;
303 next = read_block_idx(next);
304 }
305
306 uint64_t requested = 0;
307 GAIA_FOR(m_blockCnt) {
308 if ((freeMask & (uint64_t(1) << i)) != 0)
309 continue;
310
311 const auto* pMemoryBlock = (const uint8_t*)m_data + (i * mem_block_size(m_sizeType));
312 requested += block_header(pMemoryBlock).m_requestedBytes;
313 }
314
315 return requested;
316 }
317 #endif

382 void* alloc(uint32_t bytesWanted) {
383 GAIA_ASSERT(bytesWanted > 0);
384 GAIA_ASSERT(bytesWanted <= MaxMemoryBlockSize);
385 if (bytesWanted == 0 || bytesWanted > MaxMemoryBlockSize)
386 return nullptr;
387
388 const auto sizeType = mem_block_size_type(bytesWanted);
389 auto& container = m_pages[sizeType];
390
391 MemoryPageState prevState = MemoryPageState::Partial;
392 auto* pPage = container.pagesPartial.first;
393 if (pPage == nullptr) {
394 prevState = MemoryPageState::Empty;
395 pPage = container.pagesEmpty.first;
396 if (pPage == nullptr) {
397 prevState = MemoryPageState::Detached;
398 pPage = alloc_page(sizeType);
399 }
400 }
401
402 // Allocate a new chunk of memory
403 #if GAIA_DEBUG
404 void* pBlock = pPage->alloc_block(bytesWanted);
405 #else
406 void* pBlock = pPage->alloc_block();
407 #endif
408
409 move_page(container, pPage, prevState, state_for(*pPage));
410 verify();
411 return pBlock;
412 }

419 void free(void* pBlock) {
420 GAIA_ASSERT(pBlock != nullptr);
421 if (pBlock == nullptr)
422 return;
423
424 // Decode the page from the address
425 const auto& header = *(const MemoryBlockHeader*)((uint8_t*)pBlock - MemoryBlockUsableOffset);
426 const auto pageAddr = header.m_pageAddr;
427 GAIA_ASSERT(pageAddr % sizeof(uintptr_t) == 0);
428 #if GAIA_DEBUG
429 GAIA_ASSERT(header.m_requestedBytes > 0);
430 #endif
431 auto* pPage = (MemoryPage*)pageAddr;
432 const auto prevState = state_for(*pPage);
433
434 auto& container = m_pages[pPage->m_sizeType];
435
436 #if GAIA_ASSERT_ENABLED
437 if (prevState == MemoryPageState::Full) {
438 const auto res = container.pagesFull.has(pPage);
439 GAIA_ASSERT(res && "Memory page couldn't be found among full pages");
440 } else if (prevState == MemoryPageState::Partial) {
441 const auto res = container.pagesPartial.has(pPage);
442 GAIA_ASSERT(res && "Memory page couldn't be found among partial pages");
443 } else {
444 GAIA_ASSERT(false && "Allocated block can't belong to an empty page");
445 }
446 #endif
447
448 // Free the chunk
449 pPage->free_block(pBlock);
450
451 // Update lists
452 move_page(container, pPage, prevState, state_for(*pPage));
453 verify();
454
455 // Special handling for the allocator signaled to destroy itself
456 if (m_isDone) {
457 if (pPage->empty()) {
458 container.pagesEmpty.unlink(pPage);
459 free_page(pPage);
460 }
461
462 try_delete_this();
463 }
464 }

617 ChunkAllocatorPageStats page_stats(uint32_t sizeType) const {
618 ChunkAllocatorPageStats stats{};
619 const auto& container = m_pages[sizeType];
620 const auto blockSize = (uint64_t)mem_block_size(sizeType);
621 const auto pageSize = blockSize * MemoryPage::NBlocks;
622
623 stats.num_pages = (uint32_t)container.pagesEmpty.size() + (uint32_t)container.pagesPartial.size() +
624 (uint32_t)container.pagesFull.size();
625 stats.num_pages_free = (uint32_t)container.pagesEmpty.size() + (uint32_t)container.pagesPartial.size();
626 stats.mem_total = stats.num_pages * pageSize;
627 stats.mem_used = container.pagesFull.size() * pageSize;
628
629 #if GAIA_DEBUG
630 stats.num_pages_empty = (uint32_t)container.pagesEmpty.size();
631
632 for (const auto& page: container.pagesFull)
633 stats.mem_requested += page.requested_bytes();
634
635 for (const auto& page: container.pagesPartial) {
636 stats.mem_used += page.used_blocks_cnt() * blockSize;
637 stats.mem_requested += page.requested_bytes();
638 }
639 #else
640 for (const auto& page: container.pagesPartial)
641 stats.mem_used += page.used_blocks_cnt() * blockSize;
642 #endif
643
644 return stats;
645 }

649 void flushPages(MemoryPageContainer& container, uint32_t sizeType, bool releaseAll) {
650 const bool keepWarmPage = !releaseAll && warm_pages_to_keep(sizeType) != 0;
651 bool keptWarmPage = false;
652 for (auto it = container.pagesEmpty.begin(); it != container.pagesEmpty.end();) {
653 auto* pPage = &(*it);
654 ++it;
655
656 // Skip non-empty pages
657 if (!pPage->empty())
658 continue;
659
660 if (keepWarmPage && !keptWarmPage) {
661 keptWarmPage = true;
662 continue;
663 }
664
665 container.pagesEmpty.unlink(pPage);
666 free_page(pPage);
667 }
