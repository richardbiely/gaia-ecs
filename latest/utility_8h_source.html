222 template <typename T>
223 constexpr bool is_pow2(T number) {
224 static_assert(std::is_integral<T>::value, "is_pow2 must be used with integer types");
225
226 return (number & (number - 1)) == 0;
227 }
228
232 template <typename T>
233 constexpr T closest_pow2(T number) {
234 static_assert(std::is_integral<T>::value, "closest_pow2 must be used with integer types");
235
236 if (is_pow2(number))
237 return number;
238
239 // Collapse all bits below the highest set bit
240 number |= number >> 1;
241 number |= number >> 2;
242 number |= number >> 4;
243 if constexpr (sizeof(T) > 1)
244 number |= number >> 8;
245 if constexpr (sizeof(T) > 2)
246 number |= number >> 16;
247 if constexpr (sizeof(T) > 4)
248 number |= number >> 32;
249
250 // The result is now one less than the next power of two, so shift back
251 return number - (number >> 1);
252 }

675 template <auto FirstIdx, typename Tuple, typename Func, auto... Is>
676 void each_tuple_impl(Func func, std::integer_sequence<decltype(FirstIdx), Is...> /*no_name*/) {
677 if constexpr (
678 (std::is_invocable_v<
679 Func&&, decltype(std::tuple_element_t<FirstIdx + Is, Tuple>{}),
680 std::integral_constant<decltype(FirstIdx), Is>> &&
681 ...))
682 // func(Args&& arg, uint32_t idx)
683 (func(
684 std::tuple_element_t<FirstIdx + Is, Tuple>{},
685 std::integral_constant<decltype(FirstIdx), FirstIdx + Is>{}),
686 ...);
687 else
688 // func(Args&& arg)
689 (func(std::tuple_element_t<FirstIdx + Is, Tuple>{}), ...);

913 template <typename InputIt, typename T>
914 constexpr InputIt find(InputIt first, InputIt last, const T& value) {
915 if constexpr (std::is_pointer_v<InputIt>) {
916 auto size = distance(first, last);
917 for (decltype(size) i = 0; i < size; ++i) {
918 if (first[i] == value)
919 return &first[i];
920 }
921 } else if constexpr (is_random_iter_v<InputIt>) {
922 auto size = distance(first, last);
923 for (decltype(size) i = 0; i < size; ++i) {
924 if (*(first[i]) == value)
925 return first[i];
926 }
927 } else {
928 for (; first != last; ++first) {
929 if (*first == value)
930 return first;
931 }
932 }
933 return last;
934 }

957 template <typename InputIt, typename Func>
958 constexpr InputIt find_if(InputIt first, InputIt last, Func func) {
959 if constexpr (std::is_pointer_v<InputIt>) {
960 auto size = distance(first, last);
961 for (decltype(size) i = 0; i < size; ++i) {
962 if (func(first[i]))
963 return &first[i];
964 }
965 } else if constexpr (std::is_same_v<typename InputIt::iterator_category, core::random_access_iterator_tag>) {
966 auto size = distance(first, last);
967 for (decltype(size) i = 0; i < size; ++i) {
968 if (func(*(first[i])))
969 return first[i];
970 }
971 } else {
972 for (; first != last; ++first) {
973 if (func(*first))
974 return first;
975 }
