77 template <typename F>
78 static SmallFunc create(F&& f) {
79 using Fn = std::decay_t<F>;
80 static_assert(std::is_invocable_r_v<void, Fn&>, "SmallFunc requires a callable compatible with void()");
81 static_assert(std::is_move_constructible_v<Fn>, "Callable must be move-constructible");
82
83 SmallFunc func;
84
85 if constexpr (sizeof(Fn) <= BufferSize && alignof(Fn) <= alignof(std::max_align_t)) {
86 new (func.m_storage) Fn(GAIA_FWD(f));
87
88 func.m_func = [](Op op, SmallFunc* dst, SmallFunc* src) {
89 auto* pFn = reinterpret_cast<Fn*>(dst->m_storage);
90
91 switch (op) {
92 case Op::Invoke:
93 (*pFn)();
94 break;
95 case Op::Destroy:
96 pFn->~Fn();
97 break;
98 case Op::Move: {
99 GAIA_ASSERT(src != nullptr);
100 auto* pSrcFn = reinterpret_cast<Fn*>(src->m_storage);
101 new (dst->m_storage) Fn(GAIA_MOV(*pSrcFn));
102 pSrcFn->~Fn();
103 dst->m_func = src->m_func;
104 src->m_func = nullptr;
105 break;
106 }
107 }
108 };
109 } else {
110 static_assert(
111 alignof(Fn) <= alignof(std::max_align_t), "Over-aligned callables are not supported for SmallFunc");
112
113 constexpr bool UseSmallBlock = sizeof(Fn) <= MaxSize;
114 auto* pStorage = UseSmallBlock ? mem::SmallBlockAllocator::get().alloc((uint32_t)sizeof(Fn))
115 : mem::AllocHelper::alloc<Fn>();
116 GAIA_ASSERT((uintptr_t)pStorage % alignof(Fn) == 0);
117 auto* pFunc = new (pStorage) Fn(GAIA_FWD(f));
118 *reinterpret_cast<Fn**>(func.m_storage) = pFunc;
119
120 func.m_func = [](Op op, SmallFunc* dst, SmallFunc* src) {
121 auto*& pFn = *reinterpret_cast<Fn**>(dst->m_storage);
122
123 switch (op) {
124 case Op::Invoke:
125 GAIA_ASSERT(pFn != nullptr);
126 (*pFn)();
127 break;
128 case Op::Destroy:
129 GAIA_ASSERT(pFn != nullptr);
130 pFn->~Fn();
131 if constexpr (UseSmallBlock)
132 mem::SmallBlockAllocator::get().free(pFn);
133 else
134 mem::AllocHelper::free(pFn);
135 pFn = nullptr;
136 break;
137 case Op::Move:
138 GAIA_ASSERT(src != nullptr);
139 *reinterpret_cast<Fn**>(dst->m_storage) = *reinterpret_cast<Fn**>(src->m_storage);
140 dst->m_func = src->m_func;
141 *reinterpret_cast<Fn**>(src->m_storage) = nullptr;
142 src->m_func = nullptr;
143 break;
144 }
145 };
146 }
147
148 return func;
149 }
