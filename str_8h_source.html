607 GAIA_NODISCARD constexpr str_view trim(str_view expr) {
608 const auto len = expr.size();
609 if (len == 0)
610 return {};
611
612 uint32_t beg = 0;
613 while (beg < len && is_whitespace(expr.data()[beg]))
614 ++beg;
615 if (beg == len)
616 return {};
617
618 uint32_t end = len - 1;
619 while (end > beg && is_whitespace(expr.data()[end]))
620 --end;
621 return str_view(expr.data() + beg, end - beg + 1);
622 }

627 GAIA_NODISCARD constexpr std::span<const char> trim(std::span<const char> expr) {
628 const auto trimmed = trim(str_view(expr.data(), (uint32_t)expr.size()));
629 return std::span<const char>(trimmed.data(), trimmed.size());
