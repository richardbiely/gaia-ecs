89 struct QueryCmd_AddItem {
90 static constexpr QueryCmdType Id = QueryCmdType::ADD_ITEM;
91 static constexpr bool InvalidatesHash = true;
92
93 QueryInput item;
94
95 void exec(QueryCtx& ctx) const {
96 auto& ctxData = ctx.data;
97
98#if GAIA_DEBUG
99 // Unique component ids only
100 GAIA_ASSERT(!core::has(ctxData.ids_view(), item.id));
101
102 // There's a limit to the amount of query items which we can store
103 if (ctxData.idsCnt >= MAX_ITEMS_IN_QUERY) {
104 GAIA_ASSERT2(false, "Trying to create a query with too many components!");
105
106 const auto* name = ctx.cc->get(item.id).name.str();
107 GAIA_LOG_E("Trying to add component '%s' to an already full ECS query!", name);
108 return;
109 }
110#endif
111
112 // Build the read-write mask.
113 // This will be used to determine what kind of access the user wants for a given component.
114 const uint16_t isReadWrite = uint16_t(item.access == QueryAccess::Write);
115 ctxData.readWriteMask |= (isReadWrite << ctxData.idsCnt);
116
117 // The query engine is going to reorder the query items as necessary.
118 // Remapping is used so the user can still identify the items according the order in which
119 // they defined them when building the query.
120 ctxData._remapping[ctxData.idsCnt] = ctxData.idsCnt;
121
122 ctxData._ids[ctxData.idsCnt] = item.id;
123 ctxData._terms[ctxData.idsCnt] = {item.id, item.entSrc, item.entTrav, item.travKind,
124 item.travDepth, item.matchKind, nullptr, item.op};
125 ++ctxData.idsCnt;
126 }
127 };

725 QueryInfo& fetch() {
726 if constexpr (UseCaching) {
727 GAIA_PROF_SCOPE(query::fetch);
728
729 // Make sure the query was created by World::query()
730 GAIA_ASSERT(m_storage.is_initialized());
731
732 // If queryId is set it means QueryInfo was already created.
733 // This is the common case for cached queries.
734 if GAIA_LIKELY (m_storage.m_q.handle.id() != QueryIdBad) {
735 auto* pQueryInfo = m_storage.m_queryCache->try_get(m_storage.m_q.handle);
736
737 // The only time when this can be nullptr is just once after Query::destroy is called.
738 if GAIA_LIKELY (pQueryInfo != nullptr) {
739 recommit(pQueryInfo->ctx());
740 return *pQueryInfo;
741 }
742
743 m_storage.invalidate();
744 }
745
746 // No queryId is set which means QueryInfo needs to be created
747 QueryCtx ctx;
748 ctx.init(m_storage.world());
749 commit(ctx);
750 auto& queryInfo =
751 m_storage.m_queryCache->add(GAIA_MOV(ctx), *m_entityToArchetypeMap, all_archetypes_view());
752 m_storage.m_q.handle = QueryInfo::handle(queryInfo);
753 m_storage.allow_to_destroy_again();
754 return queryInfo;
755 } else {
756 GAIA_PROF_SCOPE(query::fetchu);
757
758 if GAIA_UNLIKELY (m_storage.m_queryInfo.ctx().q.handle.id() == QueryIdBad) {
759 QueryCtx ctx;
760 ctx.init(m_storage.world());
761 commit(ctx);
762 m_storage.m_queryInfo =
763 QueryInfo::create(QueryId{}, GAIA_MOV(ctx), *m_entityToArchetypeMap, all_archetypes_view());
764 } else {
765 recommit(m_storage.m_queryInfo.ctx());
766 }
767 return m_storage.m_queryInfo;
768 }

1304 void commit(QueryCtx& ctx) {
1305 GAIA_PROF_SCOPE(query::commit);
1306
1307#if GAIA_ASSERT_ENABLED
1308 if constexpr (UseCaching) {
1309 GAIA_ASSERT(m_storage.m_q.handle.id() == QueryIdBad);
1310 } else {
1311 GAIA_ASSERT(m_storage.m_queryInfo.idx == QueryIdBad);
1312 }
1313#endif
1314
1315 auto& serBuffer = m_storage.ser_buffer();
1316
1317 // Read data from buffer and execute the command stored in it
1318 serBuffer.seek(0);
1319 while (serBuffer.tell() < serBuffer.bytes()) {
1320 QueryCmdType id{};
1321 bool invalidatesHash = false;
1322 ser::load(serBuffer, id);
1323 ser::load(serBuffer, invalidatesHash);
1324 (void)invalidatesHash; // We don't care about this during commit
1325 CommandBufferRead[id](serBuffer, ctx);
1326 }
1327
1328 // Calculate the lookup hash from the provided context
1329 if constexpr (UseCaching) {
1330 ctx.data.cacheSrcTrav = m_cacheSrcTrav;
1331 normalize_cache_src_trav(ctx);
1332 auto& ctxData = ctx.data;
1333 if (ctxData.changedCnt > 1) {
1334 core::sort(ctxData._changed.data(), ctxData._changed.data() + ctxData.changedCnt, SortComponentCond{});
1335 }
1336 calc_lookup_hash(ctx);
1337 }
1338
1339 // We can free all temporary data now
1340 m_storage.ser_buffer_reset();
1341
1342 // Refresh the context
1343 ctx.refresh();
1344 }
1345
1346 void recommit(QueryCtx& ctx) {
1347 GAIA_PROF_SCOPE(query::recommit);
1348
1349 auto& serBuffer = m_storage.ser_buffer();
1350
1351 // Read data from buffer and execute the command stored in it
1352 serBuffer.seek(0);
1353 while (serBuffer.tell() < serBuffer.bytes()) {
1354 QueryCmdType id{};
1355 bool invalidatesHash = false;
1356 ser::load(serBuffer, id);
1357 ser::load(serBuffer, invalidatesHash);
1358 // Hash recalculation is not accepted here
1359 GAIA_ASSERT(!invalidatesHash);
1360 if (invalidatesHash)
1361 return;
1362 CommandBufferRead[id](serBuffer, ctx);
1363 }
1364 if constexpr (UseCaching)
1365 ctx.data.cacheSrcTrav = m_cacheSrcTrav;
1366 if constexpr (UseCaching)
1367 normalize_cache_src_trav(ctx);
1368
1369 // We can free all temporary data now
1370 m_storage.ser_buffer_reset();
1371 }

1389 GAIA_NODISCARD static bool
1390 match_filters(const Chunk& chunk, const QueryInfo& queryInfo, uint32_t changedWorldVersion) {
1391 GAIA_ASSERT(!chunk.empty() && "match_filters called on an empty chunk");
1392
1393 const auto queryVersion = changedWorldVersion;
1394 const auto& filtered = queryInfo.ctx().data.changed_view();
1395
1396 // Skip unchanged chunks
1397 if (filtered.empty())
1398 return false;
1399
1400 const auto filteredCnt = (uint32_t)filtered.size();
1401 auto ids = chunk.ids_view();
1402
1403 // This is the hot path for most change-filter queries.
1404 if (filteredCnt == 1) {
1405 const auto compIdx = core::get_index(ids, filtered[0]);
1406 if (compIdx != BadIndex && chunk.changed(queryVersion, compIdx))
1407 return true;
1408
1409 return chunk.changed(changedWorldVersion);
1410 }
1411
1412 // See if any component has changed
1413 uint32_t lastIdx = 0;
1414 for (const auto comp: filtered) {
1415 uint32_t compIdx = BadIndex;
1416 if (lastIdx < (uint32_t)ids.size()) {
1417 const auto suffixIdx =
1418 core::get_index(std::span<const Entity>(ids.data() + lastIdx, ids.size() - lastIdx), comp);
1419 if (suffixIdx != BadIndex)
1420 compIdx = lastIdx + suffixIdx;
1421 }
1422
1423 // Fallback for queries where change-filters are not monotonic in chunk column order
1424 // (e.g. OR-driven layouts).
1425 if (compIdx == BadIndex)
1426 compIdx = core::get_index(ids, comp);
1427 if (compIdx == BadIndex)
1428 continue;
1429
1430 if (chunk.changed(queryVersion, compIdx))
1431 return true;
1432
1433 lastIdx = compIdx;
1434 }
1435
1436 // If the component hasn't been modified, the entity itself still might have been moved.
1437 // For that reason we also need to check the entity version.
1438 return chunk.changed(changedWorldVersion);
1439 }

1482 static void run_query_func(World* pWorld, Func func, std::span<ChunkBatch> batches) {
1483 GAIA_PROF_SCOPE(query::run_query_func);
1484
1485 const auto chunkCnt = batches.size();
1486 GAIA_ASSERT(chunkCnt > 0);
1487
1488 // We only have one chunk to process
1489 if GAIA_UNLIKELY (chunkCnt == 1) {
1490 run_query_func<Func, TIter>(pWorld, func, batches[0]);
1491 return;
1492 }
1493
1494 // We have many chunks to process.
1495 // Chunks might be located at different memory locations. Not even in the same memory page.
1496 // Therefore, to make it easier for the CPU we give it a hint that we want to prefetch data
1497 // for the next chunk explicitly so we do not end up stalling later.
1498 // Note, this is a micro optimization and on average it brings no performance benefit. It only
1499 // helps with edge cases.
1500 // Let us be conservative for now and go with T2. That means we will try to keep our data at
1501 // least in L3 cache or higher.
1502 gaia::prefetch(batches[1].pChunk, PrefetchHint::PREFETCH_HINT_T2);
1503 run_query_func<Func, TIter>(pWorld, func, batches[0]);
1504
1505 uint32_t chunkIdx = 1;
1506 for (; chunkIdx < chunkCnt - 1; ++chunkIdx) {
1507 gaia::prefetch(batches[chunkIdx + 1].pChunk, PrefetchHint::PREFETCH_HINT_T2);
1508 run_query_func<Func, TIter>(pWorld, func, batches[chunkIdx]);
1509 }
1510
1511 run_query_func<Func, TIter>(pWorld, func, batches[chunkIdx]);
1512 }

1519 GAIA_PROF_SCOPE(query::run_query_batch_no_group_id);
1520
1521 // We are batching by chunks. Some of them might contain only few items but this state is only
1522 // temporary because defragmentation runs constantly and keeps things clean.
1523 ChunkBatchArray chunkBatches;
1524
1525 auto cacheView = queryInfo.cache_archetype_view();
1526 auto sortView = queryInfo.cache_sort_view();
1527
1528 lock(*m_storage.world());
1529
1530 if (!sortView.empty()) {
1531 for (const auto& view: sortView) {
1532 const auto chunkEntitiesCnt = TIter::size(view.pChunk);
1533 if GAIA_UNLIKELY (chunkEntitiesCnt == 0)
1534 continue;
1535
1536 const auto viewFrom = view.startRow;
1537 const auto viewTo = (uint16_t)(view.startRow + view.count);
1538
1539 const auto minStartRow = TIter::start_index(view.pChunk);
1540 const auto minEndRow = TIter::end_index(view.pChunk);
1541 const auto startRow = core::get_max(minStartRow, viewFrom);
1542 const auto endRow = core::get_min(minEndRow, viewTo);

1545 continue;

1547 if constexpr (HasFilters) {
1548 if (!match_filters(*view.pChunk, queryInfo, m_changedWorldVersion))
1549 continue;
1550 }

1553 auto indicesView = queryInfo.indices_mapping_view(view.archetypeIdx);

1556

1559 chunkBatches.clear();
1560 }
1561 }
1562 } else {
1563 for (uint32_t i = idxFrom; i < idxTo; ++i) {
1564 auto* pArchetype = const_cast<Archetype*>(cacheView[i]);

1566 continue;
1567
1568 auto indicesView = queryInfo.indices_mapping_view(i);
1569 const auto& chunks = pArchetype->chunks();
1570 uint32_t chunkOffset = 0;
1571 uint32_t itemsLeft = chunks.size();
1572 while (itemsLeft > 0) {
1573 const auto maxBatchSize = chunkBatches.max_size() - chunkBatches.size();
1574 const auto batchSize = itemsLeft > maxBatchSize ? maxBatchSize : itemsLeft;

1577 for (auto* pChunk: chunkSpan) {
1578 if GAIA_UNLIKELY (TIter::size(pChunk) == 0)
1579 continue;
1580
1581 if constexpr (HasFilters) {
1582 if (!match_filters(*pChunk, queryInfo, m_changedWorldVersion))
1583 continue;
1584 }
1585
1586 chunkBatches.push_back({pArchetype, pChunk, indicesView.data(), 0, 0, 0});
1587 }

1590 run_query_func<Func, TIter>(m_storage.world(), func, {chunkBatches.data(), chunkBatches.size()});
1591 chunkBatches.clear();
1592 }
1593
1594 itemsLeft -= batchSize;
1595 chunkOffset += batchSize;
1596 }
1597 }
1598 }
1599

1601 if (!chunkBatches.empty())
1602 run_query_func<Func, TIter>(m_storage.world(), func, {chunkBatches.data(), chunkBatches.size()});

1606 // TODO: Smart handling necessary
1607 commit_cmd_buffer_st(*m_storage.world());
1608 commit_cmd_buffer_mt(*m_storage.world());

1937 template <typename TIter, typename Func, typename... T>
1938 GAIA_FORCEINLINE void run_query_on_chunk(
1939 const QueryInfo& queryInfo, TIter& it, Func func, [[maybe_unused]] core::func_type_list<T...> types) {
1940 const auto cnt = it.size();
1941 const bool hasEntityFilters = queryInfo.has_entity_filter_terms();
1942
1943 if constexpr (sizeof...(T) > 0) {
1944 // Pointers to the respective component types in the chunk, e.g
1945 // q.each([&](Position& p, const Velocity& v) {...}
1946 // Translates to:
1947 // auto p = it.view_mut_inter<Position, true>();
1948 // auto v = it.view_inter<Velocity>();
1949 auto dataPointerTuple = std::make_tuple(it.template view_auto<T>()...);
1950
1951 // Iterate over each entity in the chunk.
1952 // Translates to:
1953 // GAIA_FOR(0, cnt) func(p[i], v[i]);
1954
1955 if (!hasEntityFilters) {
1956 GAIA_FOR(cnt) {
1957 func(std::get<decltype(it.template view_auto<T>())>(dataPointerTuple)[it.template acc_index<T>(i)]...);
1958 }
1959 } else {
1960 const auto entities = it.template view<Entity>();
1961 GAIA_FOR(cnt) {
1962 if (!match_entity_filters(*queryInfo.world(), entities[i], queryInfo))
1963 continue;
1964 func(std::get<decltype(it.template view_auto<T>())>(dataPointerTuple)[it.template acc_index<T>(i)]...);
1965 }
1966 }
1967 } else {
1968 // No functor parameters. Do an empty loop.
1969 if (!hasEntityFilters) {
1970 GAIA_FOR(cnt) {
1971 func();
1972 }
1973 } else {
1974 const auto entities = it.template view<Entity>();
1975 GAIA_FOR(cnt) {
1976 if (!match_entity_filters(*queryInfo.world(), entities[i], queryInfo))
1977 continue;
1978 func();
1979 }
1980 }
1981 }

3897 if (expr.empty())

3900 // OR-chain at top level maps to query::or_ terms.

3905 for (uint32_t i = 0; i + 1 < cnt; ++i) {
3906 const auto ch = expr[i];

3908 ++depth;
3909 else if (ch == ')') {

3913 hasOrChain = true;

3926 ++depth;

3933 const bool isEnd = i + 1 == cnt;

3935 continue;

3937 const auto partEnd = isOr ? i : (i + 1);
3938 auto partExpr = expr.subspan(partBeg, partEnd - partBeg);
3939 if (!add_term(QueryOpKind::Or, partExpr))
3940 return false;
3941
3942 if (isOr) {
3943 partBeg = i + 2;
3944 ++i;
3945 }
3946 }
