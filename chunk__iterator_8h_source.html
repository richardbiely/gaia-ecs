853 template <typename T>
854 GAIA_NODISCARD bool uses_out_of_line_component() const {
855 using Arg = std::remove_cv_t<std::remove_reference_t<T>>;
856 if constexpr (std::is_same_v<Arg, Entity>)
857 return false;
858 else {
859 using FT = typename component_type_t<Arg>::TypeFull;
860 if constexpr (is_pair<FT>::value || mem::is_soa_layout_v<Arg>)
861 return false;
862 else {
863 const auto* pItem = comp_cache(*world()).template find<FT>();
864 // Type-based iteration switches to the world/store path as soon as the
865 // payload stops being chunk-backed, even if the id still fragments.
866 return pItem != nullptr && world_is_out_of_line_component(*world(), pItem->entity);
867 }
868 }
