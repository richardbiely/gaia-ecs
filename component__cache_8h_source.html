76 template <typename T>
77 GAIA_NODISCARD GAIA_FORCEINLINE const ComponentCacheItem& add(Entity entity) {
78 static_assert(!is_pair<T>::value);
79 GAIA_ASSERT(!entity.pair());
80
81 const auto compDescId = detail::ComponentDesc<T>::id();
82
83 // Fast path for small component ids - use the array storage
84 if (compDescId < FastComponentCacheSize) {
85 auto createDesc = [&]() -> const ComponentCacheItem& {
86 const auto* pItem = ComponentCacheItem::create<T>(entity);
87 GAIA_ASSERT(compDescId == pItem->comp.id());
88 m_itemArr[compDescId] = pItem;
89 m_compByString.emplace(pItem->name, pItem);
90 m_compByEntity.emplace(pItem->entity, pItem);
91 return *pItem;
92 };
93
94 if GAIA_UNLIKELY (compDescId >= m_itemArr.size()) {
95 const auto oldSize = m_itemArr.size();
96 const auto newSize = compDescId + 1U;
97
98 // Increase the capacity by multiples of CapacityIncreaseSize
99 constexpr uint32_t CapacityIncreaseSize = 128;
100 const auto newCapacity = ((newSize / CapacityIncreaseSize) * CapacityIncreaseSize) + CapacityIncreaseSize;
101 m_itemArr.reserve(newCapacity);
102
103 // Update the size
104 m_itemArr.resize(newSize);
105
106 // Make sure unused memory is initialized to nullptr.
107 GAIA_FOR2(oldSize, newSize - 1) m_itemArr[i] = nullptr;
108
109 return createDesc();
110 }
111
112 if GAIA_UNLIKELY (m_itemArr[compDescId] == nullptr) {
113 return createDesc();
114 }
115
116 return *m_itemArr[compDescId];
117 }
118
119 // Generic path for large component ids - use the map storage
120 {
121 auto createDesc = [&]() -> const ComponentCacheItem& {
122 const auto* pItem = ComponentCacheItem::create<T>(entity);
123 GAIA_ASSERT(compDescId == pItem->comp.id());
124 m_itemByDescId.emplace(compDescId, pItem);
125 m_compByString.emplace(pItem->name, pItem);
126 m_compByEntity.emplace(pItem->entity, pItem);
127 return *pItem;
128 };
129
130 const auto it = m_itemByDescId.find(compDescId);
131 if (it == m_itemByDescId.end())
132 return createDesc();
133
134 return *it->second;
135 }
136 }

148 GAIA_NODISCARD const ComponentCacheItem&
149 add(Entity entity, const char* name, uint32_t len, uint32_t size, uint32_t alig = 1, uint32_t soa = 0,
150 const uint8_t* pSoaSizes = nullptr, ComponentLookupHash hashLookup = {}) {
151 GAIA_ASSERT(!entity.pair());
152 GAIA_ASSERT(name != nullptr);
153
154 const auto l = len == 0 ? (uint32_t)GAIA_STRLEN(name, ComponentCacheItem::MaxNameLength) : len;
155 GAIA_ASSERT(l > 0 && l < ComponentCacheItem::MaxNameLength);
156
157 {
158 const auto* pExisting = find(name, l);
159 if (pExisting != nullptr)
160 return *pExisting;
161 }
162
163 detail::ComponentDescId compDescId = m_nextRuntimeCompDescId;
164 while (find(compDescId) != nullptr) {
165 ++compDescId;
166 }
167 m_nextRuntimeCompDescId = compDescId + 1;
168
169 const auto* pItem =
170 ComponentCacheItem::create(entity, compDescId, name, l, size, alig, soa, pSoaSizes, hashLookup);
171 if (compDescId < FastComponentCacheSize) {
172 if (compDescId >= m_itemArr.size())
173 m_itemArr.resize(compDescId + 1U);
174 m_itemArr[compDescId] = pItem;
175 } else {
176 m_itemByDescId.emplace(compDescId, pItem);
177 }
178
179 m_compByString.emplace(pItem->name, pItem);
180 m_compByEntity.emplace(pItem->entity, pItem);
181 return *pItem;
182 }

187 GAIA_NODISCARD const ComponentCacheItem* find(detail::ComponentDescId compDescId) const noexcept {
188 // Fast path - array storage
189 if (compDescId < FastComponentCacheSize) {
190 if (compDescId >= m_itemArr.size())
191 return nullptr;
192
193 return m_itemArr[compDescId];
194 }
195
196 // Generic path - map storage
197 const auto it = m_itemByDescId.find(compDescId);
198 return it != m_itemByDescId.end() ? it->second : nullptr;
