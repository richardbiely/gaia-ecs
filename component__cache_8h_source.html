96 static ComponentCacheItem::SymbolLookupKey short_name_key(const ComponentCacheItem& item) noexcept {
97 const auto* pName = item.name.str();
98 const auto len = item.name.len();
99 for (uint32_t i = len; i > 1; --i) {
100 const auto idx = i - 1;
101 if (pName[idx] != ':' || pName[idx - 1] != ':')
102 continue;
103 return ComponentCacheItem::SymbolLookupKey(pName + idx + 1, len - idx - 1, 0);
104 }
105
106 return ComponentCacheItem::SymbolLookupKey();
107 }

114 static bool build_default_path(util::str& out, util::str_view symbol) {
115 out.clear();
116 bool changed = false;
117 out.reserve(symbol.size());
118 for (uint32_t i = 0; i < symbol.size(); ++i) {
119 if (i + 1 < symbol.size() && symbol.data()[i] == ':' && symbol.data()[i + 1] == ':') {
120 out.append('.');
121 ++i;
122 changed = true;
123 continue;
124 }
125
126 out.append(symbol.data()[i]);
127 }
128
129 if (!changed)
130 out.clear();
131
132 return changed;
133 }

140 static bool build_scoped_path(util::str& out, util::str_view scopePath, util::str_view leaf) {
141 out.clear();
142 if (scopePath.empty() || leaf.empty())
143 return false;
144
145 out.reserve(scopePath.size() + 1 + leaf.size());
146 out.append(scopePath);
147 out.append('.');
148 out.append(leaf);
149 return true;
150 }

155 static void initialize_names(ComponentCacheItem& item, util::str_view scopePath = {}) {
156 item.path.clear();
157 const auto symbol = util::str_view{item.name.str(), item.name.len()};
158 if (is_internal_symbol(symbol))
159 return;
160
161 const auto shortName = short_name_key(item);
162 const bool hasShortName =
163 shortName.str() != nullptr && shortName.len() != 0 && shortName.len() != item.name.len();
164 const auto leaf = hasShortName ? util::str_view{shortName.str(), shortName.len()} : symbol;
165
166 if (!build_scoped_path(item.path, scopePath, leaf))
167 (void)build_default_path(item.path, symbol);
168 }

170 template <typename ViewFunc>
171 void rebuild_lookup_map(
172 cnt::map<ComponentCacheItem::SymbolLookupKey, const ComponentCacheItem*>& map, ViewFunc&& getView) {
173 map.clear();
174 for_each_item([&](const ComponentCacheItem& item) {
175 const auto view = getView(item);
176 if (view.empty())
177 return;
178
179 const auto key = lookup_key(view);
180 const auto it = map.find(key);
181 if (it == map.end()) {
182 map.emplace(key, &item);
183 return;
184 }
185
186 if (it->second != &item)
187 it->second = nullptr;
188 });
189 }

223 template <typename T>
224 GAIA_NODISCARD GAIA_FORCEINLINE const ComponentCacheItem& add(Entity entity, util::str_view scopePath = {}) {
225 static_assert(!is_pair<T>::value);
226 GAIA_ASSERT(!entity.pair());
227
228 const auto compDescId = detail::ComponentDesc<T>::id();
229
230 // Fast path for small component ids - use the array storage
231 if (compDescId < FastComponentCacheSize) {
232 auto createDesc = [&]() -> const ComponentCacheItem& {
233 const auto* pItem = ComponentCacheItem::create<T>(entity);
234 GAIA_ASSERT(compDescId == pItem->comp.id());
235 m_itemArr[compDescId] = pItem;
236 add_name_mappings(*const_cast<ComponentCacheItem*>(pItem), scopePath);
237 m_compByEntity.emplace(pItem->entity, pItem);
238 return *pItem;
239 };
240
241 if GAIA_UNLIKELY (compDescId >= m_itemArr.size()) {
242 const auto newSize = compDescId + 1U;
243
244 // Increase the capacity by multiples of CapacityIncreaseSize
245 constexpr uint32_t CapacityIncreaseSize = 128;
246 const auto newCapacity = ((newSize / CapacityIncreaseSize) * CapacityIncreaseSize) + CapacityIncreaseSize;
247 m_itemArr.reserve(newCapacity);
248
249 // Update the size
250 m_itemArr.resize(newSize, nullptr);
251
252 return createDesc();
253 }
254
255 if GAIA_UNLIKELY (m_itemArr[compDescId] == nullptr) {
256 return createDesc();
257 }
258
259 return *m_itemArr[compDescId];
260 }
261
262 // Generic path for large component ids - use the map storage
263 {
264 auto createDesc = [&]() -> const ComponentCacheItem& {
265 const auto* pItem = ComponentCacheItem::create<T>(entity);
266 GAIA_ASSERT(compDescId == pItem->comp.id());
267 m_itemByDescId.emplace(compDescId, pItem);
268 add_name_mappings(*const_cast<ComponentCacheItem*>(pItem), scopePath);
269 m_compByEntity.emplace(pItem->entity, pItem);
270 return *pItem;
271 };
272
273 const auto it = m_itemByDescId.find(compDescId);
274 if (it == m_itemByDescId.end())
275 return createDesc();
276
277 return *it->second;
278 }
279 }

293 GAIA_NODISCARD const ComponentCacheItem&
294 add(Entity entity, const char* name, uint32_t len, uint32_t size, DataStorageType storageType, uint32_t alig = 1,
295 uint32_t soa = 0, const uint8_t* pSoaSizes = nullptr, ComponentLookupHash hashLookup = {},
296 util::str_view scopePath = {}) {
297 GAIA_ASSERT(!entity.pair());
298 GAIA_ASSERT(name != nullptr);
299
300 const auto l = len == 0 ? (uint32_t)GAIA_STRLEN(name, ComponentCacheItem::MaxNameLength) : len;
301 GAIA_ASSERT(l > 0 && l < ComponentCacheItem::MaxNameLength);
302
303 {
304 const auto* pExisting = symbol(name, l);
305 if (pExisting != nullptr)
306 return *pExisting;
307 }
308
309 detail::ComponentDescId compDescId = m_nextRuntimeCompDescId;
310 while (find(compDescId) != nullptr) {
311 ++compDescId;
312 }
313 m_nextRuntimeCompDescId = compDescId + 1;
314
315 const auto* pItem = ComponentCacheItem::create(
316 entity, compDescId, name, l, size, alig, storageType, soa, pSoaSizes, hashLookup);
317 if (compDescId < FastComponentCacheSize) {
318 if (compDescId >= m_itemArr.size())
319 m_itemArr.resize(compDescId + 1U);
320 m_itemArr[compDescId] = pItem;
321 } else {
322 m_itemByDescId.emplace(compDescId, pItem);
323 }
324
325 add_name_mappings(*const_cast<ComponentCacheItem*>(pItem), scopePath);
326 m_compByEntity.emplace(pItem->entity, pItem);
327 return *pItem;
328 }

427 const auto symbol = symbol_name(item);

526 void resolve(cnt::darray<const ComponentCacheItem*>& out, const char* name, uint32_t len = 0) const {
527 GAIA_ASSERT(name != nullptr);
528 out.clear();
529
530 const auto l = len == 0 ? (uint32_t)GAIA_STRLEN(name, ComponentCacheItem::MaxNameLength) : len;
531 GAIA_ASSERT(l < ComponentCacheItem::MaxNameLength);
532 const auto needle = util::str_view(name, l);
533
534 auto push_unique = [&](const ComponentCacheItem* pItem) {
535 if (pItem == nullptr)
536 return;
537 for (const auto* pExisting: out) {
538 if (pExisting == pItem)
539 return;
540 }
541 out.push_back(pItem);
542 };
543
544 push_unique(symbol(name, l));
545
546 if (const auto* pItem = path(name, l); pItem != nullptr) {
547 push_unique(pItem);
548 } else {
549 for_each_item([&](const ComponentCacheItem& item) {
550 if (item.path.view() == needle)
551 push_unique(&item);
552 });
553 }
554 }
