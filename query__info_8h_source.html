423 const auto maxItems = (uint32_t)m_plan.ctx.data.cacheSrcTrav;

511 if (!can_reuse_dyn_cache())
512 return;
513
514 const auto relations = m_plan.ctx.data.deps.relations_view();

538 m_state.srcTravSnapshotOverflowed = false;

542 m_state.varBindingMask = runtimeVarBindingMask;

545 template <typename TType>
546 GAIA_NODISCARD bool has_inter([[maybe_unused]] QueryOpKind op, bool isReadWrite) const {
547 using T = core::raw_t<TType>;
548
549 if constexpr (std::is_same_v<T, Entity>) {
550 // Entities are read-only.
551 GAIA_ASSERT(!isReadWrite);
552 // Skip Entity input args. Entities are always there.
553 return true;
554 } else {
555 Entity id;
556
557 if constexpr (is_pair<T>::value) {
558 const auto rel = m_plan.ctx.cc->get<typename T::rel>().entity;
559 const auto tgt = m_plan.ctx.cc->get<typename T::tgt>().entity;
560 id = (Entity)Pair(rel, tgt);
561 } else {
562 id = m_plan.ctx.cc->get<T>().entity;
563 }
564
565 const auto& ctxData = m_plan.ctx.data;
566 const auto compIdx = comp_idx<MAX_ITEMS_IN_QUERY>(ctxData.terms.data(), id, EntityBad);
567
568 if (op != ctxData.terms[compIdx].op)
569 return false;
570
571 // Read-write mask must match
572 const uint32_t maskRW = (uint32_t)ctxData.readWriteMask & (1U << compIdx);
573 const uint32_t maskXX = (uint32_t)isReadWrite << compIdx;
574 return maskRW == maskXX;
575 }
576 }

582 return has_inter<T>(op, isReadWrite);

750 template <typename ArchetypeLookup>
751 void match(
752 // entity -> archetypes mapping
753 const ArchetypeLookup& entityToArchetypeMap,
754 // all archetypes in the world
755 std::span<const Archetype*> allArchetypes,
756 // last matched archetype id
757 ArchetypeId archetypeLastId, const cnt::sarray<Entity, MaxVarCnt>& runtimeVarBindings,
758 uint8_t runtimeVarBindingMask) {
759 auto& ctxData = m_plan.ctx.data;
760
761 // Recompile if necessary
762 if ((ctxData.flags & QueryCtx::QueryFlags::Recompile) != 0)
763 recompile();
764
765 // Skip if nothing has been compiled.
766 if (!m_plan.vm.is_compiled())
767 return;
768
769 const bool hasDynamicTerms = has_dyn_terms();
770 if (hasDynamicTerms && (!can_reuse_dyn_cache() || m_state.needs_refresh() ||
771 dyn_inputs_changed(runtimeVarBindings, runtimeVarBindingMask))) {
772 // Dynamic queries keep their cached result as long as tracked runtime inputs stay stable.
773 // Source-based queries still take the conservative rebuild path until their inputs
774 // are tracked with finer-grained version metadata.
775 reset_matching_cache();
776 } else if (m_state.seed_dirty()) {
777 reset_matching_cache();
778 } else if (m_state.result_dirty()) {
779 sync_result_cache_from_seed_cache();
780 if (m_state.lastArchetypeId == archetypeLastId) {
781 sort_entities();
782 sort_cache_groups();
783 m_state.clear_dirty();
784 return;
785 }
786 }
787
788 // Skip if no new archetype appeared
789 GAIA_ASSERT(archetypeLastId >= m_state.lastArchetypeId);
790 if (!hasDynamicTerms && !m_state.needs_refresh() && m_state.lastArchetypeId == archetypeLastId) {
791 // Sort entities if necessary
792 sort_entities();
793 return;
794 }
795
796 m_state.lastArchetypeId = archetypeLastId;
797
798 GAIA_PROF_SCOPE(queryinfo::match);
799
800 auto& w = *world();

1139 void calculate_sort_data() {
1140 GAIA_PROF_SCOPE(queryinfo::calc_sort_data);
1141
1142 m_state.archetypeSortData.clear();
1143
1144 // The function doesn't do any moves and expects that all chunks have their data sorted already.
1145 // We use a min-heap / priority queue - like structure during query iteration to merge sorted tables:
1146 // - we hold a cursor into each sorted chunk
1147 // - we compare the next entity from each chunk using your sorting function
1148 // - we then pick the smallest one (like k-way merge sort), and advance that cursor
1149 // This is esentially what this function does:
1150 // while (any_chunk_has_entities) {
1151 // find_chunk_with_smallest_next_element();
1152 // yield(entity_from_that_chunk);
1153 // adv_cursor_for_that_chunk();
1154 // }
1155 // This produces a globally sorted view without modifying actual data. It's a balance between
1156 // performance and memory usage. We could also sort the data in-place across all chunks, but that
1157 // would generated too many data moves (entities + all of their components).
1158
1159 struct Cursor {
1160 uint32_t chunkIdx = 0;
1161 uint16_t row = 0;
1162 };
1163
1164 auto& archetypes = m_state.archetypeCache;
1165
1166 // Initialize cursors. We will need as many as there are archetypes.
1167 cnt::sarray_ext<Cursor, 128> cursors(archetypes.size());
1168
1169 uint32_t currArchetypeIdx = (uint32_t)-1;
1170 Chunk* pCurrentChunk = nullptr;
1171 uint16_t currentStartRow = 0;
1172 uint16_t currentRow = 0;
1173
1174 const void* pDataMin = nullptr;
1175 const void* pDataCurr = nullptr;
1176
1177 while (true) {
1178 uint32_t minArchetypeIdx = (uint32_t)-1;
1179 Entity minEntity = EntityBad;
1180
1181 // Find the next entity across all tables/chunks
1182 for (uint32_t t = 0; t < archetypes.size(); ++t) {
1183 const auto* pArchetype = archetypes[t];
1184 const auto& chunks = pArchetype->chunks();
1185 auto& cur = cursors[t];
1186
1187 while (cur.chunkIdx < chunks.size() && cur.row >= chunks[cur.chunkIdx]->size()) {
1188 ++cur.chunkIdx;
1189 cur.row = 0;
1190 }
1191
1192 if (cur.chunkIdx >= chunks.size())
1193 continue;
1194
1195 const auto* pChunk = pArchetype->chunks()[cur.chunkIdx];
1196 auto entity = pChunk->entity_view()[cur.row];
1197
1198 if (m_plan.ctx.data.sortBy != ecs::EntityBad) {
1199 auto compIdx = world_component_index_comp_idx(*m_plan.ctx.w, *pArchetype, m_plan.ctx.data.sortBy);
1200 if (compIdx == BadIndex)
1201 compIdx = pChunk->comp_idx(m_plan.ctx.data.sortBy);
1202 pDataCurr = pChunk->comp_ptr(compIdx, cur.row);
1203 } else
1204 pDataCurr = &pChunk->entity_view()[cur.row];
1205
1206 if (minEntity == EntityBad) {
1207 minEntity = entity;
1208 minArchetypeIdx = t;
1209 pDataMin = pDataCurr;
1210 continue;
1211 }
1212
1213 if (m_plan.ctx.data.sortByFunc(*m_plan.ctx.w, pDataCurr, pDataMin) < 0) {
1214 minEntity = entity;
1215 minArchetypeIdx = t;
1216 }
1217 }
1218
1219 // No more results found, we can stop
1220 if (minArchetypeIdx == (uint32_t)-1)
1221 break;
1222
1223 auto& cur = cursors[minArchetypeIdx];
1224 const auto& chunks = archetypes[minArchetypeIdx]->chunks();
1225 Chunk* pChunk = chunks[cur.chunkIdx];
1226
1227 if (minArchetypeIdx == currArchetypeIdx && pChunk == pCurrentChunk) {
1228 // Current slice
1229 } else {
1230 // End previous slice
1231 if (pCurrentChunk != nullptr) {
1232 m_state.archetypeSortData.push_back(
1233 {pCurrentChunk, currArchetypeIdx, currentStartRow, (uint16_t)(currentRow - currentStartRow)});
1234 }
1235
1236 // Start a new slice
1237 currArchetypeIdx = minArchetypeIdx;
1238 pCurrentChunk = pChunk;
1239 currentStartRow = cur.row;
1240 }
1241
1242 ++cur.row;
1243 currentRow = cur.row;
1244 }
1245
1246 if (pCurrentChunk != nullptr) {
1247 m_state.archetypeSortData.push_back(
1248 {pCurrentChunk, currArchetypeIdx, currentStartRow, (uint16_t)(currentRow - currentStartRow)});
1249 }
1250 }
